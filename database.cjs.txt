const path = require('path');
const crypto = require('crypto');
const fs = require('fs');
const { DatabaseSync } = require('node:sqlite');

let db;
const queryCache = new Map();
const statementCache = new Map();
const CACHE_TTL_MS = 1500;
function cachedPrepare(sql) { if (!statementCache.has(sql)) statementCache.set(sql, db.prepare(sql)); return statementCache.get(sql); }
function invalidateQueryCache() { queryCache.clear(); }
const ALLOWED_TABLES = new Set(['customers','suppliers','categories','warehouses','products','stock_balances','stock_movements','invoices','invoice_items','payments','quotations','quotation_items','installment_plans','installment_payments','services','expenses','partners','partner_distributions','team_members','financial_vouchers','warehouse_transfers','deleted_records','sync_queue','attendance_devices','attendance_records','employee_signatures','payroll_employees','payroll_adjustments','payroll_runs','payroll_items']);

function openDatabase(userDataPath) {
  const dataDir = path.join(userDataPath, 'data');
  fs.mkdirSync(dataDir, { recursive: true });
  db = new DatabaseSync(path.join(dataDir, 'erp-hub.sqlite'));
  db.exec('PRAGMA foreign_keys = ON; PRAGMA journal_mode = WAL; PRAGMA synchronous = NORMAL; PRAGMA busy_timeout = 5000; PRAGMA temp_store = MEMORY; PRAGMA cache_size = -32000; PRAGMA mmap_size = 134217728; PRAGMA auto_vacuum = INCREMENTAL;');
  migrate();
  migrateLegacyColumns();
  migratePlaintextPasswords();
  return db;
}

function tableColumns(tableName) {
  return new Set(db.prepare(`PRAGMA table_info(${tableName})`).all().map(row => row.name));
}
function ensureColumn(tableName, columnName, definition, backfillSql) {
  const columns = tableColumns(tableName);
  if (columns.has(columnName)) return;
  db.exec(`ALTER TABLE ${tableName} ADD COLUMN ${columnName} ${definition}`);
  if (backfillSql) db.exec(backfillSql);
}
function migrateLegacyColumns() {
  // Older installations may already contain the HR tables with fewer columns.
  // CREATE TABLE IF NOT EXISTS does not upgrade those tables, so every required
  // column is added explicitly before any list/save query is allowed to run.
  const employeeColumns = [
    ['team_member_id','TEXT'],['name',"TEXT NOT NULL DEFAULT 'موظف'"],['job_title','TEXT'],
    ['employee_role','TEXT'],['phone','TEXT'],['base_salary','REAL NOT NULL DEFAULT 0'],
    ['hourly_rate','REAL NOT NULL DEFAULT 0'],['overtime_rate','REAL NOT NULL DEFAULT 0'],
    ['standard_daily_hours','REAL NOT NULL DEFAULT 8'],['standard_monthly_days','REAL NOT NULL DEFAULT 30'],
    ['attendance_employee_id','TEXT'],['active','INTEGER NOT NULL DEFAULT 1'],
    ['created_at','TEXT'],['updated_at','TEXT']
  ];
  for (const [name, definition] of employeeColumns) ensureColumn('payroll_employees', name, definition);
  ensureColumn('payroll_adjustments','created_at','TEXT');
  ensureColumn('payroll_runs','created_at','TEXT');
  ensureColumn('payroll_runs','updated_at','TEXT');
  ensureColumn('payroll_items','created_at','TEXT');
  ensureColumn('payroll_items','updated_at','TEXT');
  ensureColumn('attendance_devices','created_at','TEXT');
  ensureColumn('attendance_records','created_at','TEXT');
  const timestampTables=['payroll_employees','payroll_adjustments','payroll_runs','payroll_items','attendance_devices','attendance_records'];
  for (const tableName of timestampTables) {
    const cols=tableColumns(tableName);
    if(cols.has('created_at')) db.exec(`UPDATE ${tableName} SET created_at=COALESCE(created_at,CURRENT_TIMESTAMP)`);
    if(cols.has('updated_at')) db.exec(`UPDATE ${tableName} SET updated_at=COALESCE(updated_at,created_at,CURRENT_TIMESTAMP)`);
  }
  db.exec('CREATE INDEX IF NOT EXISTS idx_payroll_employee_name ON payroll_employees(name COLLATE NOCASE);');
}


function closeDatabase() { if (db) { db.close(); db = undefined; statementCache.clear(); queryCache.clear(); } }

function table(name) {
  if (!ALLOWED_TABLES.has(name)) throw new Error('Unsupported table');
  return name;
}

function normalizeRow(row) {
  if (!row) return row;
  const out = { ...row };
  if ('payload_json' in out) {
    try { out.payload = JSON.parse(out.payload_json); } catch { out.payload = null; }
    delete out.payload_json;
  }
  return out;
}

function list({ table: tableName, page = 1, pageSize = 50, search = '', orderBy, direction = 'DESC' }) {
  const t = table(tableName);
  const safePage = Math.max(1, Number(page) || 1);
  const safeSize = Math.min(200, Math.max(1, Number(pageSize) || 50));
  const offset = (safePage - 1) * safeSize;
  const dir = String(direction).toUpperCase() === 'ASC' ? 'ASC' : 'DESC';
  const allowedOrder = new Set(['created_at','updated_at','quotation_date','valid_until','due_date','deleted_at','number','reference','name','invoice_date','invoice_number','occurred_at','voucher_date','transfer_date','expense_date']);
  const defaultOrder = t === 'deleted_records' ? 'deleted_at' : (t === 'installment_plans' ? 'due_date' : (t === 'quotations' ? 'quotation_date' : 'created_at'));
  let order = allowedOrder.has(orderBy) ? orderBy : defaultOrder;
  const availableColumns = tableColumns(t);
  if (!availableColumns.has(order)) order = availableColumns.has('created_at') ? 'created_at' : (availableColumns.has('name') ? 'name' : 'rowid');
  let where = '';
  let params = [];
  if (search) {
    const term = `%${String(search).trim()}%`;
    if (t === 'quotations') { where = 'WHERE number LIKE ? OR customer_name LIKE ? OR status LIKE ?'; params = [term, term, term]; }
    else if (t === 'installment_plans') { where = 'WHERE reference LIKE ? OR customer_name LIKE ? OR status LIKE ?'; params = [term, term, term]; }
    else if (t === 'deleted_records') { where = 'WHERE label LIKE ? OR entity_type LIKE ?'; params = [term, term]; }
    else if (t === 'customers' || t === 'suppliers') { where = 'WHERE name LIKE ? OR phone LIKE ? OR email LIKE ?'; params = [term, term, term]; }
    else if (t === 'products') { where = 'WHERE name LIKE ? OR code LIKE ? OR barcode LIKE ?'; params = [term, term, term]; }
    else if (t === 'invoices') { where = 'WHERE invoice_number LIKE ? OR status LIKE ? OR notes LIKE ?'; params = [term, term, term]; }
    else if (t === 'services' || t === 'partners' || t === 'team_members') { where = 'WHERE name LIKE ?'; params = [term]; }
    else if (t === 'financial_vouchers') { where = 'WHERE voucher_number LIKE ? OR party_type LIKE ? OR notes LIKE ?'; params = [term, term, term]; }
  }
  const cacheKey = JSON.stringify([t,safePage,safeSize,search,order,dir]);
  const hit = queryCache.get(cacheKey);
  if (hit && Date.now() - hit.at < CACHE_TTL_MS) return structuredClone(hit.value);
  const countSql = `SELECT COUNT(*) AS count FROM ${t} ${where}`;
  const rowsSql = `SELECT * FROM ${t} ${where} ORDER BY ${order} ${dir} LIMIT ? OFFSET ?`;
  const count = cachedPrepare(countSql).get(...params).count;
  const rows = cachedPrepare(rowsSql).all(...params, safeSize, offset).map(normalizeRow);
  const value = { rows, count, page: safePage, pageSize: safeSize, pages: Math.max(1, Math.ceil(count / safeSize)) };
  queryCache.set(cacheKey, { at: Date.now(), value });
  return structuredClone(value);
}

function executeTransaction(fn) {
  db.exec('BEGIN IMMEDIATE');
  try { const result = fn(); db.exec('COMMIT'); invalidateQueryCache(); return result; }
  catch (e) { db.exec('ROLLBACK'); throw e; }
}

function createQuotation(data) {
  return executeTransaction(() => {
    const items = Array.isArray(data.items) ? data.items : [];
    const computedTotal = items.reduce((sum, item) => sum + Math.max(0, Number(item.quantity)||0) * Math.max(0, Number(item.unitPrice)||0), 0);
    const total = items.length ? computedTotal : Math.max(0, Number(data.total)||0);
    db.prepare(`INSERT INTO quotations(id,number,customer_id,customer_name,quotation_date,valid_until,total,status,notes)
      VALUES(?,?,?,?,?,?,?,?,?)`).run(data.id,data.number,data.customerId||null,data.customerName,data.date,data.validUntil,total,data.status||'DRAFT',data.notes||null);
    const insertItem = db.prepare(`INSERT INTO quotation_items(id,quotation_id,product_id,description,quantity,unit_price,line_total) VALUES(?,?,?,?,?,?,?)`);
    for (const item of items) {
      const quantity = Math.max(0, Number(item.quantity)||0);
      const unitPrice = Math.max(0, Number(item.unitPrice)||0);
      if (!item.description || quantity <= 0) continue;
      insertItem.run(item.id || `qi_${Date.now()}_${Math.random().toString(36).slice(2,8)}`, data.id, item.productId || null, item.description, quantity, unitPrice, quantity * unitPrice);
    }
    return getQuotation(data.id);
  });
}
function updateQuotation(data) {
  return executeTransaction(() => {
    const existing=db.prepare('SELECT id FROM quotations WHERE id=?').get(data.id);
    if(!existing) throw new Error('Quotation not found');
    const items=Array.isArray(data.items)?data.items:[];
    const total=items.reduce((sum,item)=>sum+Math.max(0,Number(item.quantity)||0)*Math.max(0,Number(item.unitPrice)||0),0);
    db.prepare(`UPDATE quotations SET number=?,customer_id=?,customer_name=?,quotation_date=?,valid_until=?,total=?,status=?,notes=?,updated_at=CURRENT_TIMESTAMP WHERE id=?`).run(data.number,data.customerId||null,data.customerName,data.date,data.validUntil,total,data.status||'DRAFT',data.notes||null,data.id);
    db.prepare('DELETE FROM quotation_items WHERE quotation_id=?').run(data.id);
    const stmt=db.prepare(`INSERT INTO quotation_items(id,quotation_id,product_id,description,quantity,unit_price,line_total) VALUES(?,?,?,?,?,?,?)`);
    for(const item of items){const quantity=Math.max(0,Number(item.quantity)||0),unitPrice=Math.max(0,Number(item.unitPrice)||0);if(!item.description||quantity<=0)continue;stmt.run(item.id||`qi_${Date.now()}_${Math.random().toString(36).slice(2,8)}`,data.id,item.productId||null,item.description,quantity,unitPrice,quantity*unitPrice);}
    return getQuotation(data.id);

function createInstallment(data) {
  return executeTransaction(() => {
    const total = Math.max(0, Number(data.total) || 0);
    const paid = Math.min(total, Math.max(0, Number(data.paid) || 0));
    if (!data.id) throw new Error('Installment id is required');
    if (!String(data.customerName || '').trim()) throw new Error('Customer is required');
    if (total <= 0) throw new Error('Installment total must be greater than zero');
    const status = paid >= total ? 'PAID' : paid > 0 ? 'PARTIAL' : 'PENDING';
    db.prepare(`INSERT INTO installment_plans(id,invoice_id,customer_id,customer_name,reference,total,paid,due_date,status)
      VALUES(?,?,?,?,?,?,?,?,?)`).run(data.id,data.invoiceId||null,data.customerId||null,String(data.customerName).trim(),String(data.reference||data.id),total,paid,data.dueDate||new Date().toISOString().slice(0,10),status);
    if (paid > 0) db.prepare('INSERT INTO installment_payments(id,installment_id,amount,paid_at,payment_method,notes) VALUES(?,?,?,?,?,?)')
      .run(`pay_${Date.now()}_${Math.random().toString(36).slice(2,7)}`,data.id,paid,new Date().toISOString(),'OPENING','Initial paid amount');
    return db.prepare('SELECT * FROM installment_plans WHERE id=?').get(data.id);
  });
}
function collectInstallment({ id, amount, paymentMethod = 'CASH' }) {
  return executeTransaction(() => {
    const row = db.prepare('SELECT * FROM installment_plans WHERE id=?').get(id);
    if (!row) throw new Error('Installment not found');
    const applied = Math.min(Number(amount)||0, row.total-row.paid);
    if (applied <= 0) return row;
    db.prepare('INSERT INTO installment_payments(id,installment_id,amount,paid_at,payment_method) VALUES(?,?,?,?,?)')
      .run(`pay_${Date.now()}_${Math.random().toString(36).slice(2,7)}`, id, applied, new Date().toISOString(), paymentMethod);
    const paid = row.paid + applied;
    const status = paid >= row.total ? 'PAID' : 'PARTIAL';
    db.prepare('UPDATE installment_plans SET paid=?,status=?,updated_at=CURRENT_TIMESTAMP WHERE id=?').run(paid,status,id);
    return db.prepare('SELECT * FROM installment_plans WHERE id=?').get(id);
  });
}
function softDelete({ entity, id, label }) {
  const mapping = entity === 'quotation' ? ['quotations','number'] : ['installment_plans','reference'];
  return executeTransaction(() => {
    const row = db.prepare(`SELECT * FROM ${mapping[0]} WHERE id=?`).get(id);
    if (!row) return false;
    const payload = { row };
    if (entity === 'quotation') payload.items = db.prepare('SELECT * FROM quotation_items WHERE quotation_id=?').all(id);
    if (entity === 'installment') payload.payments = db.prepare('SELECT * FROM installment_payments WHERE installment_id=?').all(id);
    db.prepare('INSERT INTO deleted_records(id,entity_type,entity_id,label,payload_json,deleted_at) VALUES(?,?,?,?,?,?)')
      .run(`bin_${Date.now()}_${Math.random().toString(36).slice(2,7)}`, entity, id, label || row[mapping[1]], JSON.stringify(payload), new Date().toISOString());
    db.prepare(`DELETE FROM ${mapping[0]} WHERE id=?`).run(id);
    return true;
  });
}
function restoreDeleted(id) {
  return executeTransaction(() => {
    const deleted = db.prepare('SELECT * FROM deleted_records WHERE id=?').get(id);
    if (!deleted) return false;
    const payload = JSON.parse(deleted.payload_json);
    const row = payload.row || payload;
    let restored = false;
    if ((payload?.kind === 'collection' && payload?.collection) || payload?.kind === 'collection_batch') {
      restored = restoreCollectionPayload(payload);
    } else if (deleted.entity_type === 'quotation') {
      db.prepare(`INSERT OR REPLACE INTO quotations(id,number,customer_id,customer_name,quotation_date,valid_until,total,status,notes,created_at,updated_at) VALUES(?,?,?,?,?,?,?,?,?,?,?)`)
        .run(row.id,row.number,row.customer_id,row.customer_name,row.quotation_date,row.valid_until,row.total,row.status,row.notes,row.created_at,row.updated_at);
      const insertItem = db.prepare(`INSERT OR REPLACE INTO quotation_items(id,quotation_id,product_id,description,quantity,unit_price,line_total) VALUES(?,?,?,?,?,?,?)`);
      for (const item of payload.items || []) insertItem.run(item.id,item.quotation_id,item.product_id,item.description,item.quantity,item.unit_price,item.line_total);
      restored = true;
    } else if (deleted.entity_type === 'installment') {
      db.prepare(`INSERT OR REPLACE INTO installment_plans(id,invoice_id,customer_id,customer_name,reference,total,paid,due_date,status,created_at,updated_at) VALUES(?,?,?,?,?,?,?,?,?,?,?)`)
        .run(row.id,row.invoice_id,row.customer_id,row.customer_name,row.reference,row.total,row.paid,row.due_date,row.status,row.created_at,row.updated_at);
      const insertPayment = db.prepare(`INSERT OR REPLACE INTO installment_payments(id,installment_id,amount,paid_at,payment_method,notes) VALUES(?,?,?,?,?,?)`);
      for (const payment of payload.payments || []) insertPayment.run(payment.id,payment.installment_id,payment.amount,payment.paid_at,payment.payment_method,payment.notes);
      restored = true;
    }
    // Never remove an item from the recycle bin unless its payload was actually restored.
    if (!restored) throw new Error(`Unsupported or invalid recycle-bin payload: ${deleted.entity_type}`);
    db.prepare('DELETE FROM deleted_records WHERE id=?').run(id);
    return true;
  });
}

function safeArray(value) { return Array.isArray(value) ? value : []; }
function n(value, fallback=0) { const x=Number(value); return Number.isFinite(x) ? x : fallback; }
function text(value, fallback='') { return value == null ? fallback : String(value); }
function isPasswordHash(value) { return typeof value === 'string' && value.startsWith('scrypt$'); }
function hashPassword(value) {
  const password = text(value);
  if (!password || isPasswordHash(password)) return password || null;
  const salt = crypto.randomBytes(16);
  const derived = crypto.scryptSync(password, salt, 64, { N: 16384, r: 8, p: 1 });
  return `scrypt$16384$8$1$${salt.toString('base64')}$${derived.toString('base64')}`;
}
function verifyPassword(password, encoded) {
  if (!encoded) return !password;
  if (!isPasswordHash(encoded)) return crypto.timingSafeEqual(Buffer.from(String(password)), Buffer.from(String(encoded)));
  const parts = encoded.split('$');
  if (parts.length !== 6) return false;
  const [, nCost, rCost, pCost, saltB64, hashB64] = parts;
  const expected = Buffer.from(hashB64, 'base64');
  const actual = crypto.scryptSync(String(password), Buffer.from(saltB64, 'base64'), expected.length, { N:Number(nCost), r:Number(rCost), p:Number(pCost) });
  return expected.length === actual.length && crypto.timingSafeEqual(expected, actual);
}
function migratePlaintextPasswords() {
  const row = db.prepare("SELECT data_json FROM app_collections WHERE collection_name='teamMembers'").get();
  if (row) {
    const members = parseCollectionRow(row) || [];

    for (const member of members) if (member?.password && !isPasswordHash(member.password)) { member.password = hashPassword(member.password); changed = true; }
    if (changed) db.prepare("UPDATE app_collections SET data_json=?,revision=revision+1,updated_at=CURRENT_TIMESTAMP WHERE collection_name='teamMembers'").run(JSON.stringify(members));

  for (const member of plain) update.run(hashPassword(member.password), member.id);
}

function deleteMissing(table, ids) {
  db.exec('CREATE TEMP TABLE IF NOT EXISTS projection_seen(id TEXT PRIMARY KEY); DELETE FROM projection_seen;');
  const seen = db.prepare('INSERT OR IGNORE INTO projection_seen(id) VALUES(?)');
  for (const id of ids) seen.run(text(id));
  db.prepare(`DELETE FROM ${table} WHERE id NOT IN (SELECT id FROM projection_seen)`).run();
}

const COLLECTION_LABEL_FIELDS = {
  products:['name','code'], categories:['name'], suppliers:['name'], transactions:['invoiceNo','type'],
  customers:['name'], warehouses:['name'], transfers:['id'], expenses:['title','category'], teamMembers:['name','username'],
  partners:['name'], distributions:['partnerName','id'], services:['name'], vouchers:['voucherNumber','id'],
  invoices:['invoiceNo','id'], chatHistory:['title','message','id']
};
function collectionItemLabel(name,row) {
  for (const field of COLLECTION_LABEL_FIELDS[name] || ['name','id']) if (row?.[field]) return String(row[field]);
  return String(row?.id || name);
}
function captureDeletionBatch(collections) {
  const removed = [];
  const relatedChanges = [];
  const currentState = getCollections();
  // Compare the complete before/after operation state, not only one collection.
  // This preserves stock, customer balances, invoices and transaction side effects
  // when the UI sends them together as one operation.
  const afterState = { ...currentState };
  for (const [name,value] of Object.entries(collections || {})) {
    if (CORE_COLLECTIONS.has(name)) afterState[name] = sanitizeCollection(name,value);
  }
  for (const name of CORE_COLLECTIONS) {
    const current = currentState[name] || [];
    const next = afterState[name] || [];
    const currentById = new Map(current.filter(x=>x?.id != null).map(x=>[text(x.id),x]));
    const nextById = new Map(next.filter(x=>x?.id != null).map(x=>[text(x.id),x]));
    for (const [id,row] of currentById) if (!nextById.has(id)) removed.push({collection:name,row});
    for (const [id,before] of currentById) {
      const after=nextById.get(id);
      if (after && JSON.stringify(before)!==JSON.stringify(after)) relatedChanges.push({collection:name,id,before,after});
    }
  }
  if (!removed.length) return 0;
  const exists=db.prepare("SELECT 1 FROM deleted_records WHERE entity_type='operation_batch' AND entity_id=? LIMIT 1");
  const signature=removed.map(x=>`${x.collection}:${text(x.row.id)}`).sort().join('|');
  if (exists.get(signature)) return 0;
  const first=removed[0];
  const label=removed.length===1?collectionItemLabel(first.collection,first.row):`${collectionItemLabel(first.collection,first.row)} (+${removed.length-1})`;
  db.prepare('INSERT INTO deleted_records(id,entity_type,entity_id,label,payload_json,deleted_at) VALUES(?,?,?,?,?,?)')
    .run(`bin_${Date.now()}_${Math.random().toString(36).slice(2,8)}`,'operation_batch',signature,label,JSON.stringify({kind:'collection_batch',removed,relatedChanges,restoreMode:'reverse_operation'}),new Date().toISOString());
  return removed.length;
}


function normalizeUniqueCollectionNames(name, rows) {
  if (!['categories','warehouses'].includes(name)) return safeArray(rows);
  const used = new Set();
  return safeArray(rows).map((row, index) => {
    const copy = { ...row };
    const base = text(copy.name, name === 'categories' ? 'غير مصنف' : 'مخزن').trim();
    let candidate = base;
    let suffix = 2;
    while (used.has(candidate.toLocaleLowerCase())) candidate = `${base} (${suffix++})`;
    used.add(candidate.toLocaleLowerCase());
    copy.name = candidate;
    return copy;
  });
}

function saveCollectionRows(name, rows) {
  const clean=normalizeUniqueCollectionNames(name,sanitizeCollection(name,rows));
  db.prepare(`INSERT INTO app_collections(collection_name,data_json,item_count,revision,updated_at)
    VALUES(?,?,?,1,CURRENT_TIMESTAMP) ON CONFLICT(collection_name) DO UPDATE SET data_json=excluded.data_json,item_count=excluded.item_count,revision=app_collections.revision+1,updated_at=CURRENT_TIMESTAMP`)
    .run(name,JSON.stringify(clean),clean.length);
  return clean;
}
function restoreCollectionPayload(payload) {
  if (payload?.kind === 'collection_batch') {
    const impacted=new Map();
    for(const item of payload.removed||[]) {
      if(!CORE_COLLECTIONS.has(item.collection)||!item.row?.id) continue;
      const rows=impacted.get(item.collection)||getCollections([item.collection])[item.collection]||[];
      if(!rows.some(x=>text(x?.id)===text(item.row.id))) rows.push(item.row);
      impacted.set(item.collection,rows);
    }
    for(const change of payload.relatedChanges||[]) {
      if(!CORE_COLLECTIONS.has(change.collection)) continue;
      const rows=impacted.get(change.collection)||getCollections([change.collection])[change.collection]||[];
      const index=rows.findIndex(x=>text(x?.id)===text(change.id));
      // Never overwrite newer user work: reverse only when the row still equals the post-delete state.
      if(index>=0 && JSON.stringify(rows[index])===JSON.stringify(change.after)) rows[index]=change.before;
      impacted.set(change.collection,rows);
    }
    const projection={};
    for(const [name,rows] of impacted) projection[name]=saveCollectionRows(name,rows);
    // Re-project only the collections affected by the restored operation.
    // A full relational rebuild deletes parent rows and can violate foreign keys
    // held by installments, payroll, attendance, or other modules.
    projectRelationalIncremental(projection);
    return true;
  }
  const name = payload?.collection, row = payload?.row;
  if (!CORE_COLLECTIONS.has(name) || !row?.id) return false;
  const stored = getCollections([name])[name] || [];
  if (!stored.some(x => text(x?.id) === text(row.id))) stored.push(row);
  const clean=saveCollectionRows(name,stored);
  projectRelationalIncremental({[name]:clean});
  return true;
}

function saveCollections(collections, options={}) {
  return executeTransaction(() => {
    const upsert=db.prepare(`INSERT INTO app_collections(collection_name,data_json,item_count,revision,updated_at)
      VALUES(?,?,?,1,CURRENT_TIMESTAMP) ON CONFLICT(collection_name) DO UPDATE SET data_json=excluded.data_json,item_count=excluded.item_count,revision=app_collections.revision+1,updated_at=CURRENT_TIMESTAMP`);
    if(options.trackDeletes) captureDeletionBatch(collections);
    const clean={};
    for(const [name,value] of Object.entries(collections||{})) {
      if(!CORE_COLLECTIONS.has(name)) continue;
      clean[name]=sanitizeCollection(name,value);
      upsert.run(name,JSON.stringify(clean[name]),clean[name].length);
    }
    if(options.projectRelational!==false) {
      if(options.fullRebuild) replaceRelationalProjection({...getCollections(),...clean});
      else projectRelationalIncremental(clean);
    }
    if(options.markMigrated) db.prepare('INSERT OR IGNORE INTO schema_migrations(version) VALUES(2)').run();
    return Object.fromEntries(Object.entries(clean).map(([k,v])=>[k,v.length]));
  });
}

function generatePayrollRun(data={}) {
  return executeTransaction(()=>{
    const month=text(data.month,new Date().toISOString().slice(0,7));
    const start=`${month}-01`; const [y,m]=month.split('-').map(Number); const end=new Date(Date.UTC(y,m,0)).toISOString().slice(0,10);
    const runId=text(data.id,`payrun_${month}`); const generatedAt=new Date().toISOString();
    db.prepare(`INSERT INTO payroll_runs(id,month,status,generated_at,notes) VALUES(?,?,'DRAFT',?,?) ON CONFLICT(month) DO UPDATE SET generated_at=excluded.generated_at,notes=excluded.notes,status='DRAFT'`).run(runId,month,generatedAt,data.notes||null);
    const actualRun=db.prepare('SELECT id FROM payroll_runs WHERE month=?').get(month).id;
    db.prepare('DELETE FROM payroll_items WHERE run_id=?').run(actualRun);
    const employees=db.prepare('SELECT * FROM payroll_employees WHERE active=1 ORDER BY name').all();
    const adjustments=db.prepare('SELECT * FROM payroll_adjustments WHERE month=?').all(month);
    const itemStmt=db.prepare(`INSERT INTO payroll_items(id,run_id,employee_id,employee_name,base_salary,allowances,bonuses,overtime_hours,overtime_amount,absence_days,absence_deduction,deductions,loans,gross_salary,net_salary,worked_hours,attendance_days,snapshot_json) VALUES(?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?,?)`);
    let grossTotal=0,deductionsTotal=0,netTotal=0;
    for(const e of employees){
      const att=db.prepare(`SELECT substr(event_time,1,10) day,
        MIN(CASE WHEN event_type='CHECK_IN' THEN event_time END) first_in,
        MAX(CASE WHEN event_type='CHECK_OUT' THEN event_time END) last_out
        FROM attendance_records WHERE employee_id=? AND substr(event_time,1,10) BETWEEN ? AND ? GROUP BY substr(event_time,1,10)`).all(e.attendance_employee_id||e.id,start,end);
      let worked=0,days=0; for(const r of att){ if(r.first_in){days++; if(r.last_out) worked+=(new Date(r.last_out)-new Date(r.first_in))/3600000; } }
      const expected=Math.max(0,Number(e.standard_daily_hours)||8)*days; const overtimeHours=Math.max(0,worked-expected);
      const overtimeAmount=overtimeHours*(Number(e.overtime_rate)||Number(e.hourly_rate)||0);
      const absenceDays=Math.max(0,(Number(e.standard_monthly_days)||30)-days);
      const absenceDeduction=(Number(e.base_salary)||0)/(Number(e.standard_monthly_days)||30)*absenceDays;
      const own=adjustments.filter(a=>a.employee_id===e.id); const sum=t=>own.filter(a=>a.adjustment_type===t).reduce((v,a)=>v+Number(a.amount||0),0);
      const allowances=sum('ALLOWANCE')+sum('OTHER'), bonuses=sum('BONUS'), deductions=sum('DEDUCTION'), loans=sum('LOAN');
      const gross=(Number(e.base_salary)||0)+allowances+bonuses+overtimeAmount; const totalDed=absenceDeduction+deductions+loans; const net=Math.max(0,gross-totalDed);
      itemStmt.run(`payitem_${actualRun}_${e.id}`,actualRun,e.id,e.name,e.base_salary,allowances,bonuses,overtimeHours,overtimeAmount,absenceDays,absenceDeduction,deductions,loans,gross,net,worked,days,JSON.stringify({employee:e,adjustments:own,period:{start,end}}));
      grossTotal+=gross; deductionsTotal+=totalDed; netTotal+=net;
    }
    db.prepare('UPDATE payroll_runs SET gross_total=?,deductions_total=?,net_total=? WHERE id=?').run(grossTotal,deductionsTotal,netTotal,actualRun);
    invalidateQueryCache(); return {id:actualRun,month,employees:employees.length,grossTotal,deductionsTotal,netTotal};
  });
}

  return saveCollections(snapshot.collections,{projectRelational:true,fullRebuild:true,markMigrated:true});
}
