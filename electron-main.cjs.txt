// Register IPC handlers for Electron printing and printer listing
ipcMain.handle('get-printers', async (event) => {
  try {
    if (event.sender.getPrintersAsync) {
      return await event.sender.getPrintersAsync();
    } else {
      return event.sender.getPrinters();
    }
  } catch (err) {
    console.error('Failed to get printers:', err);
    return [];
  }
});

ipcMain.handle('print-document', async (event, { html, options }) => {
  return new Promise((resolve, reject) => {
    try {
      const tempFile = path.join(os.tmpdir(), `erp-hub-print-${Date.now()}-${Math.floor(Math.random() * 1000)}.html`);
      fs.writeFileSync(tempFile, html, 'utf-8');
      
      const workerWindow = new BrowserWindow({
        show: false,
        webPreferences: {
          nodeIntegration: false,
          contextIsolation: true,
          sandbox: true
        }
      });
      
      workerWindow.loadFile(tempFile);
      
      workerWindow.webContents.on('did-finish-load', () => {
        const electronPrintOptions = {
          silent: options.silent ?? false,
          printBackground: options.printBackground ?? true,
          deviceName: options.deviceName || '',
          color: options.color ?? true,
          landscape: options.landscape ?? false,
          copies: options.copies ?? 1
        };
        
        workerWindow.webContents.print(electronPrintOptions, (success, failureReason) => {
          try {
            fs.unlinkSync(tempFile);
          } catch (e) {
            console.error('Failed to remove temp file:', e);
          }
          workerWindow.destroy();
          
          if (success) {
            resolve({ success: true });
          } else {
            resolve({ success: false, error: failureReason });
          }
        });
      });
      
      workerWindow.webContents.on('did-fail-load', (err) => {
        try {
          fs.unlinkSync(tempFile);
        } catch (e) {}
        workerWindow.destroy();
        reject(new Error('Failed to load temporary print file'));
      });
    } catch (error) {
      console.error('Error during silent printing:', error);
      reject(error);
    }
  });
});

ipcMain.handle('save-pdf', async (event, { html, options }) => {
  return new Promise(async (resolve) => {
    try {
      const defaultName = `erp-document-${Date.now()}.pdf`;
      const { filePath, canceled } = await dialog.showSaveDialog({
        title: 'Save PDF File / حفظ ملف PDF',
        defaultPath: path.join(app.getPath('documents'), defaultName),
        filters: [
          { name: 'PDF Files (*.pdf)', extensions: ['pdf'] }
        ]
      });

      if (canceled || !filePath) {
        return resolve({ success: false, cancelled: true });
      }

      const tempFile = path.join(os.tmpdir(), `erp-hub-pdf-${Date.now()}-${Math.floor(Math.random() * 1000)}.html`);
      fs.writeFileSync(tempFile, html, 'utf-8');

      const workerWindow = new BrowserWindow({
        show: false,
        webPreferences: {
          nodeIntegration: false,
          contextIsolation: true,
          sandbox: true
        }
      });

      workerWindow.loadFile(tempFile);

      workerWindow.webContents.on('did-finish-load', async () => {
        try {
          const isThermal = options?.layout === 'thermal';
          const pdfOptions = {
            margins: { marginType: 'default' },
            printBackground: true,
            landscape: options?.landscape ?? false,
            pageSize: isThermal ? { width: 80000, height: 150000 } : 'A4'
          };

          const data = await workerWindow.webContents.printToPDF(pdfOptions);
          fs.writeFileSync(filePath, data);

          try {
            fs.unlinkSync(tempFile);
          } catch (e) {}
          workerWindow.destroy();

          resolve({ success: true, filePath });
        } catch (err) {
          console.error('Failed to generate PDF content:', err);
          try {
            fs.unlinkSync(tempFile);
          } catch (e) {}
          workerWindow.destroy();
          resolve({ success: false, error: err.message });
        }
      });

      workerWindow.webContents.on('did-fail-load', () => {
        try {
          fs.unlinkSync(tempFile);
        } catch (e) {}
        workerWindow.destroy();
        resolve({ success: false, error: 'Failed to load temporary PDF HTML file' });
      });

    } catch (error) {
      console.error('Error inside save-pdf IPC:', error);
      resolve({ success: false, error: error.message });
    }
  });
});

        show: false,
        webPreferences: {
          nodeIntegration: false,
          contextIsolation: true,
          sandbox: true
        }
      });

      workerWindow.loadFile(tempHtml);

      workerWindow.webContents.on('did-finish-load', async () => {
        try {
          const isThermal = options?.layout === 'thermal';
          const pdfOptions = {
            margins: { marginType: 'default' },
            printBackground: true,
            landscape: options?.landscape ?? false,
            pageSize: isThermal ? { width: 80000, height: 150000 } : 'A4'
          };

          const data = await workerWindow.webContents.printToPDF(pdfOptions);
          const tempPdf = path.join(os.tmpdir(), `erp-document-${Date.now()}.pdf`);
          fs.writeFileSync(tempPdf, data);

          try {
            fs.unlinkSync(tempHtml);
          } catch (e) {}
          workerWindow.destroy();

          // Open with default system viewer
          await shell.openPath(tempPdf);
          resolve({ success: true });
        } catch (err) {
          console.error('Failed to generate preview PDF:', err);
          try {
            fs.unlinkSync(tempHtml);
          } catch (e) {}
          workerWindow.destroy();
          resolve({ success: false, error: err.message });
        }
      });

      workerWindow.webContents.on('did-fail-load', () => {
        try {
          fs.unlinkSync(tempHtml);
        } catch (e) {}
        workerWindow.destroy();
        resolve({ success: false, error: 'Failed to load preview HTML' });
      });
    } catch (e) {
      console.error(e);
      resolve({ success: false, error: e.message });
    }
  });
});

// Focus existing window instance if user attempts to launch a second one
app.on('second-instance', () => {
  if (mainWindow) {
    if (mainWindow.isMinimized()) mainWindow.restore();
    mainWindow.focus();
  }
});

app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    app.quit();
