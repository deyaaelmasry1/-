  printerSelfTest: (deviceName) => ipcRenderer.invoke('printer-self-test', deviceName),
  database: {
    list: (args) => ipcRenderer.invoke('db-list', args),
    createQuotation: (data) => ipcRenderer.invoke('db-create-quotation', data),
    updateQuotation: (data) => ipcRenderer.invoke('db-update-quotation', data),
    getQuotation: (id) => ipcRenderer.invoke('db-get-quotation', id),

    savePayrollEmployee: (data) => ipcRenderer.invoke('db-save-payroll-employee', data),
    addPayrollAdjustment: (data) => ipcRenderer.invoke('db-add-payroll-adjustment', data),
    generatePayrollRun: (data) => ipcRenderer.invoke('db-generate-payroll-run', data),

    getPayrollRun: (id) => ipcRenderer.invoke('db-get-payroll-run', id),
    saveAttendanceDevice: (data) => ipcRenderer.invoke('db-save-attendance-device', data),
    recordAttendance: (data) => ipcRenderer.invoke('db-record-attendance', data),
    importAttendance: (rows) => ipcRenderer.invoke('db-import-attendance', rows),

    restoreDeleted: (id) => ipcRenderer.invoke('db-restore-deleted', id),
    purgeDeleted: (id) => ipcRenderer.invoke('db-purge-deleted', id),
    importLegacy: (data) => ipcRenderer.invoke('db-import-legacy', data),
    stats: () => ipcRenderer.invoke('db-stats'),
    getCollections: (names) => ipcRenderer.invoke('db-get-collections', names),
    saveCollections: (collections, options) => ipcRenderer.invoke('db-save-collections', { collections, options }),
    hasMigratedLegacy: () => ipcRenderer.invoke('db-has-migrated-legacy'),
    exportSnapshot: () => ipcRenderer.invoke('db-export-snapshot'),
    importSnapshot: (snapshot) => ipcRenderer.invoke('db-import-snapshot', snapshot),
    batchSaveCollections: (batches, options) => ipcRenderer.invoke('db-batch-save-collections', { batches, options }),
    enqueueSyncChange: (data) => ipcRenderer.invoke('db-enqueue-sync-change', data),
    pendingSync: (args) => ipcRenderer.invoke('db-pending-sync', args),
    optimize: () => ipcRenderer.invoke('db-optimize')
  },
