const fs=require('node:fs'); const path=require('node:path');
const root=path.join(__dirname,'..','src'); const files=[];
(function walk(d){for(const e of fs.readdirSync(d,{withFileTypes:true})){const p=path.join(d,e.name);e.isDirectory()?walk(p):/\.(ts|tsx)$/.test(e.name)&&files.push(p)}})(root);
const rows=files.map(file=>{const text=fs.readFileSync(file,'utf8');return {file:path.relative(path.join(__dirname,'..'),file),lines:text.split('\n').length,todos:(text.match(/TODO|FIXME/g)||[]).length}}).sort((a,b)=>b.lines-a.lines);
const duplicates=[]; const seen=new Map();
for(const f of files){const txt=fs.readFileSync(f,'utf8').replace(/\s+/g,' ').trim(); if(txt.length<200)continue; const key=require('crypto').createHash('sha1').update(txt).digest('hex'); if(seen.has(key))duplicates.push([seen.get(key),f]); else seen.set(key,f)}
console.log(JSON.stringify({files:files.length,totalLines:rows.reduce((n,r)=>n+r.lines,0),largest:rows.slice(0,20),todoCount:rows.reduce((n,r)=>n+r.todos,0),exactDuplicateFiles:duplicates},null,2));
if(duplicates.length) process.exitCode=1;
