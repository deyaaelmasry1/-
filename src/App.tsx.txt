function getTxSignature(t: any): string {
  if (!t) return '';
  const invNoMatch = t.notes ? t.notes.match(/#([A-Za-z0-9-_]+)/) : null;
  const invNo = invNoMatch ? invNoMatch[1] : '';
  const prod = t.productId || t.serviceId || '';
  const type = t.type || '';
  const qty = t.quantity || 0;
  const price = t.unitPrice || 0;
  if (invNo) {
    return `${type}_${invNo}_${prod}_${qty}_${price}`;
  }
  return String(t.id);
}

        const oldDeviceId = localStorage.getItem('license_device_id') || deviceId;
        const savedKey = localStorage.getItem('license_key');
        const wasLicensedLocally = savedKey && oldDeviceId && (savedKey === generateLicenseKey(oldDeviceId));

          if (wasLicensedLocally) {
            localStorage.setItem('license_key', expectedKey);
            setIsLicensed(true);
            console.log('Successfully migrated local license to hardware-based system.');

            // Sync the newly generated key to the cloud
            try {
              await fetch('/api/cloud-license', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ deviceId: hwId, licenseKey: expectedKey })
              });
              console.log('Migrated license successfully synchronized to cloud.');
            } catch (syncErr) {
              console.error('Failed to sync migrated license to cloud:', syncErr);
            }
            return;
          }

          // Otherwise, check if currently saved key matches the new hardware ID
          if (savedKey === expectedKey) {
            setIsLicensed(true);
          } else {
            // Attempt to automatically restore license from cloud registry
            try {
              const cloudResponse = await fetch(`/api/cloud-license?deviceId=${encodeURIComponent(hwId)}`);
              const cloudData = await cloudResponse.json();
              if (cloudData && cloudData.success && cloudData.licenseKey) {
                if (cloudData.licenseKey === expectedKey) {
                  localStorage.setItem('license_key', cloudData.licenseKey);
                  setIsLicensed(true);
                  console.log('License successfully recognized and activated automatically from the cloud.');

              console.error('Failed to query cloud license:', cloudErr);
            }
            // Fallback to unlicensed (which activates the standard 1-day trial limit)
            setIsLicensed(false);
          }

        console.error('Failed to initialize hardware-based licensing system:', err);

          {activeTab === 'transactions' && (
            <Transactions 
              products={products}
              transactions={transactions}
              language={language}
              onAddTransaction={handleAddTransaction}
              onDeleteTransaction={handleDeleteTransaction}
              currencySymbol={currencySymbol}
            />
          )}

          {activeTab === 'suppliers' && (
            <Suppliers 
              suppliers={suppliers}
              products={products}
              language={language}
              onAddSupplier={handleAddSupplier}
              onUpdateSupplier={handleUpdateSupplier}
              onDeleteSupplier={handleDeleteSupplier}
              customers={customers}
              currencySymbol={currencySymbol}
              onOpenSupplier={(supplier) => {
                setProfileOpenRequest({ type: 'supplier', id: supplier.id, token: Date.now() });
                setErpSubTab('customers');
                setActiveTab('erp');
              }}
            />
          )}

          {activeTab === 'customers' && (
            <Customers 
              customers={customers}
              transactions={transactions}
              products={products}
              language={language}
              onAddCustomer={handleAddCustomer}
              onUpdateCustomer={handleUpdateCustomer}
              onDeleteCustomer={handleDeleteCustomer}
              suppliers={suppliers}
              currencySymbol={currencySymbol}
              onOpenCustomer={(customer) => {
                setProfileOpenRequest({ type: 'customer', id: customer.id, token: Date.now() });
                setErpSubTab('customers');
                setActiveTab('erp');
              }}
            />
          )}

          <div className={activeTab === 'erp' ? "flex flex-col flex-1" : "hidden"}>
            <ErpHub
              products={products}
              categories={categories}
              suppliers={suppliers}
              transactions={transactions}
              customers={customers}
              warehouses={warehouses}
              transfers={transfers}
              expenses={expenses}
              teamMembers={teamMembers}
              partners={partners}
              distributions={distributions}
              services={services}
              language={language}
              currencySymbol={currencySymbol}
              setProducts={setProducts}
              setTransactions={setTransactions}
              setCustomers={setCustomers}
              setSuppliers={setSuppliers}
              setTransfers={setTransfers}
              setExpenses={setExpenses}
              setTeamMembers={setTeamMembers}
              setPartners={setPartners}
              setDistributions={setDistributions}
              setWarehouses={setWarehouses}
              setServices={setServices}
              vouchers={vouchers}
              setVouchers={setVouchers}
              invoices={invoices}
              setInvoices={setInvoices}
              chatHistory={chatHistory}
              setChatHistory={setChatHistory}
              setCategories={setCategories}
              showToast={showToast}
              currentUser={currentUser}
              setCurrentUser={setCurrentUser}
              activeSubTab={erpSubTab}
              setActiveSubTab={setErpSubTab}
              profileOpenRequest={profileOpenRequest}
              companyName={companyName}
              companyAddress={companyAddress}
              companyPhone={companyPhone}
              companyLogo={companyLogo}
              setCompanyName={setCompanyName}
              setCompanyAddress={setCompanyAddress}
              setCompanyPhone={setCompanyPhone}
              setCompanyLogo={setCompanyLogo}
              firebaseUrl={firebaseUrl}
              setFirebaseUrl={setFirebaseUrl}
              teamId={teamId}
              setTeamId={setTeamId}
              cloudSyncType={cloudSyncType}
              setCloudSyncType={setCloudSyncType}
              isAutoSyncEnabled={isAutoSyncEnabled}
              setIsAutoSyncEnabled={setIsAutoSyncEnabled}
              isLiveMonitoringActive={isLiveMonitoringActive}
              setIsLiveMonitoringActive={setIsLiveMonitoringActive}
              liveNotifications={liveNotifications}
              setLiveNotifications={setLiveNotifications}
            />
          </div>

          {activeTab === 'ai' && (
            <AiAdvisor 
              products={products}
              categories={categories}
              suppliers={suppliers}
              transactions={transactions}
              language={language}
              chatHistory={chatHistory}
              onAddChatMessage={handleAddChatMessage}
              onClearChatHistory={handleClearChatHistory}
            />
          )}

          {activeTab === 'reports' && (
            <Reports
              products={products}
              categories={categories}
              suppliers={suppliers}
              transactions={transactions}
              customers={customers}
              warehouses={warehouses}
              expenses={expenses}
              vouchers={vouchers}
              invoices={invoices}
              teamMembers={teamMembers}
              language={language}
              companyName={companyName}
              currencySymbol={currencySymbol}
              setProducts={setProducts}
              setSuppliers={setSuppliers}
              setTransactions={setTransactions}
              setCustomers={setCustomers}
              setWarehouses={setWarehouses}
              setExpenses={setExpenses}
              setVouchers={setVouchers}
              setInvoices={setInvoices}
            />
          )}

          {activeTab === 'settings' && (
            <Settings
              language={language}
              companyName={companyName}
              setCompanyName={setCompanyName}
              companyAddress={companyAddress}
              setCompanyAddress={setCompanyAddress}
              companyPhone={companyPhone}
              setCompanyPhone={setCompanyPhone}
              companyLogo={companyLogo}
              setCompanyLogo={setCompanyLogo}
              currencySymbol={currencySymbol}
              setCurrencySymbol={setCurrencySymbol}
              products={products}
              setProducts={setProducts}
              categories={categories}
              setCategories={setCategories}
              suppliers={suppliers}
              setSuppliers={setSuppliers}
              transactions={transactions}
              setTransactions={setTransactions}
              chatHistory={chatHistory}
              setChatHistory={setChatHistory}
              customers={customers}
              setCustomers={setCustomers}
              warehouses={warehouses}
              setWarehouses={setWarehouses}
              transfers={transfers}
              setTransfers={setTransfers}
              expenses={expenses}
              setExpenses={setExpenses}
              teamMembers={teamMembers}
              setTeamMembers={setTeamMembers}
              partners={partners}
              setPartners={setPartners}
              handleBackupDatabase={handleBackupDatabase}
              handleRestoreDatabase={handleRestoreDatabase}
              handleSyncDatabase={handleSyncDatabase}
              handleClearAllData={handleClearAllData}
              showToast={showToast}
            />
          )}
        </main>
      </div>

      {/* --- MOBILE BOTTOM NAVIGATION BAR (Android design guideline compliance) --- */}
      <div className="md:hidden fixed bottom-0 left-0 right-0 bg-slate-900/95 backdrop-blur-md border-t border-slate-800 text-slate-400 flex items-center justify-around py-2 px-1 z-40 shadow-[0_-5px_15px_-3px_rgba(0,0,0,0.3)] no-print">
        {/* Tab 1: Dashboard */}
        {hasTabAccess(currentUser.role, 'dashboard') && (
          <button 
            onClick={() => setActiveTab('dashboard')} 
            className={`flex flex-col items-center gap-0.5 flex-1 transition cursor-pointer ${
              activeTab === 'dashboard' ? 'text-indigo-400 font-bold scale-105' : 'text-slate-400 hover:text-slate-200'
            }`}
          >
            <LayoutDashboard className="w-5 h-5" />
            <span className="text-[9px] tracking-tight">{language === 'ar' ? 'الرئيسية' : 'Dashboard'}</span>
          </button>
        )}

        {/* Tab 2: Inventory */}
        {hasTabAccess(currentUser.role, 'inventory') && (
          <button 
            onClick={() => setActiveTab('inventory')} 
            className={`flex flex-col items-center gap-0.5 flex-1 transition cursor-pointer ${
              activeTab === 'inventory' ? 'text-indigo-400 font-bold scale-105' : 'text-slate-400 hover:text-slate-200'
            }`}
          >
            <Package className="w-5 h-5" />
            <span className="text-[9px] tracking-tight">{language === 'ar' ? 'المستودع' : 'Inventory'}</span>
          </button>
        )}

        {/* Tab 3: Dedicated Scanning Center */}
        <button 
          onClick={() => setIsScannerOpen(true)} 
          className="flex flex-col items-center justify-center -mt-6 bg-indigo-600 text-white rounded-full w-12 h-12 shadow-lg shadow-indigo-600/35 border-4 border-slate-900 active:scale-95 transition cursor-pointer relative"
        >
          {/* Pulse highlight */}
          <span className="absolute inset-0 rounded-full bg-indigo-500 animate-ping opacity-25"></span>
          <Camera className="w-5 h-5" />
        </button>

        {/* Tab 4: Financials & ERP */}
        {hasTabAccess(currentUser.role, 'erp') && (
          <button 
            onClick={() => setActiveTab('erp')} 
            className={`flex flex-col items-center gap-0.5 flex-1 transition cursor-pointer ${
              activeTab === 'erp' ? 'text-indigo-400 font-bold scale-105' : 'text-slate-400 hover:text-slate-200'
            }`}
          >
            <Briefcase className="w-5 h-5" />
            <span className="text-[9px] tracking-tight">{language === 'ar' ? 'المالية' : 'ERP'}</span>
          </button>
        )}

        {/* Tab 4.5: Reports */}
        {hasTabAccess(currentUser.role, 'reports') && currentUser.role === 'admin' && (
          <button 
            onClick={() => setActiveTab('reports')} 
            className={`flex flex-col items-center gap-0.5 flex-1 transition cursor-pointer ${
              activeTab === 'reports' ? 'text-indigo-400 font-bold scale-105' : 'text-slate-400 hover:text-slate-200'
            }`}
          >
            <FileBarChart className="w-5 h-5 text-amber-400" />
            <span className="text-[9px] tracking-tight">{language === 'ar' ? 'التقارير' : 'Reports'}</span>
          </button>
        )}

        {/* Tab 5: AI Advisor */}
        {hasTabAccess(currentUser.role, 'ai') && (
          <button 
            onClick={() => setActiveTab('ai')} 
            className={`flex flex-col items-center gap-0.5 flex-1 transition cursor-pointer ${
              activeTab === 'ai' ? 'text-indigo-400 font-bold scale-105' : 'text-slate-400 hover:text-slate-200'
            }`}
          >
            <BrainCircuit className="w-5 h-5" />
            <span className="text-[9px] tracking-tight">{language === 'ar' ? 'الذكاء' : 'AI'}</span>
          </button>
        )}
      </div>

      {/* --- FLOATING CAM SCANNER FAB FOR HANDHELD TAB USERS (For easy thumb access on medium tablets/phones) --- */}
      <button
        onClick={() => setIsScannerOpen(true)}
        className={`fixed hidden md:flex lg:hidden bottom-6 ${
          isRtl ? 'left-6' : 'right-6'
        } bg-indigo-600 hover:bg-indigo-700 text-white p-3.5 rounded-2xl shadow-xl shadow-indigo-600/30 z-40 active:scale-95 transition hover:scale-105 border border-indigo-500/30 cursor-pointer`}
        title={language === 'ar' ? 'قارئ باركود الكاميرا' : 'Camera Barcode Scanner'}
      >
        <Camera className="w-5 h-5" />
      </button>

      {/* --- INTERACTIVE CAMERA BARCODE SCANNER MODAL --- */}
      <BarcodeScanner 
        isOpen={isScannerOpen}
        onClose={() => setIsScannerOpen(false)}
        products={products}
        language={language}
        onScanSuccess={handleScanSuccess}
      />

      {/* --- HIGH-CONTRAST FLOATING TOAST NOTIFICATION --- */}
      {toast && (
        <div 
          className={`fixed bottom-20 left-1/2 -translate-x-1/2 z-50 flex items-center gap-2.5 px-4.5 py-3 rounded-2xl shadow-xl text-xs font-bold border animate-in slide-in-from-bottom duration-350 ${
            toast.type === 'success' 
              ? 'bg-slate-900 border-emerald-500/30 text-white' 
              : 'bg-red-950 border-red-500/30 text-red-100'
          }`}
        >
          {toast.type === 'success' ? (
            <CheckCircle className="w-4.5 h-4.5 text-emerald-400 shrink-0" />
          ) : (
            <AlertCircle className="w-4.5 h-4.5 text-red-400 shrink-0" />
          )}
          <span>{toast.message}</span>
        </div>
      )}
      {/* --- REAL-TIME INTUITIVE PRINT PREVIEW MODAL FOR WINDOWS & POS DEVICES --- */}
      <PrintPreviewModal
        isOpen={printPreview.isOpen}
        onClose={() => setPrintPreview(prev => ({ ...prev, isOpen: false }))}
        html={printPreview.html}
        initialLayout={printPreview.layout}
        title={printPreview.title}
        language={language}
        showToast={showToast}
      />

      {/* --- INSTALLATION DATA RECOVERY OR FRESH START DIALOG --- */}
      {showInstallPrompt && (
        <div className="fixed inset-0 z-[100] flex items-center justify-center p-4 bg-slate-950/85 backdrop-blur-md animate-fade-in">
          <div className="bg-slate-900 border border-slate-800 rounded-3xl max-w-lg w-full p-6 shadow-2xl text-slate-100 flex flex-col gap-6 text-center animate-in zoom-in-95 duration-200">
            <div className="mx-auto bg-indigo-950/50 p-4 rounded-full border border-indigo-500/20 text-indigo-400">
              <Database className="w-8 h-8" />
            </div>
            
            <div className="space-y-2">
              <h3 className="text-lg font-bold text-slate-100">
                {language === 'ar' ? 'تنبيه تحديث وتثبيت البيانات' : 'System Update & Setup Detected'}
              </h3>
              <p className="text-xs text-slate-400 leading-relaxed">
                {language === 'ar' 
                  ? 'تم رصد وجود بيانات ومستندات محفوظة من نسخة سابقة للبرنامج على هذا الجهاز. كيف تود استكمال المتابعة؟' 
                  : 'We detected saved database records from a previous installation of the software on this machine. How would you like to continue?'}
              </p>
            </div>

            <div className="grid grid-cols-1 gap-3.5 mt-2">
              {/* Option 1: Keep Old Data */}
              <button
                onClick={handleKeepOldData}
                className="group flex flex-col items-center justify-center p-4 bg-slate-800/50 hover:bg-slate-800 border border-slate-700/60 hover:border-slate-600 rounded-2xl transition active:scale-[0.99] cursor-pointer"
              >
                <div className="flex items-center gap-2.5 w-full justify-center">
                  <span className="p-1.5 bg-emerald-950/40 border border-emerald-500/20 text-emerald-400 rounded-lg group-hover:scale-110 transition">
                    <CheckCircle className="w-4 h-4" />
                  </span>
                  <span className="text-xs font-bold text-slate-200">
                    {language === 'ar' ? 'الاحتفاظ بالبيانات القديمة ومتابعة العمل' : 'Retain Existing Data & Continue'}
                  </span>
                </div>
                <p className="text-[10px] text-slate-400 mt-1 text-center">
                  {language === 'ar' 
                    ? 'سيتم الحفاظ على جميع المنتجات، المبيعات، الفواتير، والعملاء بدون أي تعديل.' 
                    : 'All previous inventory records, sales, invoices, and customers will be kept.'}
                </p>
              </button>

              {/* Option 2: Clean Slate / Fresh Start */}
              <button
                onClick={handleStartFresh}
                className="group flex flex-col items-center justify-center p-4 bg-red-950/15 hover:bg-red-950/25 border border-red-900/20 hover:border-red-950/40 rounded-2xl transition active:scale-[0.99] cursor-pointer"
              >
                <div className="flex items-center gap-2.5 w-full justify-center">
                  <span className="p-1.5 bg-red-950/40 border border-red-500/20 text-red-400 rounded-lg group-hover:scale-110 transition">
                    <Trash2 className="w-4 h-4" />
                  </span>
                  <span className="text-xs font-bold text-red-200">
                    {language === 'ar' ? 'مسح البيانات السابقة والبدء من الصفر' : 'Full Clear & Fresh Start'}
                  </span>
                </div>
                <p className="text-[10px] text-red-400/80 mt-1 text-center">
                  {language === 'ar' 
                    ? '⚠️ تحذير: سيتم مسح كافة البيانات وتصفير النظام بالكامل لتأسيس قاعدة بيانات جديدة ونظيفة.' 
                    : '⚠️ Warning: This will permanently wipe all existing local database history to start from scratch.'}
                </p>
              </button>
            </div>
            
            <p className="text-[9px] text-slate-500">
              {language === 'ar' 
                ? '* تضمن هذه الخطوة نجاح واستقرار التثبيت وتجنب أي تعارض في قواعد البيانات القديمة.' 
                : '* This choice ensures safe database structure integrity and prevents installation mismatch errors.'}
            </p>
          </div>
        </div>
      )}
    </div>
  );
}
