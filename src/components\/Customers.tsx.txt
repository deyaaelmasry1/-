import React, { useState, useMemo } from 'react';
import { Customer, Transaction, Product, Supplier } from '../types';
import { Language, translations } from '../utils/translations';
import { printHtml } from '../utils/printHelper';
import { 
  Users, 
  Phone, 
  Mail, 
  MapPin, 
  Plus, 
  Trash2, 
  Search, 
  UserPlus, 
  X, 
  TrendingUp, 
  Wallet,
  History,
  TrendingDown,
  Percent,
  Layers,
  ArrowUpRight,
  ArrowDownLeft,
  Edit2,
  Printer,
  Share2,
  Download
} from 'lucide-react';

interface CustomersProps {
  customers: Customer[];
  transactions: Transaction[];
  products: Product[];
  language: Language;
  onAddCustomer: (customer: Omit<Customer, 'id'>) => void;
  onUpdateCustomer: (customer: Customer) => void;
  onDeleteCustomer: (id: string) => void;
  suppliers?: Supplier[];
  currencySymbol?: string;
  onOpenCustomer?: (customer: Customer) => void;
}

export default function Customers({
  customers,
  transactions,
  products,
  language,
  onAddCustomer,
  onUpdateCustomer,
  onDeleteCustomer,
  suppliers = [],
  currencySymbol = language === 'ar' ? 'ج.م' : '$',
  onOpenCustomer
}: CustomersProps) {
  const t = translations[language];
  const isRtl = language === 'ar';

  // Search & Modal State
  const [searchQuery, setSearchQuery] = useState('');
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [editingCustomer, setEditingCustomer] = useState<Customer | null>(null);
  const [selectedCustomerId, setSelectedCustomerId] = useState<string | null>(null);

  // Import from Suppliers State
  const [isImportModalOpen, setIsImportModalOpen] = useState(false);
  const [supplierSearchQuery, setSupplierSearchQuery] = useState('');

  // New Customer form state
  const [custName, setCustName] = useState('');
  const [custPhone, setCustPhone] = useState('');
  const [custEmail, setCustEmail] = useState('');
  const [custAddress, setCustAddress] = useState('');
  const [initialDebt, setInitialDebt] = useState('0');
  const [balanceType, setBalanceType] = useState<'DEBIT' | 'CREDIT'>('DEBIT');

  // Filter customers based on search query
  const filteredCustomers = useMemo(() => {
    return customers.filter(c => {
      const query = searchQuery.toLowerCase();
      return (
        c.name.toLowerCase().includes(query) ||
        (c.phone && c.phone.includes(query)) ||
        (c.email && c.email.toLowerCase().includes(query)) ||
        (c.address && c.address.toLowerCase().includes(query))
      );
    });
  }, [customers, searchQuery]);

  // Precompute customer sales count in O(T) instead of O(C * T) for maximum CPU efficiency
  const customerSalesCounts = useMemo(() => {
    const counts: Record<string, number> = {};
    transactions.forEach(tx => {
      if (tx.customerId && tx.type === 'OUT') {
        counts[tx.customerId] = (counts[tx.customerId] || 0) + 1;
      }
    });
    return counts;
  }, [transactions]);

  // Filter suppliers for import
  const filteredSuppliersForImport = useMemo(() => {
    return suppliers.filter(s => {
      const q = supplierSearchQuery.toLowerCase();
      return s.name.toLowerCase().includes(q) || (s.company && s.company.toLowerCase().includes(q));
    });
  }, [suppliers, supplierSearchQuery]);

  // Calculations for Metrics
  const metrics = useMemo(() => {
    const totalCustomers = customers.length;
    const indebtedCustomers = customers.filter(c => c.balance > 0);
    const totalReceivables = customers.reduce((sum, c) => sum + (c.balance || 0), 0);
    const averageDebt = indebtedCustomers.length > 0 ? totalReceivables / indebtedCustomers.length : 0;
    return {
      totalCustomers,
      indebtedCount: indebtedCustomers.length,
      totalReceivables,
      averageDebt
    };
  }, [customers]);

  // Filter transactions for selected customer
  const customerTransactions = useMemo(() => {
    if (!selectedCustomerId) return [];
    return transactions.filter(tx => tx.customerId === selectedCustomerId);
  }, [transactions, selectedCustomerId]);

  // Form submission
  const handleSubmitCustomer = (e: React.FormEvent) => {
    e.preventDefault();
    if (!custName.trim()) {
      alert(language === 'ar' ? 'اسم العميل مطلوب!' : 'Customer name is required!');
      return;
    }

    const numericBalance = parseFloat(initialDebt) || 0;
    const finalBalance = (balanceType === 'CREDIT' ? -1 : 1) * numericBalance;

    if (editingCustomer) {
      onUpdateCustomer({
        ...editingCustomer,
        name: custName.trim(),
        phone: custPhone.trim() || undefined,
        email: custEmail.trim() || undefined,
        address: custAddress.trim() || undefined,
        balance: finalBalance
      });
    } else {
      onAddCustomer({
        name: custName.trim(),
        phone: custPhone.trim() || undefined,
        email: custEmail.trim() || undefined,
        address: custAddress.trim() || undefined,
        balance: finalBalance
      });
    }

    // Reset Form
    setCustName('');
    setCustPhone('');
    setCustEmail('');
    setCustAddress('');
    setInitialDebt('0');
    setBalanceType('DEBIT');
    setEditingCustomer(null);
    setIsModalOpen(false);
  };

  const getProductName = (id: string) => {
    const p = products.find(prod => prod.id === id);
    return p ? p.name : (language === 'ar' ? 'منتج غير معروف' : 'Unknown Product');
  };

  const selectedCustomer = customers.find(c => c.id === selectedCustomerId);

  return (
    <div className="space-y-6 animate-in fade-in duration-200" dir={isRtl ? 'rtl' : 'ltr'}>
      {/* Guided Onboarding Banner if empty */}
      {customers.length === 0 && (
        <div className="bg-gradient-to-r from-teal-500 via-emerald-500 to-teal-600 rounded-3xl p-6 text-white shadow-md flex flex-col sm:flex-row items-center justify-between gap-5 animate-in fade-in zoom-in-95 duration-200">
          <div className="space-y-1.5 text-center sm:text-start">
            <h4 className="font-black text-base flex items-center justify-center sm:justify-start gap-2">
              <UserPlus className="w-5 h-5 text-emerald-200 animate-pulse" />
              {isRtl ? 'بوابة العملاء والزبائن اللطفاء! 👥' : 'Customers & Clients Portal! 👥'}
            </h4>
            <p className="text-xs text-white/95 font-semibold leading-relaxed max-w-2xl">
              {isRtl 
                ? 'لا يوجد أي عملاء مسجلين حالياً. اضغط على الزر المتوهج في اليسار لتسجيل العميل الأول لمتجرك لتبدأ في طباعة فواتير مبيعات مخصصة باسمائهم وتتبع مديونيات الكاشير بكل بساطة.' 
                : 'Your customer registry is completely empty! Add your first cash or credit client to start generating beautiful personalized invoices.'}
            </p>
          </div>
          <button
            onClick={() => setIsModalOpen(true)}
            className="bg-white text-emerald-600 hover:bg-emerald-50 font-black text-xs px-6 py-3.5 rounded-2xl transition shadow-lg shrink-0 cursor-pointer animate-bounce border-none outline-none"
          >
            {isRtl ? '✨ أضف أول عميل لمتجرك!' : '✨ Add First Customer!'}
          </button>
        </div>
      )}

      {/* Header bar */}
      <div className="bg-white p-5 rounded-2xl border border-slate-200/60 shadow-xs flex flex-col sm:flex-row sm:items-center justify-between gap-4">
        <div>
          <h2 className="text-xl font-black text-slate-800 font-sans tracking-tight">
            {language === 'ar' ? 'إدارة العملاء والديون' : 'Customers & Accounts Receivable'}
          </h2>
          <p className="text-xs text-slate-400 mt-1 font-semibold">
            {language === 'ar' ? 'سجل العملاء الكامل، المديونيات، والدفعات المستحقة للشركة' : 'Manage your client directories, customer ledgers and balances.'}
          </p>
        </div>

        <div className="flex flex-wrap items-center gap-2 shrink-0">
          <button
            onClick={() => setIsModalOpen(true)}
            className="flex items-center justify-center gap-1.5 bg-indigo-600 hover:bg-indigo-700 text-white px-4 py-2.5 rounded-xl text-xs font-bold shadow-md shadow-indigo-600/10 transition cursor-pointer border-none outline-none"
          >
            <UserPlus className="w-4 h-4" />
            {language === 'ar' ? 'إضافة عميل جديد' : 'Register New Customer'}
          </button>

          <button
            onClick={() => setIsImportModalOpen(true)}
            className="flex items-center justify-center gap-1.5 bg-emerald-600 hover:bg-emerald-700 text-white px-4 py-2.5 rounded-xl text-xs font-bold shadow-md shadow-emerald-600/10 transition cursor-pointer border-none outline-none"
          >
            <Download className="w-4 h-4" />
            {language === 'ar' ? 'جلب من الموردين 📥' : 'Import from Suppliers 📥'}
          </button>
        </div>
      </div>

      {/* KPI Cards */}
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
        {/* Metric 1 */}
        <div className="bg-white p-4 rounded-2xl border border-slate-200/60 shadow-xs flex items-center gap-3">
          <div className="p-3 bg-indigo-50 text-indigo-600 rounded-xl">
            <Users className="w-5 h-5" />
          </div>
          <div>
            <span className="text-[10px] text-slate-400 font-bold block uppercase tracking-wider">
              {language === 'ar' ? 'إجمالي العملاء' : 'Total Customers'}
            </span>
            <span className="text-lg font-black text-slate-800 font-sans leading-none mt-1 block">
              {metrics.totalCustomers}
            </span>
          </div>
        </div>

        {/* Metric 2 */}
        <div className="bg-white p-4 rounded-2xl border border-slate-200/60 shadow-xs flex items-center gap-3">
          <div className="p-3 bg-amber-50 text-amber-600 rounded-xl">
            <Wallet className="w-5 h-5" />
          </div>
          <div>
            <span className="text-[10px] text-slate-400 font-bold block uppercase tracking-wider">
              {language === 'ar' ? 'إجمالي الديون المعلقة' : 'Total Outstanding Debt'}
            </span>
            <span className="text-lg font-black text-amber-600 font-mono leading-none mt-1 block">
              {currencySymbol}{metrics.totalReceivables.toFixed(2)}
            </span>
          </div>
        </div>

        {/* Metric 3 */}
        <div className="bg-white p-4 rounded-2xl border border-slate-200/60 shadow-xs flex items-center gap-3">
          <div className="p-3 bg-rose-50 text-rose-600 rounded-xl">
            <TrendingUp className="w-5 h-5" />
          </div>
          <div>
            <span className="text-[10px] text-slate-400 font-bold block uppercase tracking-wider">
              {language === 'ar' ? 'عملاء مدينين' : 'Indebted Clients'}
            </span>
            <span className="text-lg font-black text-slate-800 font-sans leading-none mt-1 block">
              {metrics.indebtedCount}
            </span>
          </div>
        </div>

        {/* Metric 4 */}
        <div className="bg-white p-4 rounded-2xl border border-slate-200/60 shadow-xs flex items-center gap-3">
          <div className="p-3 bg-emerald-50 text-emerald-600 rounded-xl">
            <Percent className="w-5 h-5" />
          </div>
          <div>
            <span className="text-[10px] text-slate-400 font-bold block uppercase tracking-wider">
              {language === 'ar' ? 'متوسط مديونية العميل' : 'Average Balance'}
            </span>
            <span className="text-lg font-black text-emerald-600 font-mono leading-none mt-1 block">
              {currencySymbol}{metrics.averageDebt.toFixed(2)}
            </span>
          </div>
        </div>
      </div>

      {/* Search Input Filter */}
      <div className="bg-white p-4 rounded-2xl border border-slate-200/60 shadow-xs">
        <div className="relative">
          <Search className={`absolute top-2.5 w-4 h-4 text-slate-400 ${isRtl ? 'right-3' : 'left-3'}`} />
          <input 
            type="text" 
            placeholder={language === 'ar' ? 'البحث عن عميل بالاسم، الهاتف، البريد أو العنوان...' : 'Search customers by name, phone, email, address...'}
            value={searchQuery}
            onChange={(e) => setSearchQuery(e.target.value)}
            className={`w-full bg-slate-50 border border-slate-200/60 rounded-xl py-2 text-xs focus:bg-white focus:outline-none focus:ring-1 focus:ring-indigo-500 text-slate-700 ${isRtl ? 'pr-9 pl-3' : 'pl-9 pr-3'}`}
          />
        </div>
      </div>

      {/* Directory Grid & Quick History */}
      <div className={`grid grid-cols-1 ${onOpenCustomer ? 'lg:grid-cols-1' : 'lg:grid-cols-12'} gap-6`}>
        {/* Customers Directory - Left Col */}
        <div className="lg:col-span-8 grid grid-cols-1 md:grid-cols-2 gap-4">
          {filteredCustomers.length === 0 ? (
            <div className="bg-white p-8 rounded-2xl border border-slate-200/60 shadow-xs text-center text-slate-400 col-span-full">
              {t.noData}
            </div>
          ) : (
            filteredCustomers.map(c => {
              const customerSalesCount = customerSalesCounts[c.id] || 0;
              return (
                <div 
                  key={c.id} 
                  onClick={() => {
                    if (onOpenCustomer) {
                      onOpenCustomer(c);
                    } else {
                      setSelectedCustomerId(c.id);
                    }
                  }}
                  className={`bg-white rounded-2xl border transition-all flex flex-col overflow-hidden cursor-pointer ${
                    !onOpenCustomer && selectedCustomerId === c.id 
                      ? 'border-indigo-500 shadow-md ring-1 ring-indigo-500' 
                      : 'border-slate-200/65 shadow-xs hover:shadow-md hover:border-slate-350'
                  }`}
                >
                  {/* Card Header */}
                  <div className="bg-[#1e293b] p-4 flex items-start justify-between text-white">
                    <div className="flex items-center gap-3 overflow-hidden">
                      <div className="p-2.5 bg-slate-800 text-indigo-400 rounded-xl shrink-0">
                        <Users className="w-5 h-5" />
                      </div>
                      <div className="overflow-hidden">
                        <h4 className="font-bold text-slate-100 text-xs truncate leading-tight">{c.name}</h4>
                        <p className="text-[10px] text-slate-300 mt-1 flex items-center gap-1 font-semibold">
                          {language === 'ar' ? 'عدد الفواتير:' : 'Invoices count:'} {customerSalesCount}
                        </p>
                      </div>
                    </div>

                    <div className="flex gap-1" onClick={(e) => e.stopPropagation()}>
                      {/* Edit button */}
                      <button
                        onClick={() => {
                          setEditingCustomer(c);
                          setCustName(c.name);
                          setCustPhone(c.phone || '');
                          setCustEmail(c.email || '');
                          setCustAddress(c.address || '');
                          setInitialDebt(Math.abs(c.balance).toString());
                          setBalanceType(c.balance < 0 ? 'CREDIT' : 'DEBIT');
                          setIsModalOpen(true);
                        }}
                        className="text-slate-400 hover:text-indigo-400 p-1.5 rounded-lg hover:bg-slate-800 transition shrink-0 cursor-pointer"
                        title={language === 'ar' ? 'تعديل بيانات العميل' : 'Edit Customer'}
                      >
                        <Edit2 className="w-4 h-4" />
                      </button>

                      {/* Delete button */}
                      <button
                        onClick={() => {
                          if (window.confirm(language === 'ar' ? 'هل تريد حذف هذا العميل؟' : 'Are you sure you want to delete this customer?')) {
                            onDeleteCustomer(c.id);
                            if (selectedCustomerId === c.id) setSelectedCustomerId(null);
                          }
                        }}
                        className="text-slate-400 hover:text-red-400 p-1.5 rounded-lg hover:bg-slate-800 transition shrink-0 cursor-pointer"
                        title={language === 'ar' ? 'حذف العميل' : 'Delete Customer'}
                      >
                        <Trash2 className="w-4 h-4" />
                      </button>
                    </div>
                  </div>

                  {/* Body details */}
                  <div className="p-4 space-y-2.5 flex-1 text-[11px]">
                    {/* Phone */}
                    {c.phone && (
                      <div className="flex items-center gap-2 text-slate-655 font-mono">
                        <Phone className="w-3.5 h-3.5 text-indigo-500 shrink-0" />
                        <span>{c.phone}</span>
                      </div>
                    )}

                    {/* Email */}
                    {c.email && (
                      <div className="flex items-center gap-2 text-slate-655 font-mono truncate">
                        <Mail className="w-3.5 h-3.5 text-indigo-500 shrink-0" />
                        <span className="truncate">{c.email}</span>
                      </div>
                    )}

                    {/* Address */}
                    {c.address && (
                      <div className="flex items-start gap-2 text-slate-600 font-medium leading-relaxed">
                        <MapPin className="w-3.5 h-3.5 text-indigo-500 shrink-0 mt-0.5" />
                        <span>{c.address}</span>
                      </div>
                    )}

                    {/* Debt Display */}
                    <div className="pt-2 border-t border-slate-100 mt-3 flex items-center justify-between">
                      <span className="text-[10px] font-bold text-slate-400 uppercase">
                        {language === 'ar' ? 'الرصيد المالي' : 'Financial Balance'}
                      </span>
                      {c.balance > 0 ? (
                        <span className="px-2 py-0.5 rounded text-xs font-mono font-bold bg-amber-100 text-amber-800">
                          {language === 'ar' ? 'مدين (مطلوب منه):' : 'Due:'} {currencySymbol}{c.balance.toFixed(2)}
                        </span>
                      ) : c.balance < 0 ? (
                        <span className="px-2 py-0.5 rounded text-xs font-mono font-bold bg-emerald-100 text-emerald-800">
                          {language === 'ar' ? 'دائن (له رصيد زائد):' : 'Credit:'} {currencySymbol}{Math.abs(c.balance).toFixed(2)}
                        </span>
                      ) : (
                        <span className="px-2 py-0.5 rounded text-xs font-mono font-bold bg-slate-100 text-slate-500">
                          {currencySymbol}0.00
                        </span>
                      )}
                    </div>
                  </div>
                </div>
              );
            })
          )}
        </div>

        {!onOpenCustomer && (<>
        {/* Ledger - Right Col */}
        <div className="lg:col-span-4">
          <div className="bg-white p-5 rounded-2xl border border-slate-200/60 shadow-xs space-y-4 sticky top-4">
            <div className="flex items-center gap-2 border-b pb-3">
              <History className="w-5 h-5 text-indigo-600" />
              <h3 className="font-bold text-slate-800 text-xs">
                {language === 'ar' ? 'دفتر الأستاذ للعميل' : 'Customer Account Ledger'}
              </h3>
            </div>

            {!selectedCustomerId ? (
              <div className="py-12 text-center text-slate-400 text-xs">
                {language === 'ar' ? 'اختر عميلاً من القائمة لعرض كشف حسابه وحركاته بالتفصيل' : 'Select a customer from the directory to view transaction ledgers and audit records.'}
              </div>
            ) : (
              <div className="space-y-4">
                <div className="p-3 bg-slate-50 rounded-xl border border-slate-150 space-y-2">
                  <div className="flex justify-between items-center">
                    <p className="font-black text-slate-800 text-xs">{selectedCustomer?.name}</p>
                    <div className="flex gap-1">
                      {/* Print Ledger */}
                      <button
                        onClick={() => {
                          if (!selectedCustomer) return;
                          const listHtml = customerTransactions.map(tx => `
                            <tr>
                              <td style="padding: 8px; border-bottom: 1px solid #e2e8f0;">${new Date(tx.date).toLocaleDateString()}</td>
                              <td style="padding: 8px; border-bottom: 1px solid #e2e8f0; font-weight: bold;">${getProductName(tx.productId)}</td>
                              <td style="padding: 8px; border-bottom: 1px solid #e2e8f0;">${tx.type === 'OUT' ? 'Sale' : 'Return'}</td>
                              <td style="padding: 8px; border-bottom: 1px solid #e2e8f0; text-align: center;">${tx.quantity}</td>
                              <td style="padding: 8px; border-bottom: 1px solid #e2e8f0; text-align: right;">${currencySymbol}${tx.unitPrice.toFixed(2)}</td>
                              <td style="padding: 8px; border-bottom: 1px solid #e2e8f0; text-align: right; font-weight: bold;">${currencySymbol}${(tx.quantity * tx.unitPrice).toFixed(2)}</td>
                            </tr>
                          `).join('');

                          const htmlString = `
                            <html>
                              <head>
                                <title>Customer Ledger - ${selectedCustomer.name}</title>
                                <style>
                                  @page {
                                    size: A4 portrait;
                                    margin: 10mm;
                                  }
                                  body { 
                                    font-family: sans-serif; 
                                    padding: 20px; 
                                    color: #1e293b; 
                                    direction: ${isRtl ? 'rtl' : 'ltr'}; 
                                    background: #fff;
                                  }
                                  table { width: 100%; border-collapse: collapse; margin-top: 20px; }
                                  th { background-color: #f1f5f9; padding: 10px; text-align: left; font-weight: bold; }
                                  td { padding: 10px; }
                                  .header { border-bottom: 2px solid #e2e8f0; padding-bottom: 15px; margin-bottom: 20px; }
                                  .total-box { background-color: #f8fafc; padding: 15px; border-radius: 8px; margin-top: 20px; text-align: right; font-weight: bold; border: 1px solid #e2e8f0; }
                                  
                                  @media print {
                                    html, body {
                                      margin: 0 !important;
                                      padding: 0 !important;
                                      height: auto !important;
                                      background: #fff !important;
                                    }
                                    body {
                                      padding: 10mm !important;
                                    }
                                  }
                                </style>
                              </head>
                              <body>
                                <div class="header">
                                  <h2>Customer Account Statement</h2>
                                  <p><strong>Customer Name:</strong> ${selectedCustomer.name}</p>
                                  <p><strong>Phone:</strong> ${selectedCustomer.phone || '-'}</p>
                                  <p><strong>Email:</strong> ${selectedCustomer.email || '-'}</p>
                                </div>
                                <table>
                                  <thead>
                                    <tr>
                                      <th>Date</th>
                                      <th>Product</th>
                                      <th>Type</th>
                                      <th style="text-align: center;">Qty</th>
                                      <th style="text-align: right;">Unit Price</th>
                                      <th style="text-align: right;">Amount</th>
                                    </tr>
                                  </thead>
                                  <tbody>
                                    ${listHtml || '<tr><td colspan="6" style="text-align: center; color: #94a3b8; font-style: italic;">No transactions recorded.</td></tr>'}
                                  </tbody>
                                </table>
                                <div class="total-box">
                                  Outstanding Due Balance: ${currencySymbol}${selectedCustomer.balance.toFixed(2)}
                                </div>
                              </body>
                            </html>
                          `;
                          printHtml(htmlString, 'a4', language === 'ar' ? 'كشف حساب العميل' : 'Customer Account Statement');
                        }}
                        className="p-1 text-slate-500 hover:text-indigo-600 hover:bg-slate-200/50 rounded transition cursor-pointer"
                        title={language === 'ar' ? 'طباعة كشف الحساب بالكامل' : 'Print Account Statement'}
                      >
                        <Printer className="w-3.5 h-3.5" />
                      </button>

                      {/* Share Ledger */}
                      <button
                        onClick={() => {
                          if (!selectedCustomer) return;
                          const logLines = customerTransactions.map(tx => `• ${new Date(tx.date).toLocaleDateString()} | ${getProductName(tx.productId)} | ${tx.type === 'OUT' ? 'Sale' : 'Return'} | ${tx.quantity} pcs @ ${currencySymbol}${tx.unitPrice} = ${currencySymbol}${(tx.quantity * tx.unitPrice).toFixed(2)}`).join('\n');
                          const shareText = `📋 [Account Statement] ${selectedCustomer.name}\nOutstanding Receivable Debt: ${currencySymbol}${selectedCustomer.balance.toFixed(2)}\n\nRecent Ledger Items:\n${logLines || 'No transactions'}`;
                          navigator.clipboard.writeText(shareText);
                          alert(language === 'ar' ? 'تم نسخ كشف الحساب الكامل للحافظة!' : 'Full statement ledger copied to clipboard!');
                        }}
                        className="p-1 text-slate-500 hover:text-indigo-600 hover:bg-slate-200/50 rounded transition cursor-pointer"
                        title={language === 'ar' ? 'مشاركة كشف الحساب' : 'Share Account Statement'}
                      >
                        <Share2 className="w-3.5 h-3.5" />
                      </button>
                    </div>
                  </div>
                  <div className="flex justify-between items-center text-[11px]">
                    <span className="text-slate-500">{language === 'ar' ? 'الرصيد المدين:' : 'Receivables Debit Balance:'}</span>
                    <span className="font-mono font-black text-amber-600">{currencySymbol}{selectedCustomer?.balance.toFixed(2)}</span>
                  </div>
                </div>

                <div className="space-y-2 max-h-[300px] overflow-y-auto pr-1">
                  <p className="text-[10px] font-bold text-slate-400 uppercase tracking-wider">
                    {language === 'ar' ? 'سجل العمليات والطلبيات' : 'Transaction Logs'} ({customerTransactions.length})
                  </p>

                  {customerTransactions.length === 0 ? (
                    <p className="text-[11px] text-slate-350 italic text-center py-6">{language === 'ar' ? 'لا يوجد عمليات مسجلة لهذا العميل' : 'No recorded transactions found for this customer'}</p>
                  ) : (
                    customerTransactions.map((tx) => (
                      <div key={tx.id} className="p-2.5 bg-white border border-slate-100 hover:border-slate-200 rounded-xl transition text-[10px] space-y-1">
                        <div className="flex justify-between items-center">
                          <span className="font-bold text-slate-700 max-w-[120px] truncate">{getProductName(tx.productId)}</span>
                          <span className={`px-1.5 py-0.5 rounded-sm font-bold uppercase text-[8px] ${
                            tx.type === 'OUT' ? 'bg-red-50 text-red-600' : 'bg-emerald-50 text-emerald-600'
                          }`}>
                            {tx.type === 'OUT' ? (language === 'ar' ? 'مبيعات' : 'Sale') : (language === 'ar' ? 'مرتجع' : 'Return')}
                          </span>
                        </div>
                        <div className="flex justify-between text-slate-450 font-mono">
                          <span>{tx.quantity} pcs @ {currencySymbol}{tx.unitPrice.toFixed(2)}</span>
                          <span>Total: {currencySymbol}{(tx.quantity * tx.unitPrice).toFixed(2)}</span>
                        </div>
                        <p className="text-[9px] text-slate-400 truncate">{tx.notes || '-'}</p>
                        <p className="text-[8px] text-slate-350 text-right">{new Date(tx.date).toLocaleDateString()}</p>
                      </div>
                    ))
                  )}
                </div>
              </div>
            )}
          </div>
        </div>
        </>)}
      </div>

      {/* --- ADD CUSTOMER FORM MODAL --- */}
      {isModalOpen && (
        <div className="fixed inset-0 bg-slate-900/60 backdrop-blur-xs flex items-center justify-center p-4 z-50 animate-in fade-in duration-200">
          <div className="bg-white rounded-2xl shadow-xl w-full max-w-md overflow-hidden border border-slate-200" dir={isRtl ? 'rtl' : 'ltr'}>
            <div className="flex items-center justify-between p-4 border-b border-slate-100 bg-slate-50">
              <h3 className="font-bold text-slate-800 text-xs">
                {editingCustomer
                  ? (language === 'ar' ? 'تعديل بيانات العميل 📝' : 'Edit Customer Account Details 📝')
                  : (language === 'ar' ? 'تسجيل عميل جديد بالشركة' : 'Register New Customer Account')}
              </h3>
              <button 
                onClick={() => {
                  setEditingCustomer(null);
                  setCustName('');
                  setCustPhone('');
                  setCustEmail('');
                  setCustAddress('');
                  setInitialDebt('0');
                  setIsModalOpen(false);
                }} 
                className="text-slate-400 hover:text-slate-600 cursor-pointer"
              >
                <X className="w-5 h-5" />
              </button>
            </div>

            <form onSubmit={handleSubmitCustomer} className="p-5 space-y-4 text-xs">
              {/* Customer Name */}
              <div>
                <label className="block text-[11px] font-bold text-slate-655 mb-1">{language === 'ar' ? 'اسم العميل الكامل *' : 'Customer Full Name *'}</label>
                <input 
                  type="text" 
                  required
                  placeholder="e.g. Ahmed Al-Otaibi"
                  value={custName}
                  onChange={(e) => setCustName(e.target.value)}
                  className="w-full bg-slate-50 border border-slate-200 rounded-xl py-2 px-3 focus:bg-white focus:outline-none focus:ring-1 focus:ring-indigo-500 font-semibold text-slate-800"
                />
              </div>

              {/* Contact Phone */}
              <div>
                <label className="block text-[11px] font-bold text-slate-655 mb-1">{language === 'ar' ? 'رقم الهاتف / الجوال' : 'Contact Phone'}</label>
                <input 
                  type="text" 
                  placeholder="+966..."
                  value={custPhone}
                  onChange={(e) => setCustPhone(e.target.value)}
                  className="w-full bg-slate-50 border border-slate-200 rounded-xl py-2 px-3 focus:bg-white focus:outline-none focus:ring-1 focus:ring-indigo-500 font-mono text-slate-800"
                />
              </div>

              {/* Email */}
              <div>
                <label className="block text-[11px] font-bold text-slate-655 mb-1">{language === 'ar' ? 'البريد الإلكتروني' : 'Email Address'}</label>
                <input 
                  type="email" 
                  placeholder="name@company.com"
                  value={custEmail}
                  onChange={(e) => setCustEmail(e.target.value)}
                  className="w-full bg-slate-50 border border-slate-200 rounded-xl py-2 px-3 focus:bg-white focus:outline-none focus:ring-1 focus:ring-indigo-500 font-mono text-slate-800"
                />
              </div>

              {/* Location Address */}
              <div>
                <label className="block text-[11px] font-bold text-slate-655 mb-1">{language === 'ar' ? 'العنوان الجغرافي' : 'Location Address'}</label>
                <input 
                  type="text" 
                  placeholder="Olaya, Riyadh, KSA"
                  value={custAddress}
                  onChange={(e) => setCustAddress(e.target.value)}
                  className="w-full bg-slate-50 border border-slate-200 rounded-xl py-2 px-3 focus:bg-white focus:outline-none focus:ring-1 focus:ring-indigo-500 text-slate-800"
                />
              </div>

              {/* Initial Debt with Type (Debit/Credit) */}
              <div className="grid grid-cols-1 sm:grid-cols-2 gap-3">
                <div>
                  <label className="block text-[11px] font-bold text-slate-655 mb-1">
                    {editingCustomer
                      ? (language === 'ar' ? `الرصيد المالي (${currencySymbol})` : `Financial Balance (${currencySymbol})`)
                      : (language === 'ar' ? `الرصيد الافتتاحي (${currencySymbol})` : `Opening Balance (${currencySymbol})`)}
                  </label>
                  <div className="relative">
                    <span className={`absolute top-2 text-slate-400 font-bold ${isRtl ? 'left-3' : 'right-3'}`}>{currencySymbol}</span>
                    <input 
                      type="number" 
                      min="0" 
                      step="0.01"
                      value={initialDebt}
                      onChange={(e) => setInitialDebt(e.target.value)}
                      className="w-full bg-slate-50 border border-slate-200 rounded-xl py-2 px-3 focus:bg-white focus:outline-none focus:ring-1 focus:ring-indigo-500 font-bold text-slate-800"
                    />
                  </div>
                </div>

                <div>
                  <label className="block text-[11px] font-bold text-slate-655 mb-1">
                    {language === 'ar' ? 'طبيعة الرصيد' : 'Balance Type'}
                  </label>
                  <select
                    value={balanceType}
                    onChange={(e) => setBalanceType(e.target.value as 'DEBIT' | 'CREDIT')}
                    className="w-full bg-slate-50 border border-slate-200 rounded-xl py-2 px-3 focus:bg-white focus:outline-none focus:ring-1 focus:ring-indigo-500 font-bold text-slate-700"
                  >
                    <option value="DEBIT">
                      {language === 'ar' ? 'مدين (عليه دين للمحل)' : 'Debit (Owes us)'}
                    </option>
                    <option value="CREDIT">
                      {language === 'ar' ? 'دائن (له رصيد زائد)' : 'Credit (We owe them)'}
                    </option>
                  </select>
                </div>
              </div>

              <button
                type="submit"
                className="w-full bg-indigo-600 hover:bg-indigo-700 text-white font-bold py-2.5 px-4 rounded-xl shadow-md transition cursor-pointer"
              >
                {editingCustomer
                  ? (language === 'ar' ? 'حفظ التعديلات' : 'Save Changes')
                  : (language === 'ar' ? 'إضافة وتثبيت العميل الجديد' : 'Confirm Customer Onboarding')}
              </button>
            </form>
          </div>
        </div>
      )}

      {/* Import from Suppliers Modal */}
      {isImportModalOpen && (
        <div className="fixed inset-0 bg-slate-900/60 backdrop-blur-xs flex items-center justify-center p-4 z-50 animate-in fade-in duration-200">
          <div className="bg-white rounded-3xl max-w-lg w-full overflow-hidden shadow-2xl border border-slate-100 flex flex-col max-h-[85vh] animate-in zoom-in-95 duration-200">
            {/* Header */}
            <div className="bg-slate-50 border-b border-slate-100 p-5 flex items-center justify-between">
              <div>
                <h3 className="font-extrabold text-slate-900 text-sm flex items-center gap-2">
                  <Download className="w-5 h-5 text-emerald-600 animate-pulse" />
                  {language === 'ar' ? 'جلب واستيراد من سجل الموردين' : 'Import from Suppliers Directory'}
                </h3>
                <p className="text-[10px] text-slate-450 mt-1 font-semibold">
                  {language === 'ar' 
                    ? 'اختر أي مورد حالي لتحويله وتكرار بياناته كعميل في متجرك بضغطة زر واحدة.' 
                    : 'Select any registered supplier to import and clone their profile as a customer.'}
                </p>
              </div>
              <button 
                onClick={() => {
                  setIsImportModalOpen(false);
                  setSupplierSearchQuery('');
                }} 
                className="text-slate-400 hover:text-slate-600 cursor-pointer border-none bg-transparent outline-none"
              >
                <X className="w-5 h-5" />
              </button>
            </div>

            {/* Search Input */}
            <div className="p-4 border-b border-slate-50 bg-slate-50/50">
              <div className="relative">
                <Search className="w-4 h-4 text-slate-450 absolute left-3 top-3.5" />
                <input
                  type="text"
                  placeholder={language === 'ar' ? 'بحث باسم المورد أو الشركة...' : 'Search supplier name or company...'}
                  value={supplierSearchQuery}
                  onChange={(e) => setSupplierSearchQuery(e.target.value)}
                  className="w-full bg-white border border-slate-200 rounded-xl py-2.5 pl-9 pr-4 text-xs font-semibold focus:outline-none focus:ring-1 focus:ring-emerald-500 text-slate-800"
                />
              </div>
            </div>

            {/* Suppliers List */}
            <div className="p-4 overflow-y-auto flex-1 space-y-2 max-h-[45vh]">
              {suppliers.length === 0 ? (
                <div className="text-center py-8">
                  <p className="text-slate-400 text-xs font-bold">
                    {language === 'ar' ? 'لا يوجد أي موردين مسجلين في النظام حالياً!' : 'No suppliers registered in the system yet!'}
                  </p>
                </div>
              ) : filteredSuppliersForImport.length === 0 ? (
                <div className="text-center py-8">
                  <p className="text-slate-400 text-xs font-bold">
                    {language === 'ar' ? 'لا توجد نتائج تطابق البحث!' : 'No suppliers match your search!'}
                  </p>
                </div>
              ) : (
                filteredSuppliersForImport.map(sup => {
                  const alreadyExists = customers.some(c => c.name.trim().toLowerCase() === sup.name.trim().toLowerCase());
                  return (
                    <div 
                      key={sup.id} 
                      className={`flex items-center justify-between p-3.5 rounded-2xl border transition ${
                        alreadyExists 
                          ? 'bg-slate-50/70 border-slate-100 opacity-60' 
                          : 'bg-white border-slate-150 hover:border-emerald-300 hover:shadow-xs'
                      }`}
                    >
                      <div className="space-y-1 min-w-0 flex-1">
                        <div className="flex items-center gap-1.5 flex-wrap">
                          <p className="font-extrabold text-slate-850 text-xs truncate">
                            {sup.name}
                          </p>
                          {sup.company && (
                            <span className="bg-slate-100 text-slate-600 text-[9px] px-1.5 py-0.5 rounded font-bold">
                              {sup.company}
                            </span>
                          )}
                          {alreadyExists && (
                            <span className="bg-amber-50 text-amber-700 text-[8px] px-1.5 py-0.5 rounded font-black uppercase">
                              {language === 'ar' ? 'موجود كعميل' : 'Already Customer'}
                            </span>
                          )}
                        </div>
                        <div className="flex items-center gap-3 text-[10px] text-slate-450 font-semibold">
                          {sup.phone && <span className="flex items-center gap-1">📞 {sup.phone}</span>}
                          {sup.address && <span className="flex items-center gap-1">📍 {sup.address}</span>}
                        </div>
                      </div>

                      <button
                        type="button"
                        onClick={() => {
                          if (alreadyExists) {
                            if (!window.confirm(language === 'ar' ? 'هذا الاسم مسجل بالفعل كعميل. هل تريد إضافته مرة أخرى على أي حال؟' : 'This name is already in customers list. Add anyway?')) {
                              return;
                            }
                          }
                          onAddCustomer({
                            name: sup.name,
                            phone: sup.phone,
                            email: sup.email,
                            address: sup.address,
                            balance: 0
                          });
                          setIsImportModalOpen(false);
                          setSupplierSearchQuery('');
                        }}
                        className={`px-3 py-1.5 rounded-xl font-bold text-[10px] cursor-pointer transition border-none outline-none ${
                          alreadyExists
                            ? 'bg-amber-100 hover:bg-amber-200 text-amber-800'
                            : 'bg-emerald-600 hover:bg-emerald-700 text-white shadow-xs'
                        }`}
                      >
                        {language === 'ar' ? 'جلب واستيراد' : 'Import Customer'}
                      </button>
                    </div>
                  );
                })
              )}
            </div>

            {/* Footer */}
            <div className="bg-slate-50 p-4 border-t border-slate-100 flex justify-end">
              <button
                type="button"
                onClick={() => {
                  setIsImportModalOpen(false);
                  setSupplierSearchQuery('');
                }}
                className="bg-slate-200 hover:bg-slate-300 text-slate-700 text-xs font-bold px-4 py-2.5 rounded-xl transition cursor-pointer border-none outline-none"
              >
                {language === 'ar' ? 'إغلاق النافذة' : 'Close'}
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
