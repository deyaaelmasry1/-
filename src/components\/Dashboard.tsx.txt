          <div className="p-3 bg-violet-50/70 text-violet-650 rounded-xl">
            <TrendingUp className="w-6 h-6" />
          </div>
        </div>

        {/* Card 5: Expected Profits */}
        <div className="bg-white p-5 rounded-2xl border border-slate-200/60 shadow-xs flex items-center justify-between">
          <div className="space-y-1">
            <p className="text-xs text-slate-400 font-bold uppercase tracking-wider">{t.expectedProfit}</p>
            <p className="text-3xl font-black text-emerald-600 font-mono">{currencySymbol} {metrics.expectedProfit.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 })}</p>
          </div>
          <div className="p-3 bg-teal-50/70 text-teal-650 rounded-xl">
            <TrendingUp className="w-6 h-6" />
          </div>
        </div>
      </div>

      {/* Comprehensive ERP Quick Access Bento Grid Section */}
      {quickAccessBentoGrid}

      {/* Visual Analytics Charts Section */}
      <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
        {/* Transaction Flow Area Chart */}
        <div className="bg-white p-5 rounded-2xl border border-slate-200/60 shadow-xs lg:col-span-2 space-y-4">
          <div className="flex items-center justify-between border-b border-slate-100 pb-3">
            <h3 className="font-bold text-slate-800 text-sm flex items-center gap-2">
              <ClipboardList className="w-4 h-4 text-indigo-500" />
              {t.recentTransactions} (7 {language === 'ar' ? 'أيام الأخيرة' : 'Days Flow'})
            </h3>
            <div className="flex gap-4 text-xs font-semibold">
              <span className="flex items-center gap-1.5 text-emerald-600">
                <span className="w-2.5 h-2.5 rounded-full bg-emerald-500 inline-block"></span>
                {t.stockIn}
              </span>
              <span className="flex items-center gap-1.5 text-amber-600">
                <span className="w-2.5 h-2.5 rounded-full bg-amber-500 inline-block"></span>
                {t.stockOut}
              </span>
            </div>
          </div>
          <div className="h-72 w-full">
            <ResponsiveContainer width="100%" height="100%">
              <AreaChart data={transactionChartData}>
                <defs>
                  <linearGradient id="colorIncoming" x1="0" y1="0" x2="0" y2="1">
                    <stop offset="5%" stopColor="#10b981" stopOpacity={0.2}/>
                    <stop offset="95%" stopColor="#10b981" stopOpacity={0}/>
                  </linearGradient>
                  <linearGradient id="colorOutgoing" x1="0" y1="0" x2="0" y2="1">
                    <stop offset="5%" stopColor="#f59e0b" stopOpacity={0.2}/>
                    <stop offset="95%" stopColor="#f59e0b" stopOpacity={0}/>
                  </linearGradient>
                </defs>
                <CartesianGrid strokeDasharray="3 3" vertical={false} stroke="#f1f5f9" />
                <XAxis 
                  dataKey="date" 
                  stroke="#94a3b8" 
                  fontSize={10} 
                  tickLine={false} 
                  tickFormatter={(tick) => {
                    const parts = tick.split('-');
                    return `${parts[1]}/${parts[2]}`;
                  }}
                />
                <YAxis stroke="#94a3b8" fontSize={10} tickLine={false} />
                <Tooltip 
                  contentStyle={{ background: '#fff', border: '1px solid #e2e8f0', borderRadius: '8px', fontSize: '11px' }}
                  labelFormatter={(label) => `${t.txDate}: ${label}`}
                />
                <Area type="monotone" name={t.stockIn} dataKey="incoming" stroke="#10b981" strokeWidth={2} fillOpacity={1} fill="url(#colorIncoming)" />
                <Area type="monotone" name={t.stockOut} dataKey="outgoing" stroke="#f59e0b" strokeWidth={2} fillOpacity={1} fill="url(#colorOutgoing)" />
              </AreaChart>
            </ResponsiveContainer>
          </div>
        </div>

        {/* Category distribution Pie Chart */}
        <div className="bg-white p-5 rounded-2xl border border-slate-200/60 shadow-xs space-y-4">
          <div className="border-b border-slate-100 pb-3">
            <h3 className="font-bold text-slate-800 text-sm">
              {language === 'ar' ? 'توزيع الكميات حسب الفئة' : 'Stock by Category'}
            </h3>
          </div>
          <div className="h-60 w-full relative flex items-center justify-center">
            {categoryData.length === 0 ? (
              <p className="text-slate-400 text-xs font-sans text-center">{t.noData}</p>
            ) : (
              <ResponsiveContainer width="100%" height="100%">
                <PieChart>
                  <Pie
                    data={categoryData}
                    cx="50%"
                    cy="50%"
                    innerRadius={60}
                    outerRadius={80}
                    paddingAngle={3}
                    dataKey="value"
                  >
                    {categoryData.map((entry, index) => (
                      <Cell key={`cell-${index}`} fill={COLORS[index % COLORS.length]} />
                    ))}
                  </Pie>
                  <Tooltip formatter={(value) => [`${value} units`, 'Quantity']} />
                </PieChart>
              </ResponsiveContainer>
            )}
            {/* Center label */}
            <div className="absolute flex flex-col items-center justify-center">
              <span className="text-[10px] uppercase font-bold text-slate-400">{language === 'ar' ? 'إجمالي القطع' : 'Total Units'}</span>
              <span className="text-2xl font-black text-slate-800 font-mono leading-none mt-1">{metrics.totalQty}</span>
            </div>
          </div>
          {/* Legend */}
          <div className="grid grid-cols-2 gap-2 text-xs">
            {categoryData.map((item, index) => (
              <div key={item.name} className="flex items-center gap-1.5 text-slate-600 truncate font-semibold">
                <span 
                  className="w-2.5 h-2.5 rounded-full shrink-0" 
                  style={{ backgroundColor: COLORS[index % COLORS.length] }}
                ></span>
                <span className="truncate">{item.name}</span>
                <span className="font-mono text-[10px] text-slate-450">({item.value})</span>
              </div>
            ))}
          </div>
        </div>
      </div>

      {/* Low Stock Notifications Alert Modal Overlay */}
      {showLowStockModal && (
        <div className="fixed inset-0 bg-slate-900/60 backdrop-blur-xs z-50 flex items-center justify-center p-4" dir={isRtl ? 'rtl' : 'ltr'}>
          <div className="bg-white rounded-2xl border border-slate-200 shadow-2xl w-full max-w-2xl max-h-[85vh] overflow-y-auto flex flex-col animate-in fade-in zoom-in-95 duration-150">
            {/* Modal Header */}
            <div className="p-5 border-b border-slate-100 flex items-center justify-between bg-slate-50 rounded-t-2xl">
              <div className="flex items-center gap-2">
                <Bell className="w-5 h-5 text-amber-500 animate-bounce" />
                <div>
                  <h3 className="font-extrabold text-slate-800 text-base">{isRtl ? 'جرس تنبيهات المخزون المنخفض' : 'Low Stock Notifications & Alerts'}</h3>
                  <p className="text-[10px] text-slate-450 font-bold">{isRtl ? 'إدارة النواقص والطلب السريع والربط بالمستودعات' : 'Manage stock shortages & quick restock orders mapped to warehouses'}</p>
                </div>
              </div>
              <button 
                onClick={() => { setShowLowStockModal(false); setSelectedRestockProd(null); }}
                className="text-slate-400 hover:text-slate-600 font-bold text-lg p-1.5 hover:bg-slate-150 rounded-xl transition cursor-pointer"
              >
                ✕
              </button>
            </div>

            {/* Modal Body */}
            <div className="p-5 overflow-y-auto flex-1 space-y-4">
              {products.filter(p => p.quantity <= p.minQuantity).length === 0 ? (
                <div className="text-center py-10 space-y-3">
                  <span className="text-4xl">🎉</span>
                  <p className="text-slate-800 font-extrabold text-sm">{isRtl ? 'المستودعات آمنة بنسبة 100%!' : 'Warehouses are 100% safe!'}</p>
                  <p className="text-slate-450 text-xs font-semibold">{isRtl ? 'جميع المنتجات كمياتها ممتازة وفوق حدود الأمان والتنبيه.' : 'All products have adequate stock levels above warning limits.'}</p>
                </div>
              ) : (
                <div className="space-y-3">
                  <p className="text-xs text-slate-500 font-semibold">{isRtl ? 'المنتجات التالية انخفضت كميتها عن حد الأمان بالكامل:' : 'The following products are running low on stock:'}</p>
                  
                  {products.filter(p => p.quantity <= p.minQuantity).map(p => {
                    const wh = warehouses.find(w => w.id === p.warehouseId);
                    const sup = suppliers.find(s => s.id === p.supplierId);
                    return (
                      <div key={p.id} className="p-4 bg-slate-50 hover:bg-slate-100/70 rounded-xl border border-slate-200/80 transition flex flex-col md:flex-row md:items-center justify-between gap-3">
                        <div className="space-y-1">
                          <div className="flex items-center gap-2">
                            <span className="w-2.5 h-2.5 rounded-full bg-rose-500"></span>
                            <span className="font-extrabold text-slate-800 text-sm">{p.name}</span>
                            <span className="text-[10px] bg-slate-200 text-slate-700 font-mono px-1.5 py-0.5 rounded">SKU: {p.code}</span>
                          </div>
                          
                          <div className="text-xs text-slate-500 font-medium space-y-0.5">
                            <p className="flex items-center gap-1">
                              <span className="font-bold">{isRtl ? 'المستودع المرتبط به:' : 'Associated Warehouse:'}</span> 
                              <span className="text-indigo-600 font-bold">{wh ? wh.name : (isRtl ? 'غير محدد' : 'Not Specified')}</span>
                              {wh?.location && <span className="text-slate-400">({wh.location})</span>}
                            </p>
                            <p className="flex items-center gap-1">
                              <span className="font-bold">{isRtl ? 'المورد المعتمد:' : 'Preferred Supplier:'}</span>
                              <span className="text-slate-700">{sup ? sup.name : (isRtl ? 'غير محدد' : 'Not Specified')}</span>
                            </p>
                          </div>
                        </div>

                        <div className="flex items-center justify-between md:justify-end gap-4 border-t md:border-t-0 pt-2 md:pt-0">
                          <div className="text-right">
                            <p className="text-xs font-bold text-rose-600 font-mono">
                              {isRtl ? 'الكمية الحالية:' : 'Current Qty:'} {p.quantity} / {isRtl ? 'الحد الأدنى:' : 'Min Limit:'} {p.minQuantity}
                            </p>
                            <span className="inline-block px-1.5 py-0.5 text-[9px] font-black rounded bg-rose-100 text-rose-700 mt-0.5">
                              {p.quantity === 0 ? (isRtl ? 'نفذت الكمية 🚫' : 'Out of Stock 🚫') : (isRtl ? 'مخزون منخفض ⚠️' : 'Low Stock ⚠️')}
                            </span>
                          </div>

                          <button
                            onClick={() => {
                              setSelectedRestockProd(p);
                              setRestockQty(Math.max(1, p.minQuantity * 2 - p.quantity));
                              setRestockCost(p.costPrice);
                              setRestockSupplierId(p.supplierId || (suppliers[0]?.id || ''));
                              setRestockWarehouseId(p.warehouseId || (warehouses[0]?.id || ''));
                            }}
                            className="bg-indigo-600 hover:bg-indigo-700 text-white font-bold text-xs px-3.5 py-2 rounded-xl transition cursor-pointer"
                          >
                            {isRtl ? 'اتخاذ إجراء 🛠️' : 'Take Action 🛠️'}
                          </button>
                        </div>
                      </div>
                    );
                  })}
                </div>
              )}

              {/* Subform: Action/Quick Restock */}
              {selectedRestockProd && (
                <form onSubmit={handleQuickRestock} className="mt-6 border-t border-indigo-100 pt-5 space-y-4 bg-indigo-50/40 p-4 rounded-xl border border-indigo-50 animate-in slide-in-from-bottom duration-150">
                  <div className="flex items-center justify-between">
                    <h4 className="font-bold text-xs text-indigo-950 flex items-center gap-1.5">
                      <Package className="w-4 h-4 text-indigo-500" />
                      {isRtl ? `طلب تمويل مستودع مستعجل للسلعة: ${selectedRestockProd.name}` : `Emergency Stock Funding for: ${selectedRestockProd.name}`}
                    </h4>
                    <button 
                      type="button"
                      onClick={() => setSelectedRestockProd(null)}
                      className="text-indigo-650 hover:text-indigo-800 text-xs font-bold animate-pulse"
                    >
                      {isRtl ? 'إلغاء الإجراء' : 'Cancel Action'}
                    </button>
                  </div>

                  <div className="grid grid-cols-1 md:grid-cols-2 gap-3 text-xs">
                    <div>
                      <label className="block text-[11px] font-bold text-slate-650 mb-1">{isRtl ? 'الكمية المشتراة المراد إضافتها للمستودع *' : 'Purchase Qty to Add *'}</label>
                      <input 
                        type="number" 
                        required
                        min="1"
                        value={restockQty}
                        onChange={(e) => setRestockQty(Math.max(1, Number(e.target.value)))}
                        className="w-full bg-white border border-slate-200 rounded-xl py-2 px-3 font-bold text-slate-800"
                      />
                    </div>

                    <div>
                      <label className="block text-[11px] font-bold text-slate-650 mb-1">{isRtl ? `تكلفة شراء القطعة الواحد (${currencySymbol}) *` : `Unit Cost Price (${currencySymbol}) *`}</label>
                      <input 
                        type="number" 
                        required
                        step="0.01"
                        min="0.01"
                        value={restockCost}
                        onChange={(e) => setRestockCost(Math.max(0.01, Number(e.target.value)))}
                        className="w-full bg-white border border-slate-200 rounded-xl py-2 px-3 font-bold text-slate-800"
                      />
                    </div>

                    <div>
                      <label className="block text-[11px] font-bold text-slate-650 mb-1">{isRtl ? 'المورد المعتمد للفاتورة *' : 'Supplier Registry *'}</label>
                      <select
                        required
                        value={restockSupplierId}
                        onChange={(e) => setRestockSupplierId(e.target.value)}
                        className="w-full bg-white border border-slate-200 rounded-xl py-2 px-3 font-bold text-slate-800"
                      >
                        <option value="">{isRtl ? '-- حدد مورد الفاتورة --' : '-- Select Supplier --'}</option>
                        {suppliers.map(s => (
                          <option key={s.id} value={s.id}>{s.name} {s.company ? `(${s.company})` : ''}</option>
                        ))}
                      </select>
                    </div>

                    <div>
                      <label className="block text-[11px] font-bold text-slate-650 mb-1">{isRtl ? 'المستودع المستهدف للإيداع *' : 'Target Warehouse *'}</label>
                      <select
                        required
                        value={restockWarehouseId}
                        onChange={(e) => setRestockWarehouseId(e.target.value)}
                        className="w-full bg-white border border-slate-200 rounded-xl py-2 px-3 font-bold text-slate-800"
                      >
                        <option value="">{isRtl ? '-- حدد المستودع --' : '-- Select Warehouse --'}</option>
                        {warehouses.map(w => (
                          <option key={w.id} value={w.id}>{w.name}</option>
                        ))}
                      </select>
                    </div>

                    <div>
                      <label className="block text-[11px] font-bold text-slate-650 mb-1">{isRtl ? 'طريقة دفع تكلفة الشحنة' : 'Payment Type'}</label>
                      <div className="grid grid-cols-2 gap-2">
                        <button
                          type="button"
                          onClick={() => setRestockPayMethod('CASH')}
                          className={`py-1.5 rounded-xl border text-center font-bold transition cursor-pointer ${
                            restockPayMethod === 'CASH' ? 'bg-indigo-600 text-white border-indigo-600' : 'bg-white text-slate-600 border-slate-200'
                          }`}
                        >
                          {isRtl ? 'نقداً من الصندوق' : 'Cash Safe Box'}
                        </button>
                        <button
                          type="button"
                          onClick={() => setRestockPayMethod('DEBT')}
                          className={`py-1.5 rounded-xl border text-center font-bold transition cursor-pointer ${
                            restockPayMethod === 'DEBT' ? 'bg-indigo-600 text-white border-indigo-600' : 'bg-white text-slate-600 border-slate-200'
                          }`}
                        >
                          {isRtl ? 'آجل على الحساب' : 'Supplier Debt'}
                        </button>
                      </div>
                    </div>

                    <div className="flex flex-col justify-end">
                      <div className="p-2.5 bg-indigo-100/50 border border-indigo-200 rounded-xl text-center">
                        <p className="text-[10px] text-indigo-950 font-semibold">{isRtl ? 'إجمالي قيمة الشراء والتوريد:' : 'Total Restock Billing cost:'}</p>
                        <p className="text-base font-black text-indigo-700 font-mono">{currencySymbol} {(restockQty * restockCost).toFixed(2)}</p>
                      </div>
                    </div>
                  </div>

                  <button
                    type="submit"
                    className="w-full bg-emerald-600 hover:bg-emerald-700 text-white font-bold text-xs py-2.5 rounded-xl transition cursor-pointer shadow-md shadow-emerald-600/10"
                  >
                    {isRtl ? '✓ تأكيد عملية الشراء والتسجيل الفوري لحساب السلع' : '✓ Confirm & Register Emergency Stock Purchase'}
                  </button>
                </form>
              )}
            </div>

            {/* Modal Footer */}
            <div className="p-4 border-t border-slate-100 bg-slate-50 flex justify-end rounded-b-2xl">
              <button 
                onClick={() => { setShowLowStockModal(false); setSelectedRestockProd(null); }}
                className="bg-slate-200 hover:bg-slate-300 text-slate-800 font-bold text-xs px-4 py-2 rounded-xl transition cursor-pointer"
              >
                {isRtl ? 'إغلاق' : 'Close'}
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
