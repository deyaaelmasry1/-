function safeJson<T>(key:string, fallback:T):T { try { const raw=localStorage.getItem(key); return raw?JSON.parse(raw):fallback; } catch { return fallback; } }
function useDbPage<T>(table:string, search:string, orderBy:string){
  const [rows,setRows]=useState<T[]>([]); const [page,setPage]=useState(1); const [pages,setPages]=useState(1); const [count,setCount]=useState(0); const [loading,setLoading]=useState(true); const [error,setError]=useState('');
  const reload=useCallback(async()=>{ setLoading(true); setError(''); try { if(!desktopDatabase.available()) throw new Error('SQLite bridge is unavailable. Open the application through Electron, not directly in a browser.'); const result=await desktopDatabase.list<T>({table,page,pageSize:PAGE_SIZE,search,orderBy,direction:'DESC'}); setRows(result.rows); setPages(result.pages); setCount(result.count); } catch(e:any) { setError(e?.message||String(e)); setRows([]); setCount(0); setPages(1); } finally { setLoading(false); } },[table,page,search,orderBy]);
  useEffect(()=>{ const t=setTimeout(reload,120); return()=>clearTimeout(t); },[reload]); useEffect(()=>setPage(1),[search]);
  return {rows,page,setPage,pages,count,loading,error,reload};
}
async function migrateLegacyOnce(){
  if(!desktopDatabase.available() || localStorage.getItem('desktop_sqlite_migration_v1')==='done') return;
  await desktopDatabase.importLegacy({ quotations:safeJson('store_quotations',[]), installments:safeJson('store_installments',[]), deleted:safeJson('store_recycle_bin',[]) });
  localStorage.setItem('desktop_sqlite_migration_v1','done');
  localStorage.setItem('desktop_sqlite_migration_backup',JSON.stringify({ quotations:safeJson('store_quotations',[]), installments:safeJson('store_installments',[]), deleted:safeJson('store_recycle_bin',[]), migratedAt:new Date().toISOString() }));
}

export const InstallmentsModule=memo(function InstallmentsModule({customers,currencySymbol='ج.م',language='ar'}:{customers:Customer[];currencySymbol?:string;language?:string}){
 const ar=language==='ar';
 const [q,setQ]=useState(''); const [open,setOpen]=useState(false); const [saving,setSaving]=useState(false); const [formError,setFormError]=useState('');
 const [f,setF]=useState<any>({customerId:'',reference:'',total:'',paid:'0',dueDate:today()}); const data=useDbPage<InstallmentDb>('installment_plans',q,'due_date');
 useEffect(()=>{migrateLegacyOnce().then(data.reload).catch(()=>data.reload())},[]);
 const save=useCallback(async()=>{
   setFormError('');
   const c=customers.find(x=>String(x.id)===String(f.customerId)); const total=Number(f.total),paid=Number(f.paid||0);
   if(!c){setFormError(ar?'اختر العميل أولاً':'Select a customer first');return;}
   if(!Number.isFinite(total)||total<=0){setFormError(ar?'أدخل إجماليًا أكبر من صفر':'Enter a total greater than zero');return;}
   if(!Number.isFinite(paid)||paid<0||paid>total){setFormError(ar?'المدفوع يجب أن يكون بين صفر والإجمالي':'Paid must be between zero and total');return;}
   setSaving(true);
   try { await desktopDatabase.createInstallment({id:uid('ins'),customerId:c.id,customerName:c.name,reference:String(f.reference||'').trim()||`INS-${Date.now().toString().slice(-6)}`,total,paid,dueDate:f.dueDate||today()}); setOpen(false);setF({customerId:'',reference:'',total:'',paid:'0',dueDate:today()});await data.reload(); }
   catch(e:any){setFormError(e?.message||String(e));}
   finally{setSaving(false);}
 },[customers,f,data.reload,ar]);
 const collect=useCallback(async(x:InstallmentDb)=>{try{await desktopDatabase.collectInstallment({id:x.id,amount:x.total-x.paid});await data.reload();}catch(e:any){alert(e?.message||String(e));}},[data.reload]);
 const del=useCallback(async(x:InstallmentDb)=>{try{await desktopDatabase.softDelete({entity:'installment',id:x.id,label:x.reference});await data.reload();}catch(e:any){alert(e?.message||String(e));}},[data.reload]);
 return <Module title={ar?'الأقساط والتحصيل':'Installments'} icon={<CalendarClock/>} q={q} setQ={setQ} onAdd={()=>{setFormError('');setOpen(true)}} count={data.count}><Table loading={data.loading} headers={[ar?'العميل':'Customer',ar?'المرجع':'Reference',ar?'الإجمالي':'Total',ar?'المدفوع':'Paid',ar?'المتبقي':'Remaining',ar?'الاستحقاق':'Due',ar?'الحالة':'Status','']} rows={data.rows.map(x=>[x.customer_name,x.reference,`${Number(x.total).toLocaleString()} ${currencySymbol}`,`${Number(x.paid).toLocaleString()} ${currencySymbol}`,`${Math.max(0,Number(x.total)-Number(x.paid)).toLocaleString()} ${currencySymbol}`,x.due_date,x.status,<div className="flex gap-1"><button type="button" onClick={()=>collect(x)} title={ar?'تحصيل كامل':'Collect full'} className="p-2 text-emerald-600 hover:bg-emerald-50 rounded-lg"><CheckCircle2 size={17}/></button><button type="button" onClick={()=>del(x)} className="p-2 text-rose-600 hover:bg-rose-50 rounded-lg"><Trash2 size={17}/></button></div>])}/><Pager page={data.page} pages={data.pages} setPage={data.setPage}/>{open&&<Modal title={ar?'قسط جديد':'New installment'} onClose={()=>!saving&&setOpen(false)} onSave={save} saving={saving} saveLabel={saving?(ar?'جارٍ الحفظ...':'Saving...'):(ar?'حفظ القسط':'Save installment')} error={formError}><Select label={ar?'العميل *':'Customer *'} value={f.customerId} onChange={v=>setF((old:any)=>({...old,customerId:v}))} options={customers.map(x=>[x.id,x.name])}/><Input label={ar?'المرجع':'Reference'} value={f.reference} onChange={v=>setF((old:any)=>({...old,reference:v}))}/><Input label={ar?'الإجمالي *':'Total *'} type="number" value={f.total} onChange={v=>setF((old:any)=>({...old,total:v}))}/><Input label={ar?'المدفوع':'Paid'} type="number" value={f.paid} onChange={v=>setF((old:any)=>({...old,paid:v}))}/><Input label={ar?'تاريخ الاستحقاق':'Due date'} type="date" value={f.dueDate} onChange={v=>setF((old:any)=>({...old,dueDate:v}))}/></Modal>}</Module>;
});
