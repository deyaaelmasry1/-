// Deterministic activation code generator
export function generateLicenseKey(deviceId: string): string {
  const salt = "DiyaElmasri_@Retal9488_ERP_System";
  let hash1 = 0;
  let hash2 = 0;
  const combined = (deviceId || '').trim().toUpperCase() + salt;
  
  for (let i = 0; i < combined.length; i++) {
    const char = combined.charCodeAt(i);
    hash1 = ((hash1 << 5) - hash1) + char;
    hash1 = hash1 & hash1; // Convert to 32bit integer
    if (i % 2 === 0) {
      hash2 = ((hash2 << 7) - hash2) + (char * 13);
      hash2 = hash2 & hash2;
    }
  }
  
  const part1 = Math.abs(hash1).toString(16).toUpperCase().padStart(8, '0').slice(0, 4);
  const part2 = Math.abs(hash2).toString(16).toUpperCase().padStart(8, '0').slice(4, 8);
  const part3 = Math.abs(hash1 ^ hash2).toString(16).toUpperCase().padStart(8, '0').slice(2, 6);
  
  return `LIC-${part1}-${part2}-${part3}`;
}

// Generate device ID
export function generateDeviceId(): string {
  const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
  const randPart = (len: number) => Array.from({length: len}, () => chars[Math.floor(Math.random() * chars.length)]).join('');
  return `DEV-${randPart(4)}-${randPart(4)}-${randPart(4)}`;
}
