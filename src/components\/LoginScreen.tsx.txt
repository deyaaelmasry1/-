import React, { useState, useEffect, useRef } from 'react';
import { TeamMember } from '../types';
import { LogIn, User, Lock, Globe, Shield, Check, UserCheck } from 'lucide-react';
import { desktopDatabase } from '../utils/desktopDatabase';

interface LoginScreenProps {
  language: 'ar' | 'en';
  teamMembers: TeamMember[];
  onLogin: (user: TeamMember, updatedPassword?: string) => void;
  onToggleLanguage: () => void;
}

export default function LoginScreen({
  language,
  teamMembers,
  onLogin,
  onToggleLanguage,
}: LoginScreenProps) {
  const activeMembers = teamMembers.filter(m => m.active);
  
  // State variables
  const [selectedUserId, setSelectedUserId] = useState<string>(() => {
    // Default to the last logged in user or first active member
    const lastUser = localStorage.getItem('store_last_logged_in_user');
    if (lastUser && activeMembers.some(m => m.id === lastUser)) {
      return lastUser;
    }
    return activeMembers[0]?.id || '';
  });
  
  const [usernameInput, setUsernameInput] = useState<string>('');
  const [passwordInput, setPasswordInput] = useState<string>('');
  const [rememberPassword, setRememberPassword] = useState<boolean>(true);
  const [errorMsg, setErrorMsg] = useState<string>('');
  
  const passwordInputRef = useRef<HTMLInputElement>(null);

  const selectedMember = activeMembers.find(m => m.id === selectedUserId);

  // Load remembered passwords map from localStorage (high speed cached dictionary)
  const getRememberedUsers = (): Record<string, boolean> => {
    try {
      const saved = localStorage.getItem('store_remembered_users');
      return saved ? JSON.parse(saved) : {};
    } catch {
      return {};
    }
  };

  // Sync inputs and check for remembered credentials for instant 1-click login
  useEffect(() => {
    if (selectedMember) {
      setUsernameInput(selectedMember.username || '');
      setErrorMsg('');
      
      setPasswordInput('');
      
      // Auto-focus password input for swift workflow
      setTimeout(() => {
        if (passwordInputRef.current) {
          passwordInputRef.current.focus();
        }
      }, 50);
    }
  }, [selectedUserId]);

  const handleSubmit = async (e?: React.FormEvent) => {
    if (e) e.preventDefault();
    setErrorMsg('');

    if (!selectedMember) {
      setErrorMsg(language === 'ar' ? 'يرجى اختيار مستخدم أولاً' : 'Please select a user first');
      return;
    }

    // Auto-match username
    if (selectedMember.username && usernameInput.trim().toLowerCase() !== selectedMember.username.toLowerCase()) {
      setErrorMsg(language === 'ar' ? 'اسم المستخدم غير متطابق' : 'Username does not match');
      return;
    }

    const hasNoPassword = !selectedMember.password;

    if (hasNoPassword) {
      const newPass = passwordInput.trim() || '123';
      const hashed = await desktopDatabase.hashPassword(newPass);
      const remembered = getRememberedUsers();
      if (rememberPassword) remembered[selectedMember.id] = true; else delete remembered[selectedMember.id];
      localStorage.setItem('store_remembered_users', JSON.stringify(remembered));
      localStorage.removeItem('store_remembered_passwords');
      localStorage.setItem('store_last_logged_in_user', selectedMember.id);
      onLogin({ ...selectedMember, password: hashed || undefined }, hashed || undefined);
    } else {
      const valid = await desktopDatabase.verifyPassword(passwordInput, selectedMember.password || '');
      if (valid) {
        const remembered = getRememberedUsers();
        if (rememberPassword) remembered[selectedMember.id] = true; else delete remembered[selectedMember.id];
        localStorage.setItem('store_remembered_users', JSON.stringify(remembered));
        localStorage.removeItem('store_remembered_passwords');
        localStorage.setItem('store_last_logged_in_user', selectedMember.id);
        onLogin(selectedMember);
      } else {
        setErrorMsg(language === 'ar' ? 'كلمة المرور غير صحيحة!' : 'Incorrect password!');
      }
    }
  };

  // Quick profile selection never stores or reuses plaintext credentials.
  const handleQuickLogin = (member: TeamMember) => {
    setSelectedUserId(member.id);
    setErrorMsg('');
    if (!member.password) {
      setPasswordInput('');
      setTimeout(() => passwordInputRef.current?.focus(), 0);
    }
  };

  const isRtl = language === 'ar';

  return (
    <div 
      className="min-h-screen bg-slate-950 flex flex-col items-center justify-center p-4 antialiased font-sans text-slate-100"
      dir={isRtl ? 'rtl' : 'ltr'}
    >
      {/* Top Bar for language and brand indicator */}
      <div className="absolute top-6 left-6 right-6 flex justify-between items-center no-print">
        <div className="flex items-center gap-2">
          <Shield className="w-5 h-5 text-indigo-500" />
          <span className="text-xs font-black tracking-wider text-slate-400">ERP SECURE SHIELD</span>
        </div>
        <button
          onClick={onToggleLanguage}
          className="bg-slate-900 hover:bg-slate-800 text-slate-300 font-bold px-4 py-2 rounded-xl text-xs flex items-center gap-1.5 border border-slate-800 transition cursor-pointer active:scale-95"
        >
          <Globe className="w-4 h-4 text-indigo-400" />
          {language === 'ar' ? 'English' : 'العربية'}
        </button>
      </div>

      <div className="w-full max-w-4xl grid grid-cols-1 md:grid-cols-12 gap-8 items-stretch my-8">
        
        {/* Left/Right Column: Interactive Profiles Grid (7 cols) */}
        <div className="md:col-span-7 bg-slate-900 rounded-3xl p-6 sm:p-8 border border-slate-800 flex flex-col justify-between space-y-6">
          <div>
            <span className="text-[10px] font-black tracking-wider text-indigo-400 uppercase block mb-1">
              {isRtl ? 'بوابة التحقق السريع للموظفين' : 'Instant Staff Validation Gate'}
            </span>
            <h1 className="text-lg sm:text-xl font-black text-white tracking-tight">
              {isRtl ? 'اختر حسابك لبدء العمل 🔓' : 'Select Profile to Access ERP 🔓'}
            </h1>
            <p className="text-xs text-slate-400 mt-1 leading-relaxed">
              {isRtl 
                ? 'انقر على بطاقتك الشخصية. يدعم النظام الدخول السريع بنقرة واحدة للحسابات الموثقة والمحفوظة.' 
                : 'Click your profile card. Fully supports blazing-fast 1-click login for remembered credentials.'}
            </p>
          </div>

          {/* Profiles Grid */}
          <div className="grid grid-cols-1 sm:grid-cols-2 gap-4 my-2">
            {activeMembers.map(m => {
              const isSelected = m.id === selectedUserId;
              const hasRemembered = !!getRememberedUsers()[m.id];
              return (
                <div
                  key={m.id}
                  onClick={() => setSelectedUserId(m.id)}
                  onDoubleClick={() => handleQuickLogin(m)}
                  className={`relative p-4 rounded-2xl border transition-all duration-200 cursor-pointer flex flex-col justify-between ${
                    isSelected 
                      ? 'bg-indigo-600/10 border-indigo-500 shadow-lg shadow-indigo-950/40 ring-1 ring-indigo-500/50' 
                      : 'bg-slate-950 hover:bg-slate-850 border-slate-800 hover:border-slate-700'
                  }`}
                >
                  <div className="flex items-center gap-3">
                    {/* Fast Avatar with initial */}
                    <div className={`w-10 h-10 rounded-full flex items-center justify-center font-black text-xs uppercase border ${
                      isSelected 
                        ? 'bg-indigo-600 border-indigo-400 text-white' 
                        : 'bg-slate-800 border-slate-700 text-slate-300'
                    }`}>
                      {m.name.slice(0, 2)}
                    </div>
                    <div className="overflow-hidden">
                      <h3 className="font-extrabold text-xs text-slate-100 truncate">{m.name}</h3>
                      <span className="text-[9px] text-slate-500 font-bold block capitalize mt-0.5">
                        {m.role === 'admin' && (isRtl ? '👑 مدير النظام' : '👑 Administrator')}
                        {m.role === 'cashier' && (isRtl ? '💵 كاشير مبيعات' : '💵 Sales Cashier')}
                        {m.role === 'storekeeper' && (isRtl ? '📦 أمين مستودع' : '📦 Storekeeper')}
                      </span>
                    </div>
                  </div>

                  <div className="flex justify-between items-center mt-4 pt-3 border-t border-slate-800/60 text-[9px]">
                    <span className="text-slate-400 font-mono">@{m.username || 'user'}</span>
                    
                    {/* Quick login Badge */}
                    {hasRemembered ? (
                      <button
                        type="button"
                        onClick={(e) => {
                          e.stopPropagation();
                          handleQuickLogin(m);
                        }}
                        className="bg-emerald-500/15 text-emerald-400 border border-emerald-500/30 px-2 py-1 rounded-lg font-bold hover:bg-emerald-500 hover:text-white transition cursor-pointer flex items-center gap-1 shrink-0"
                        title={isRtl ? 'دخول سريع فوري بنقرة واحدة' : 'Instant 1-click access'}
                      >
                        <UserCheck className="w-3 h-3" />
                        <span>{isRtl ? 'دخول سريع' : '1-Click'}</span>
                      </button>
                    ) : !m.password ? (
                      <span className="text-amber-400 bg-amber-500/10 px-1.5 py-0.5 rounded border border-amber-500/20 font-bold shrink-0">
                        {isRtl ? 'بدون كلمة مرور' : 'No Password'}
                      </span>
                    ) : (
                      <span className="text-slate-500 shrink-0">
                        {isRtl ? 'مؤمن' : 'Secured'}
                      </span>
                    )}
                  </div>

                  {isSelected && (
                    <div className="absolute top-3 right-3 w-4 h-4 rounded-full bg-indigo-500 text-white flex items-center justify-center shadow-md">
                      <Check className="w-2.5 h-2.5 stroke-[3]" />
                    </div>
                  )}
                </div>
              );
            })}
          </div>

          <div className="text-[10px] text-slate-500 border-t border-slate-800/60 pt-3 flex items-center justify-between">
            <span>{isRtl ? 'تلميح: انقر نقراً مزدوجاً للدخول السريع التلقائي.' : 'Hint: Double click card for instant validation.'}</span>
            <span className="font-mono text-indigo-500 font-bold">ERP v4.2 Stable</span>
          </div>
        </div>

        {/* Right/Left Column: Security Authentication Pad (5 cols) */}
        <div className="md:col-span-5 bg-slate-900 rounded-3xl p-6 sm:p-8 border border-slate-800 flex flex-col justify-between space-y-6">
          <div className="space-y-4">
            <span className="bg-indigo-600/15 border border-indigo-500/20 text-indigo-400 px-3 py-1.5 rounded-xl text-[10px] font-black inline-flex items-center gap-1">
              <Lock className="w-3.5 h-3.5" />
              <span>{isRtl ? 'التحقق الثنائي والوصول' : 'SECURE CREDENTIAL INPUT'}</span>
            </span>
            
            <div>
              <h2 className="text-sm font-extrabold text-slate-100">
                {isRtl ? 'تأكيد الهوية الشخصية' : 'Credential Challenge'}
              </h2>
              <p className="text-[11px] text-slate-400 mt-0.5">
                {isRtl ? 'أدخل كلمة المرور الخاصة بحسابك لإلغاء القفل' : 'Type your system password to unlock the console.'}
              </p>
            </div>

            {selectedMember ? (
              <div className="bg-slate-950 p-4 rounded-2xl border border-slate-800 space-y-2 text-xs">
                <div className="flex justify-between items-center border-b border-slate-900 pb-2">
                  <span className="text-slate-500 font-bold">{isRtl ? 'الموظف المختار:' : 'Selected Profile:'}</span>
                  <span className="font-black text-white text-xs">{selectedMember.name}</span>
                </div>
                <div className="flex justify-between items-center text-[11px]">
                  <span className="text-slate-500 font-bold">{isRtl ? 'الرتبة المعتمدة:' : 'System Access Role:'}</span>
                  <span className="font-bold text-indigo-400 uppercase tracking-wider">
                    {selectedMember.role === 'admin' ? (isRtl ? 'المدير العام' : 'Administrator') : selectedMember.role === 'cashier' ? (isRtl ? 'الكاشير' : 'POS Cashier') : (isRtl ? 'أمين المستودع' : 'Storekeeper')}
                  </span>
                </div>
              </div>
            ) : (
              <div className="bg-slate-950 p-4 rounded-2xl border border-slate-800/80 text-center text-slate-500 font-bold text-xs">
                {isRtl ? '⚠️ الرجاء تحديد موظف من اليسار' : '⚠️ Please select an employee profile'}
              </div>
            )}

            {/* Main Inputs Form */}
            {selectedMember && (
              <form onSubmit={handleSubmit} className="space-y-4 text-xs">
                
                {/* Username hidden/disabled for clean look */}
                <div>
                  <label className="block text-[9px] font-black text-slate-500 uppercase tracking-wider mb-1">
                    {isRtl ? 'اسم المستخدم المعتمد' : 'Login Username'}
                  </label>
                  <div className="relative">
                    <User className={`absolute ${isRtl ? 'right-3.5' : 'left-3.5'} top-2.5 w-4 h-4 text-slate-500`} />
                    <input
                      type="text"
                      disabled
                      value={usernameInput}
                      className={`w-full bg-slate-950 border border-slate-800 rounded-xl py-2.5 ${isRtl ? 'pr-10 pl-4' : 'pl-10 pr-4'} text-slate-400 outline-none cursor-not-allowed font-bold`}
                    />
                  </div>
                </div>

                {/* Password Box */}
                <div>
                  <label className="block text-[9px] font-black text-slate-400 uppercase tracking-wider mb-1">
                    {isRtl ? 'كلمة المرور المشفرة *' : 'Security Password *'}
                  </label>
                  <div className="relative">
                    <Lock className={`absolute ${isRtl ? 'right-3.5' : 'left-3.5'} top-2.5 w-4 h-4 text-slate-400`} />
                    <input
                      ref={passwordInputRef}
                      type="password"
                      required
                      value={passwordInput}
                      onChange={(e) => setPasswordInput(e.target.value)}
                      placeholder={!selectedMember.password ? (isRtl ? 'اكتب كلمة مرور لحفظها' : 'Type to set password') : '••••••••'}
                      className={`w-full bg-slate-950 border border-slate-750 focus:border-indigo-500 rounded-xl py-2.5 ${isRtl ? 'pr-10 pl-4' : 'pl-10 pr-4'} text-white outline-none font-mono text-xs focus:ring-1 focus:ring-indigo-500/50`}
                    />
                  </div>
                </div>

                {/* Password Setting Info */}
                {!selectedMember.password && (
                  <div className="bg-amber-500/10 border border-amber-500/20 rounded-xl p-3 text-amber-400 text-[10px] leading-relaxed">
                    💡 {isRtl 
                      ? 'هذا الحساب بدون كلمة مرور حالياً. اكتب أي كلمة مرور وسيقوم النظام بتعيينها تلقائياً!'
                      : 'This profile is password-free. Enter any password and the system will assign it immediately!'}
                  </div>
                )}

                {/* Remember password option */}
                <div className="flex items-center justify-between py-1">
                  <label className="flex items-center gap-2 cursor-pointer text-[10px] font-bold text-slate-400 select-none">
                    <input
                      type="checkbox"
                      checked={rememberPassword}
                      onChange={(e) => setRememberPassword(e.target.checked)}
                      className="rounded border-slate-800 text-indigo-600 focus:ring-0 bg-slate-950"
                    />
                    <span>{isRtl ? 'حفظ كلمة المرور للدخول السريع فورا' : 'Remember me for 1-click login'}</span>
                  </label>
                </div>

                {/* Error Box */}
                {errorMsg && (
                  <div className="bg-rose-500/15 border border-rose-500/25 text-rose-400 rounded-xl p-3 text-[11px] font-bold">
                    ❌ {errorMsg}
                  </div>
                )}

                {/* Action button */}
                <button
                  type="submit"
                  className="w-full bg-indigo-600 hover:bg-indigo-500 text-white font-black py-3 rounded-xl shadow-lg hover:shadow-indigo-950/40 active:scale-95 transition flex items-center justify-center gap-2 text-xs cursor-pointer"
                >
                  <LogIn className="w-4.5 h-4.5" />
                  <span>{isRtl ? 'تسجيل الدخول الآمن للنظام 🔓' : 'Authorize & Unlock Console 🔓'}</span>
                </button>

              </form>
            )}
          </div>

          <div className="text-[9px] text-slate-600 text-center leading-relaxed font-mono">
            SECURE DEVICE STORAGE ACTIVE • ENCRYPTED SESSION
          </div>
        </div>

      </div>

      <p className="text-[10px] text-slate-500 tracking-wider mt-2">
        {isRtl ? 'نظام المحاسبة السحابي والتحقق الذاتي من الهوية' : 'Cloud Secure Enterprise Retail Ledger Systems'}
      </p>
    </div>
  );
}
