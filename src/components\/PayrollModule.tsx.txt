import React,{useCallback,useEffect,useMemo,useState} from 'react';
import {AlertTriangle,Banknote,CheckCircle2,Plus,RefreshCw,Receipt,Users} from 'lucide-react';
import {desktopDatabase} from '../utils/desktopDatabase';
import type {TeamMember} from '../types';

type Employee={id:string;team_member_id?:string;name:string;job_title?:string;employee_role?:string;phone?:string;base_salary:number;overtime_rate:number;standard_daily_hours:number;standard_monthly_days:number;attendance_employee_id?:string;active:number};
type Run={id:string;month:string;status:string;gross_total:number;deductions_total:number;net_total:number;generated_at:string};
type Item={id:string;employee_name:string;base_salary:number;allowances:number;bonuses:number;overtime_hours:number;overtime_amount:number;absence_days:number;absence_deduction:number;deductions:number;loans:number;gross_salary:number;net_salary:number;worked_hours:number;attendance_days:number};
const numberValue=(value:unknown,fallback=0)=>{const normalized=String(value??'').replace(/,/g,'.').replace(/[^0-9.-]/g,'');const n=Number(normalized);return Number.isFinite(n)?n:fallback};
const initialForm={teamMemberId:'',name:'',jobTitle:'',employeeRole:'',phone:'',baseSalary:'',overtimeRate:'',standardDailyHours:'8',standardMonthlyDays:'30',active:true};
function Field({label,hint,children}:{label:string;hint?:string;children:React.ReactNode}){return <label className="block space-y-1"><span className="block text-xs font-black text-slate-700">{label}</span>{hint&&<span className="block text-[10px] leading-4 text-slate-500">{hint}</span>}{children}</label>}
export const PayrollModule=({teamMembers,language,currencySymbol,mode='all'}:{teamMembers:TeamMember[];language:string;currencySymbol:string;mode?:'all'|'employees'|'payroll'})=>{
 const ar=language==='ar', now=new Date().toISOString().slice(0,7);
 const [employees,setEmployees]=useState<Employee[]>([]),[runs,setRuns]=useState<Run[]>([]),[month,setMonth]=useState(now),[selectedRun,setSelectedRun]=useState<any>(null);
 const [form,setForm]=useState<any>(initialForm);
 const [adj,setAdj]=useState<any>({employeeId:'',adjustmentType:'ALLOWANCE',amount:'',description:''});
 const [status,setStatus]=useState<{type:'ok'|'error'|'loading';message:string}>({type:'loading',message:ar?'جارٍ الاتصال بقاعدة البيانات…':'Connecting to database…'});
 const reload=useCallback(async()=>{if(!desktopDatabase.available()){setStatus({type:'error',message:ar?'قاعدة SQLite غير متاحة. افتح النظام من تطبيق سطح المكتب وليس المتصفح.':'SQLite is unavailable. Open the desktop application.'});return}try{setStatus({type:'loading',message:ar?'جارٍ تحميل بيانات الموظفين والمرتبات…':'Loading employees and payroll…'});const [e,r]=await Promise.all([desktopDatabase.list<Employee>({table:'payroll_employees',page:1,pageSize:500,orderBy:'name'}),desktopDatabase.list<Run>({table:'payroll_runs',page:1,pageSize:100,orderBy:'generated_at',direction:'DESC'})]);setEmployees(e.rows);setRuns(r.rows);setStatus({type:'ok',message:ar?`تم الاتصال بقاعدة البيانات — ${e.rows.length} موظف`:`Database connected — ${e.rows.length} employees`})}catch(error){setStatus({type:'error',message:`${ar?'فشل تحميل قاعدة البيانات':'Database load failed'}: ${error instanceof Error?error.message:String(error)}`})}},[ar]);
 useEffect(()=>{reload()},[reload]);
 const saveEmployee=async()=>{const tm=teamMembers.find(x=>x.id===form.teamMemberId);const name=(tm?.name||form.name||'').trim();if(!name)return alert(ar?'اكتب اسم الموظف':'Enter employee name');try{await desktopDatabase.savePayrollEmployee({...form,name,baseSalary:numberValue(form.baseSalary),overtimeRate:numberValue(form.overtimeRate),standardDailyHours:numberValue(form.standardDailyHours,8),standardMonthlyDays:numberValue(form.standardMonthlyDays,30),attendanceEmployeeId:form.teamMemberId||undefined});setForm(initialForm);await reload();alert(ar?'تم حفظ الموظف بنجاح':'Employee saved')}catch(error){alert(`${ar?'تعذر حفظ الموظف':'Could not save employee'}: ${error instanceof Error?error.message:String(error)}`)}};
 const addAdjustment=async()=>{const amount=numberValue(adj.amount);if(!adj.employeeId||amount<=0)return alert(ar?'اختر الموظف واكتب مبلغًا أكبر من صفر':'Select employee and enter an amount above zero');try{await desktopDatabase.addPayrollAdjustment({...adj,amount,month});setAdj({...adj,amount:'',description:''});alert(ar?'تمت إضافة الحركة':'Adjustment added')}catch(error){alert(`${ar?'تعذر إضافة الحركة':'Could not add adjustment'}: ${error instanceof Error?error.message:String(error)}`)}};
 const generate=async()=>{try{await desktopDatabase.generatePayrollRun({month});await reload();setSelectedRun(await desktopDatabase.getPayrollRun(month))}catch(error){alert(`${ar?'تعذر إنشاء المسير':'Could not generate payroll'}: ${error instanceof Error?error.message:String(error)}`)}};
 const openRun=async(id:string)=>setSelectedRun(await desktopDatabase.getPayrollRun(id));
 const money=(v:number)=>`${Number(v||0).toFixed(2)} ${currencySymbol}`;
 const totals=useMemo(()=>employees.reduce((a,e)=>a+Number(e.base_salary||0),0),[employees]);
 const inputClass='w-full border border-slate-300 rounded-xl px-3 py-2.5 bg-white text-slate-900 focus:outline-none focus:ring-2 focus:ring-indigo-500';
 return <div className="space-y-5" dir={ar?'rtl':'ltr'}>
  <div className={`rounded-xl border px-4 py-3 text-xs font-bold flex items-center gap-2 ${status.type==='error'?'border-rose-300 bg-rose-50 text-rose-700':status.type==='ok'?'border-emerald-300 bg-emerald-50 text-emerald-700':'border-amber-300 bg-amber-50 text-amber-700'}`}>{status.type==='error'?<AlertTriangle size={17}/>:status.type==='ok'?<CheckCircle2 size={17}/>:<RefreshCw size={17} className="animate-spin"/>}{status.message}</div>
  <div className="flex flex-wrap justify-between gap-3"><div><h2 className="font-black text-xl flex gap-2"><Banknote/>{mode==='employees'?(ar?'الموظفون':'Employees'):(ar?'نظام المرتبات':'Payroll')}</h2><p className="text-xs text-slate-500">{ar?'الموظف مستقل عن مستخدم النظام، ويمكن ربطه بالحضور والمرتبات':'Employees can exist independently of system users'}</p></div><button type="button" onClick={reload} className="p-2 border rounded-xl"><RefreshCw size={18}/></button></div>
  {mode!=='payroll'&&<div className="grid lg:grid-cols-3 gap-4"><div className="lg:col-span-2 border rounded-2xl p-4 space-y-4"><h3 className="font-black flex gap-2"><Plus size={18}/>{ar?'بيانات الموظف':'Employee details'}</h3>
   {teamMembers.length>0&&<Field label={ar?'ربط اختياري بحساب من فريق العمل':'Optional system-user link'} hint={ar?'اتركه فارغًا إذا كان الموظف لا يستخدم البرنامج.':'Leave empty when the employee has no system login.'}><select value={form.teamMemberId} onChange={e=>{const id=e.target.value,tm=teamMembers.find(x=>x.id===id);setForm({...form,teamMemberId:id,name:tm?.name||form.name,jobTitle:tm?.role||form.jobTitle,phone:tm?.phone||form.phone})}} className={inputClass}><option value="">{ar?'لا يوجد حساب دخول — موظف مستقل':'No login account — independent employee'}</option>{teamMembers.map(x=><option key={x.id} value={x.id}>{x.name} — {x.role}</option>)}</select></Field>}
   <div className="grid md:grid-cols-2 gap-3"><Field label={ar?'اسم الموظف *':'Employee name *'} hint={ar?'الاسم الذي سيظهر في الحضور وقسيمة الراتب.':'Shown in attendance and payroll slips.'}><input value={form.name} onChange={e=>setForm({...form,name:e.target.value})} className={inputClass}/></Field><Field label={ar?'رقم الهاتف':'Phone number'} hint={ar?'اختياري، اكتب الرقم كما هو دون عمليات حسابية.':'Optional contact number.'}><input inputMode="tel" value={form.phone} onChange={e=>setForm({...form,phone:e.target.value})} className={inputClass}/></Field><Field label={ar?'المسمى الوظيفي':'Job title'} hint={ar?'مثال: محاسب، سائق، عامل مخزن.':'Example: accountant, driver, warehouse worker.'}><input value={form.jobTitle} onChange={e=>setForm({...form,jobTitle:e.target.value})} className={inputClass}/></Field><Field label={ar?'دور الموظف وأعماله':'Employee role and duties'} hint={ar?'وصف حر لمهام الموظف، ولا يمنحه صلاحيات داخل البرنامج.':'Free-text duties; does not grant system permissions.'}><input value={form.employeeRole} onChange={e=>setForm({...form,employeeRole:e.target.value})} className={inputClass}/></Field></div>
   <div className="rounded-xl bg-slate-50 border p-3"><h4 className="font-black text-sm mb-3">{ar?'إعدادات الراتب والحساب':'Salary calculation settings'}</h4><div className="grid md:grid-cols-2 gap-3"><Field label={`${ar?'الراتب الأساسي الشهري':'Monthly base salary'} (${currencySymbol})`} hint={ar?'المبلغ الثابت قبل البدلات والخصومات.':'Fixed amount before allowances and deductions.'}><input inputMode="decimal" value={form.baseSalary} onChange={e=>setForm({...form,baseSalary:e.target.value})} placeholder="0.00" className={inputClass}/></Field><Field label={`${ar?'قيمة ساعة العمل الإضافي':'Overtime hourly rate'} (${currencySymbol})`} hint={ar?'المبلغ المستحق عن كل ساعة إضافية محسوبة.':'Amount paid for each overtime hour.'}><input inputMode="decimal" value={form.overtimeRate} onChange={e=>setForm({...form,overtimeRate:e.target.value})} placeholder="0.00" className={inputClass}/></Field><Field label={ar?'ساعات العمل القياسية يوميًا':'Standard working hours per day'} hint={ar?'تستخدم لحساب الساعات الإضافية. مثال: 8 ساعات.':'Used to calculate overtime. Example: 8 hours.'}><input inputMode="decimal" value={form.standardDailyHours} onChange={e=>setForm({...form,standardDailyHours:e.target.value})} className={inputClass}/></Field><Field label={ar?'أيام العمل المستحقة في الشهر':'Expected paid days per month'} hint={ar?'تستخدم لحساب خصم الغياب. مثال: 30 أو 26 يومًا حسب نظامك.':'Used for absence deductions. Example: 30 or 26 days.'}><input inputMode="numeric" value={form.standardMonthlyDays} onChange={e=>setForm({...form,standardMonthlyDays:e.target.value})} className={inputClass}/></Field></div></div>
   <button type="button" disabled={status.type==='error'} onClick={saveEmployee} className="bg-indigo-600 disabled:bg-slate-400 text-white rounded-xl px-5 py-3 font-bold">{ar?'حفظ بيانات الموظف':'Save employee'}</button></div><div className="border rounded-2xl p-4"><h3 className="font-black flex gap-2"><Users size={18}/>{ar?'ملخص الموظفين':'Employees summary'}</h3><div className="text-3xl font-black mt-4">{employees.length}</div><div className="text-xs text-slate-500">{ar?'إجمالي الرواتب الأساسية الشهرية':'Total monthly base salaries'}</div><div className="font-bold">{money(totals)}</div></div></div>}
  {mode!=='employees'&&<><div className="border rounded-2xl p-4 space-y-3"><h3 className="font-black">{ar?'بدلات وخصومات الشهر':'Monthly adjustments'}</h3><div className="grid md:grid-cols-2 lg:grid-cols-4 gap-3"><Field label={ar?'الموظف':'Employee'}><select value={adj.employeeId} onChange={e=>setAdj({...adj,employeeId:e.target.value})} className={inputClass}><option value="">{ar?'اختر الموظف':'Select employee'}</option>{employees.map(e=><option key={e.id} value={e.id}>{e.name}</option>)}</select></Field><Field label={ar?'نوع الحركة':'Adjustment type'}><select value={adj.adjustmentType} onChange={e=>setAdj({...adj,adjustmentType:e.target.value})} className={inputClass}><option value="ALLOWANCE">{ar?'بدل يضاف للراتب':'Allowance — adds'}</option><option value="BONUS">{ar?'مكافأة تضاف للراتب':'Bonus — adds'}</option><option value="DEDUCTION">{ar?'خصم من الراتب':'Deduction — subtracts'}</option><option value="LOAN">{ar?'قسط سلفة يخصم':'Loan installment — subtracts'}</option></select></Field><Field label={`${ar?'قيمة الحركة':'Amount'} (${currencySymbol})`} hint={ar?'اكتب قيمة موجبة؛ نوع الحركة يحدد الإضافة أو الخصم.':'Enter a positive value; type controls add/subtract.'}><input inputMode="decimal" value={adj.amount} onChange={e=>setAdj({...adj,amount:e.target.value})} placeholder="0.00" className={inputClass}/></Field><Field label={ar?'البيان أو السبب':'Description / reason'}><input value={adj.description} onChange={e=>setAdj({...adj,description:e.target.value})} className={inputClass}/></Field></div><button type="button" onClick={addAdjustment} className="bg-slate-900 text-white rounded-xl px-5 py-2.5 font-bold">{ar?'إضافة الحركة':'Add adjustment'}</button></div>
  <div className="border rounded-2xl p-4 space-y-3"><div className="flex flex-wrap gap-3 items-end"><Field label={ar?'شهر مسير المرتبات':'Payroll month'} hint={ar?'اختر الشهر الذي سيحسب له الحضور والبدلات والخصومات.':'Attendance and adjustments are calculated for this month.'}><input type="month" value={month} onChange={e=>setMonth(e.target.value)} className={inputClass}/></Field><button type="button" onClick={generate} className="bg-emerald-600 text-white rounded-xl px-4 py-2.5 font-bold">{ar?'إنشاء أو إعادة احتساب المسير':'Generate / recalculate payroll'}</button></div><div className="overflow-auto"><table className="w-full text-xs"><thead><tr className="bg-slate-50"><th className="p-2 text-right">{ar?'الشهر':'Month'}</th><th>{ar?'الحالة':'Status'}</th><th>{ar?'إجمالي الاستحقاقات':'Gross'}</th><th>{ar?'إجمالي الخصومات':'Deductions'}</th><th>{ar?'صافي المرتبات':'Net'}</th><th></th></tr></thead><tbody>{runs.map(r=><tr key={r.id} className="border-t"><td className="p-2 font-bold">{r.month}</td><td>{r.status}</td><td>{money(r.gross_total)}</td><td>{money(r.deductions_total)}</td><td>{money(r.net_total)}</td><td><button onClick={()=>openRun(r.id)} className="p-2"><Receipt size={16}/></button></td></tr>)}</tbody></table></div></div></>}
  {mode!=='employees'&&selectedRun&&<div className="border rounded-2xl overflow-auto"><div className="p-4 font-black">{ar?'قسائم مسير':'Payroll slips'} {selectedRun.month}</div><table className="w-full text-xs"><thead className="bg-slate-50"><tr><th className="p-2 text-right">{ar?'الموظف':'Employee'}</th><th>{ar?'الأساسي':'Base'}</th><th>{ar?'البدلات والمكافآت':'Allowances'}</th><th>{ar?'الإضافي':'Overtime'}</th><th>{ar?'خصم الغياب':'Absence'}</th><th>{ar?'خصومات وسلف':'Other deductions'}</th><th>{ar?'الصافي':'Net'}</th></tr></thead><tbody>{(selectedRun.items||[]).map((i:Item)=><tr key={i.id} className="border-t"><td className="p-2 font-bold">{i.employee_name}<div className="text-slate-400">{i.attendance_days} {ar?'يوم حضور':'attendance days'} · {Number(i.worked_hours).toFixed(1)}h</div></td><td>{money(i.base_salary)}</td><td>{money(i.allowances+i.bonuses)}</td><td>{money(i.overtime_amount)}</td><td>{money(i.absence_deduction)}</td><td>{money(i.deductions+i.loans)}</td><td className="font-black">{money(i.net_salary)}</td></tr>)}</tbody></table></div>}
 </div>
};
