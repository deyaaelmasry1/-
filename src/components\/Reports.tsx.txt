import { 
  Product, 
  Transaction, 
  Category, 
  Supplier, 
  Customer, 
  Warehouse, 
  Expense, 
  FinancialVoucher, 
  Invoice,
  TeamMember
} from '../types';
import { Language } from '../utils/translations';
import { printHtml } from '../utils/printHelper';

import { 
  TrendingUp,

interface ReportsProps {
  products: Product[];
  categories: Category[];
  suppliers: Supplier[];
  transactions: Transaction[];
  customers: Customer[];
  warehouses: Warehouse[];
  expenses: Expense[];
  vouchers: FinancialVoucher[];
  invoices: Invoice[];
  teamMembers: TeamMember[];
  language: Language;
  companyName: string;
  currencySymbol?: string;
  setProducts: React.Dispatch<React.SetStateAction<Product[]>>;
  setSuppliers: React.Dispatch<React.SetStateAction<Supplier[]>>;
  setTransactions: React.Dispatch<React.SetStateAction<Transaction[]>>;
  setCustomers: React.Dispatch<React.SetStateAction<Customer[]>>;
  setWarehouses: React.Dispatch<React.SetStateAction<Warehouse[]>>;
  setExpenses: React.Dispatch<React.SetStateAction<Expense[]>>;
  setVouchers: React.Dispatch<React.SetStateAction<FinancialVoucher[]>>;
  setInvoices: React.Dispatch<React.SetStateAction<Invoice[]>>;
}

type ReportTab = 'sales_profits' | 'inventory_valuation' | 'expenses_cash' | 'tax_vat' | 'performance_entities' | 'records';

export default function Reports({
  products,
  categories,
  suppliers,
  transactions,
  customers,
  warehouses,
  expenses,
  vouchers,
  invoices,
  teamMembers,
  language,
  companyName,
  currencySymbol = language === 'ar' ? 'ج.م' : '$',
  setProducts, setSuppliers, setTransactions, setCustomers, setWarehouses, setExpenses, setVouchers, setInvoices
}: ReportsProps) {
  const isRtl = language === 'ar';

  // State
  const [activeReportTab, setActiveReportTab] = useState<ReportTab>('sales_profits');
  const [dateRange, setDateRange] = useState<'today' | 'yesterday' | '7days' | 'month' | 'last_month' | 'custom'>('month');
  const [startDate, setStartDate] = useState<string>(() => {
    const d = new Date();
    d.setDate(1); // Default to start of current month
    return d.toISOString().split('T')[0];
  });
  const [endDate, setEndDate] = useState<string>(() => {
    return new Date().toISOString().split('T')[0];
  });
  const [searchQuery, setSearchQuery] = useState<string>('');

  // Pre-sets handler
  const handleDatePreset = (preset: 'today' | 'yesterday' | '7days' | 'month' | 'last_month' | 'all') => {
    const today = new Date();
    const todayStr = today.toISOString().split('T')[0];

    if (preset === 'today') {
      setDateRange('today');
      setStartDate(todayStr);
      setEndDate(todayStr);
    } else if (preset === 'yesterday') {
      setDateRange('yesterday');
      const yesterday = new Date();
      yesterday.setDate(today.getDate() - 1);
      const yestStr = yesterday.toISOString().split('T')[0];
      setStartDate(yestStr);
      setEndDate(yestStr);
    } else if (preset === '7days') {
      setDateRange('7days');
      const lastWeek = new Date();
      lastWeek.setDate(today.getDate() - 7);
      setStartDate(lastWeek.toISOString().split('T')[0]);
      setEndDate(todayStr);
    } else if (preset === 'month') {
      setDateRange('month');
      const startOfMonth = new Date(today.getFullYear(), today.getMonth(), 1);
      setStartDate(startOfMonth.toISOString().split('T')[0]);
      setEndDate(todayStr);
    } else if (preset === 'last_month') {
      setDateRange('last_month');
      const startOfLastMonth = new Date(today.getFullYear(), today.getMonth() - 1, 1);
      const endOfLastMonth = new Date(today.getFullYear(), today.getMonth(), 0);
      setStartDate(startOfLastMonth.toISOString().split('T')[0]);
      setEndDate(endOfLastMonth.toISOString().split('T')[0]);
    } else if (preset === 'all') {
      setDateRange('custom');
      setStartDate('2020-01-01');
      setEndDate(todayStr);
    }
  };

  // Filter Helper for dates
  const isWithinDateRange = (itemDateStr: string) => {
    const itemDate = new Date(itemDateStr);
    itemDate.setHours(0, 0, 0, 0);

    const start = new Date(startDate);
    start.setHours(0, 0, 0, 0);

    const end = new Date(endDate);
    end.setHours(23, 59, 59, 999);

    return itemDate >= start && itemDate <= end;
  };

  // 1. --- FILTERED DATA SETS ---
  const filteredInvoices = useMemo(() => {
    return invoices.filter(inv => {
      // Date filter
      const matchesDate = isWithinDateRange(inv.date);
      // Search filter
      const matchesSearch = searchQuery === '' || 
        inv.id.toLowerCase().includes(searchQuery.toLowerCase()) ||
        inv.customerName.toLowerCase().includes(searchQuery.toLowerCase()) ||
        inv.customerPhone.includes(searchQuery);

      return matchesDate && matchesSearch;
    });
  }, [invoices, startDate, endDate, searchQuery]);

  const filteredTransactions = useMemo(() => {
    return transactions.filter(tx => {
      const matchesDate = isWithinDateRange(tx.date);
      const matchesSearch = searchQuery === '' || 
        (tx.id && tx.id.toLowerCase().includes(searchQuery.toLowerCase())) ||
        (tx.notes && tx.notes.toLowerCase().includes(searchQuery.toLowerCase()));
      return matchesDate && matchesSearch;
    });
  }, [transactions, startDate, endDate, searchQuery]);

  const filteredExpenses = useMemo(() => {
    return expenses.filter(exp => {
      const matchesDate = isWithinDateRange(exp.date);
      const matchesSearch = searchQuery === '' || 
        exp.title.toLowerCase().includes(searchQuery.toLowerCase()) ||
        exp.notes?.toLowerCase().includes(searchQuery.toLowerCase()) ||
        exp.category.toLowerCase().includes(searchQuery.toLowerCase());
      return matchesDate && matchesSearch;
    });
  }, [expenses, startDate, endDate, searchQuery]);

  const filteredVouchers = useMemo(() => {
    return vouchers.filter(v => {
      const matchesDate = isWithinDateRange(v.date);
      const matchesSearch = searchQuery === '' || 
        v.voucherNumber.toLowerCase().includes(searchQuery.toLowerCase()) ||
        v.notes?.toLowerCase().includes(searchQuery.toLowerCase());
      return matchesDate && matchesSearch;
    });
  }, [vouchers, startDate, endDate, searchQuery]);

  // 2. --- DETAILED CALCULATIONS ---

  // Sales & Profits Metrics
  const salesMetrics = useMemo(() => {
    let totalSalesVal = 0;
    let totalSalesQty = 0;
    let totalReturnsVal = 0;
    let totalDiscountsVal = 0;
    let totalCOGS = 0;
    let salesCount = filteredInvoices.length;

    // From Invoices
    filteredInvoices.forEach(inv => {
      totalSalesVal += inv.total;
      totalDiscountsVal += inv.discount;
      inv.items.forEach(it => {
        totalSalesQty += it.quantity;
        // Calculate COGS if product is found
        if (it.productId) {
          const prod = products.find(p => p.id === it.productId);
          if (prod) {
            totalCOGS += (prod.costPrice * it.quantity);
          }
        }
      });
    });

    // From Returns (SALE_RETURN)
    filteredTransactions.forEach(tx => {
      if (tx.type === 'SALE_RETURN') {
        totalReturnsVal += (tx.quantity * tx.unitPrice);
        const prod = products.find(p => p.id === tx.productId);
        if (prod) {
          totalCOGS -= (prod.costPrice * tx.quantity); // Deduct COGS for returned goods
        }
      }
    });

    const netSales = totalSalesVal - totalReturnsVal;
    const grossProfit = netSales - totalCOGS;
    const avgOrderValue = salesCount > 0 ? totalSalesVal / salesCount : 0;
    const profitMargin = netSales > 0 ? (grossProfit / netSales) * 100 : 0;

    // Payment methods breakdown
    let cashSales = 0;
    let cardSales = 0;
    let debtSales = 0;
    filteredInvoices.forEach(inv => {
      if (inv.paymentMethod === 'CASH') cashSales += inv.total;
      else if (inv.paymentMethod === 'CARD') cardSales += inv.total;
      else if (inv.paymentMethod === 'DEBT') debtSales += inv.total;
    });

    return {
      totalSalesVal,
      totalSalesQty,
      totalReturnsVal,
      totalDiscountsVal,
      totalCOGS,
      netSales,
      grossProfit,
      avgOrderValue,
      profitMargin,
      cashSales,
      cardSales,
      debtSales,
      salesCount
    };
  }, [filteredInvoices, filteredTransactions, products]);

  // Inventory & Valuation Metrics
  const inventoryMetrics = useMemo(() => {
    let totalItemsStock = 0;
    let totalCostValuation = 0;
    let totalRetailValuation = 0;
    let lowStockCount = 0;

    products.forEach(p => {
      totalItemsStock += p.quantity;
      totalCostValuation += (p.costPrice * p.quantity);
      totalRetailValuation += (p.sellingPrice * p.quantity);
      if (p.quantity <= p.minQuantity) {
        lowStockCount++;
      }
    });

    const potentialProfit = totalRetailValuation - totalCostValuation;
    const returnOnInvestment = totalCostValuation > 0 ? (potentialProfit / totalCostValuation) * 100 : 0;

    // Category distribution of stock value
    const categoryData: { [key: string]: { name: string; cost: number; qty: number } } = {};
    products.forEach(p => {
      const cat = categories.find(c => c.id === p.categoryId)?.name || (language === 'ar' ? 'عام' : 'General');
      if (!categoryData[p.categoryId]) {
        categoryData[p.categoryId] = { name: cat, cost: 0, qty: 0 };
      }
      categoryData[p.categoryId].cost += (p.costPrice * p.quantity);
      categoryData[p.categoryId].qty += p.quantity;
    });

    // Warehouse valuation
    const warehouseValuation: { [key: string]: { name: string; cost: number; qty: number } } = {};
    products.forEach(p => {
      const whId = p.warehouseId || 'default';
      const whName = warehouses.find(w => w.id === whId)?.name || (language === 'ar' ? 'المستودع الرئيسي' : 'Main Warehouse');
      if (!warehouseValuation[whId]) {
        warehouseValuation[whId] = { name: whName, cost: 0, qty: 0 };
      }
      warehouseValuation[whId].cost += (p.costPrice * p.quantity);
      warehouseValuation[whId].qty += p.quantity;
    });

    return {
      totalItemsStock,
      totalCostValuation,
      totalRetailValuation,
      potentialProfit,
      lowStockCount,
      returnOnInvestment,
      categoryValue: Object.values(categoryData),
      warehouseValue: Object.values(warehouseValuation)
    };
  }, [products, categories, warehouses, language]);

  // Expenses & Cash Metrics
  const cashMetrics = useMemo(() => {
    let totalSpentExpenses = 0;
    const categoryExpenses: { [key: string]: number } = {
      salaries: 0,
      rent: 0,
      utilities: 0,
      marketing: 0,
      other: 0
    };

    filteredExpenses.forEach(exp => {
      totalSpentExpenses += exp.amount;
      if (categoryExpenses[exp.category] !== undefined) {
        categoryExpenses[exp.category] += exp.amount;
      } else {
        categoryExpenses.other += exp.amount;
      }
    });

    // Safe Flow via Vouchers
    let totalVoucherReceipts = 0; // القبض
    let totalVoucherPayments = 0; // الصرف
    let cashierReceipts: { [key: string]: number } = {};

    filteredVouchers.forEach(v => {
      if (v.type === 'RECEIPT') {
        totalVoucherReceipts += v.amount;
      } else if (v.type === 'PAYMENT') {
        totalVoucherPayments += v.amount;
      }

      if (v.createdBy) {
        cashierReceipts[v.createdBy] = (cashierReceipts[v.createdBy] || 0) + v.amount;
      }
    });

    const netOperatingCashFlow = salesMetrics.netSales - totalSpentExpenses;

    return {
      totalSpentExpenses,
      categoryExpenses,
      totalVoucherReceipts,
      totalVoucherPayments,
      netOperatingCashFlow
    };
  }, [filteredExpenses, filteredVouchers, salesMetrics.netSales]);

  // Tax/VAT calculations (Saudi VAT is 15% standard)
  const taxMetrics = useMemo(() => {
    // Assuming prices in system are standard retail tax-inclusive, let's calculate back the tax (e.g. 15%)
    // Base amount = Total / 1.15
    // Tax amount = Total - Base amount
    const vatRate = 0.15; // 15% VAT

    const taxableSales = salesMetrics.netSales / (1 + vatRate);
    const collectedVat = salesMetrics.netSales - taxableSales;

    // Total Purchase cost from IN transactions
    let totalPurchasesVal = 0;
    filteredTransactions.forEach(tx => {
      if (tx.type === 'IN') {
        totalPurchasesVal += (tx.quantity * tx.unitPrice);
      } else if (tx.type === 'PURCHASE_RETURN') {
        totalPurchasesVal -= (tx.quantity * tx.unitPrice);
      }
    });

    const taxablePurchases = totalPurchasesVal / (1 + vatRate);
    const paidVat = totalPurchasesVal - taxablePurchases;

    const netVatPayable = collectedVat - paidVat;

    return {
      taxableSales,
      collectedVat,
      totalPurchasesVal,
      taxablePurchases,
      paidVat,
      netVatPayable
    };
  }, [salesMetrics.netSales, filteredTransactions]);

  // Entity & Staff Performance Metrics
  const performanceMetrics = useMemo(() => {
    // Cashier performance
    const cashierPerformance: { [key: string]: { name: string; sales: number; bills: number } } = {};
    
    // Seed cashiers
    teamMembers.forEach(m => {
      cashierPerformance[m.id] = { name: String(m?.name ?? '').split('|')[0].trim(), sales: 0, bills: 0 };
    });
    // Add default admin/unknown
    cashierPerformance['admin'] = { name: language === 'ar' ? 'المدير العام' : 'General Manager', sales: 0, bills: 0 };

    filteredInvoices.forEach(inv => {
      const cId = inv.cashierId || 'admin';
      if (!cashierPerformance[cId]) {
        cashierPerformance[cId] = { name: inv.cashierId || (language === 'ar' ? 'مبيعات خارجية' : 'External Sales'), sales: 0, bills: 0 };
      }
      cashierPerformance[cId].sales += inv.total;
      cashierPerformance[cId].bills += 1;
    });

    // Top selling products (accumulate quantities from invoices)
    const topProducts: { [key: string]: { name: string; qty: number; revenue: number } } = {};
    filteredInvoices.forEach(inv => {
      inv.items.forEach(it => {
        if (it.productId) {
          if (!topProducts[it.productId]) {
            topProducts[it.productId] = { name: it.name, qty: 0, revenue: 0 };
          }
          topProducts[it.productId].qty += it.quantity;
          topProducts[it.productId].revenue += (it.quantity * it.unitPrice);
        }
      });
    });

    // Top Debt Customers
    const customerDebts = customers
      .filter(c => c.balance > 0)
      .sort((a, b) => b.balance - a.balance)
      .slice(0, 10);

    // Top Supplier Debts
    const supplierPayables = suppliers
      .filter(s => (s.balance || 0) < 0)
      .sort((a, b) => (a.balance || 0) - (b.balance || 0)) // larger negative first
      .slice(0, 10);

    return {
      cashiers: Object.values(cashierPerformance).filter(c => c.bills > 0 || c.sales > 0),
      topProducts: Object.values(topProducts).sort((a, b) => b.revenue - a.revenue).slice(0, 10),
      customerDebts,
      supplierPayables
    };
  }, [filteredInvoices, teamMembers, customers, suppliers, language]);


  // 3. --- PRINT AND COPY PLAIN TEXT UTILITIES ---

  const generateTextReport = () => {
    let reportTitle = '';
    let reportContent = '';

    const header = `==================================================\n` +
                   `          ${companyName || 'AI ERP SYSTEM'}          \n` +
                   `               ${language === 'ar' ? 'تقرير تفصيلي شامل' : 'Comprehensive ERP Report'} \n` +
                   `==================================================\n` +
                   `${language === 'ar' ? 'تاريخ الاستخراج:' : 'Date Issued:'} ${new Date().toLocaleString()}\n` +
                   `${language === 'ar' ? 'الفترة الزمنية:' : 'Period:'} ${startDate} ${language === 'ar' ? 'إلى' : 'to'} ${endDate}\n` +
                   `--------------------------------------------------\n\n`;

    if (activeReportTab === 'sales_profits') {
      reportTitle = language === 'ar' ? 'تقرير المبيعات والأرباح' : 'Sales & Profits Report';
      reportContent = 
        `[1] ${language === 'ar' ? 'المؤشرات المالية الرئيسية:' : 'Key Financial Indicators:'}\n` +
        ` - ${language === 'ar' ? 'إجمالي المبيعات:' : 'Gross Sales:'} ${currencySymbol}${salesMetrics.totalSalesVal.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'المرتجعات المخصومة:' : 'Sales Returns:'} ${currencySymbol}${salesMetrics.totalReturnsVal.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'صافي قيمة المبيعات:' : 'Net Revenue:'} ${currencySymbol}${salesMetrics.netSales.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'تكلفة البضاعة المباعة COGS:' : 'Cost of Goods Sold (COGS):'} ${currencySymbol}${salesMetrics.totalCOGS.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'مجمل الربح التشغيلي:' : 'Gross Operating Profit:'} ${currencySymbol}${salesMetrics.grossProfit.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'هامش الربح الكلي:' : 'Overall Profit Margin:'} ${salesMetrics.profitMargin.toFixed(1)}%\n` +
        ` - ${language === 'ar' ? 'متوسط قيمة الفاتورة:' : 'Average Ticket Size:'} ${currencySymbol}${salesMetrics.avgOrderValue.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'عدد الفواتير المصدرة:' : 'Invoices Count:'} ${salesMetrics.salesCount}\n\n` +
        `[2] ${language === 'ar' ? 'تحليل وسائل الدفع المقبولة:' : 'Payment Methods Breakdown:'}\n` +
        ` - ${language === 'ar' ? 'نقدي (كاش):' : 'Cash Safe:'} ${currencySymbol}${salesMetrics.cashSales.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'شبكة وعبر البنك:' : 'Card & Bank POS:'} ${currencySymbol}${salesMetrics.cardSales.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'آجل مبيعات مديونية:' : 'Debt (Outstanding Credit):'} ${currencySymbol}${salesMetrics.debtSales.toFixed(2)}\n`;
    } 
    else if (activeReportTab === 'inventory_valuation') {
      reportTitle = language === 'ar' ? 'تقرير تقييم حركة المخزون' : 'Inventory Valuation Report';
      reportContent = 
        `[1] ${language === 'ar' ? 'ملخص موازنة وحركة المستودعات:' : 'Warehouse & Stock Summary:'}\n` +
        ` - ${language === 'ar' ? 'إجمالي عدد القطع المتوفرة:' : 'Total Pieces in Stock:'} ${inventoryMetrics.totalItemsStock} ${language === 'ar' ? 'قطعة' : 'units'}\n` +
        ` - ${language === 'ar' ? 'تقييم المخزون بسعر التكلفة:' : 'Stock Valuation (At Cost):'} ${currencySymbol}${inventoryMetrics.totalCostValuation.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'تقييم المخزون بسعر البيع النهائي:' : 'Stock Valuation (At Retail):'} ${currencySymbol}${inventoryMetrics.totalRetailValuation.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'الأرباح الكامنة المتوقعة:' : 'Potential Projected Profit:'} ${currencySymbol}${inventoryMetrics.potentialProfit.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'العائد المتوقع للاستثمار ROI:' : 'Projected ROI:'} ${inventoryMetrics.returnOnInvestment.toFixed(1)}%\n` +
        ` - ${language === 'ar' ? 'عدد الأصناف شحيحة المخزون:' : 'Low Stock Alert Items Count:'} ${inventoryMetrics.lowStockCount}\n\n` +
        `[2] ${language === 'ar' ? 'توزيع المخزون حسب الفروع والمخازن:' : 'Warehouse Inventory Breakdown:'}\n` +
        inventoryMetrics.warehouseValue.map(wh => ` - ${wh.name}: Qty ${wh.qty} | Value at Cost: ${currencySymbol}${wh.cost.toFixed(2)}`).join('\n') + '\n';
    } 
    else if (activeReportTab === 'expenses_cash') {
      reportTitle = language === 'ar' ? 'تقرير المصاريف والسيولة' : 'Expenses & Cashflow Report';
      reportContent = 
        `[1] ${language === 'ar' ? 'ملخص المصروفات التشغيلية:' : 'Overhead Operating Expenses:'}\n` +
        ` - ${language === 'ar' ? 'إجمالي المصاريف المصروفة:' : 'Total Spent Expenses:'} ${currencySymbol}${cashMetrics.totalSpentExpenses.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'مصاريف الرواتب والأجور:' : 'Salaries & Wages Overhead:'} ${currencySymbol}${cashMetrics.categoryExpenses.salaries.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'إيجار العقارات والمقرات:' : 'Rent Expenses:'} ${currencySymbol}${cashMetrics.categoryExpenses.rent.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'الخدمات العامة والمرافق:' : 'Utilities & Power:'} ${currencySymbol}${cashMetrics.categoryExpenses.utilities.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'التسويق والدعاية:' : 'Marketing overhead:'} ${currencySymbol}${cashMetrics.categoryExpenses.marketing.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'مصاريف وبنود أخرى:' : 'Other Miscellaneous:'} ${currencySymbol}${cashMetrics.categoryExpenses.other.toFixed(2)}\n\n` +
        `[2] ${language === 'ar' ? 'حركات السندات المالية:' : 'Financial Safe Vouchers:'}\n` +
        ` - ${language === 'ar' ? 'إجمالي المقبوضات (سندات القبض):' : 'Total Receipts (Inflow):'} ${currencySymbol}${cashMetrics.totalVoucherReceipts.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'إجمالي المدفوعات (سندات الصرف):' : 'Total Payments (Outflow):'} ${currencySymbol}${cashMetrics.totalVoucherPayments.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'التدفق المالي الصافي المتبقي:' : 'Net Safe Operations Cashflow:'} ${currencySymbol}${cashMetrics.netOperatingCashFlow.toFixed(2)}\n`;
    } 
    else if (activeReportTab === 'tax_vat') {
      reportTitle = language === 'ar' ? 'التقرير الضريبي التفصيلي (ضريبة القيمة المضافة)' : 'Detailed VAT Report';
      reportContent = 
        `[1] ${language === 'ar' ? 'ضريبة القيمة المضافة المحصلة (المبيعات) - 15%:' : 'Output VAT (On Sales) - 15%:'}\n` +
        ` - ${language === 'ar' ? 'إجمالي المبيعات الخاضعة للضريبة:' : 'Total Taxable Sales (Base):'} ${currencySymbol}${taxMetrics.taxableSales.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'ضريبة القيمة المضافة المحصلة:' : 'Collected Output VAT:'} ${currencySymbol}${taxMetrics.collectedVat.toFixed(2)}\n\n` +
        `[2] ${language === 'ar' ? 'ضريبة القيمة المضافة المدفوعة (المشتريات) - 15%:' : 'Input VAT (On Purchases) - 15%:'}\n` +
        ` - ${language === 'ar' ? 'إجمالي المشتريات الخاضعة للضريبة:' : 'Total Taxable Purchases (Base):'} ${currencySymbol}${taxMetrics.taxablePurchases.toFixed(2)}\n` +
        ` - ${language === 'ar' ? 'ضريبة القيمة المضافة المدفوعة للموردين:' : 'Paid Input VAT:'} ${currencySymbol}${taxMetrics.paidVat.toFixed(2)}\n\n` +

                {language === 'ar' 
                  ? 'تم حساب القيمة المضافة المحصلة والمدفوعة بالاعتماد على النسبة القياسية للضريبة (15%) بناء على الفواتير وحركات الشراء والبيع التفصيلية المجرية في الفترة الزمنية المحددة بالفلتر.' 
                  : 'VAT calculation is estimated based on the standard 15% rate applied to net sales and purchase movements in the selected range.'}
              </p>
            </div>

            {/* Metrics cards */}
            <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
              <div className="bg-slate-50 rounded-2xl border border-slate-150 p-4">
                <span className="text-[10px] text-slate-450 font-extrabold uppercase tracking-wider block">
                  {language === 'ar' ? 'إجمالي المبيعات الخاضعة للضريبة' : 'Total Taxable Sales (Base)'}
                </span>
                <span className="text-xl font-black text-slate-900 block mt-1">
                  {currencySymbol}{taxMetrics.taxableSales.toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 })}
                </span>
                <span className="text-[10px] text-slate-450 mt-1 block">
                  {language === 'ar' ? 'المجموع خالي من الضريبة' : 'Excluding 15% VAT'}
                </span>
                <div className="mt-2.5 pt-2 border-t flex justify-between text-xs text-indigo-600 font-bold">
                  <span>{language === 'ar' ? 'ضريبة محصلة (مخرجات):' : 'Output VAT Collected:'}</span>
                  <span>{currencySymbol}{taxMetrics.collectedVat.toFixed(2)}</span>
                </div>
              </div>

              <div className="bg-slate-50 rounded-2xl border border-slate-150 p-4">
                <span className="text-[10px] text-slate-450 font-extrabold uppercase tracking-wider block">
                  {language === 'ar' ? 'إجمالي المشتريات الخاضعة للضريبة' : 'Total Taxable Purchases (Base)'}
                </span>
                <span className="text-xl font-black text-slate-750 block mt-1">
                  {currencySymbol}{taxMetrics.taxablePurchases.toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 })}
                </span>
                <span className="text-[10px] text-slate-450 mt-1 block">
                  {language === 'ar' ? 'المجموع خالي من الضريبة' : 'Excluding 15% VAT'}
                </span>
                <div className="mt-2.5 pt-2 border-t flex justify-between text-xs text-rose-500 font-bold">
                  <span>{language === 'ar' ? 'ضريبة مدفوعة (مدخلات):' : 'Input VAT Paid:'}</span>
                  <span>{currencySymbol}{taxMetrics.paidVat.toFixed(2)}</span>
                </div>
              </div>

              <div className="bg-slate-50 rounded-2xl border border-slate-150 p-4 bg-teal-50/40 border-teal-200">
                <span className="text-[10px] text-teal-650 font-extrabold uppercase tracking-wider block">
                  {language === 'ar' ? 'صافي المخرجات الضريبية المستحقة' : 'Net Tax Payable'}
                </span>
                <span className="text-xl font-black text-teal-700 block mt-1">
                  {currencySymbol}{taxMetrics.netVatPayable.toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 })}
                </span>
                <span className="text-[10px] text-teal-600 mt-1 block">
                  {language === 'ar' ? 'الواجب سدادها لهيئة الزكاة والضريبة' : 'Net payable to authority'}
                </span>
                <div className="mt-2.5 pt-2 border-t flex justify-between text-xs text-teal-600 font-bold">
                  <span>{language === 'ar' ? 'الوضع الضريبي الحالي:' : 'Tax Status:'}</span>
                  <span>{taxMetrics.netVatPayable >= 0 ? (language === 'ar' ? 'سداد مستحق' : 'Payable') : (language === 'ar' ? 'رصيد دائن' : 'Credit / Refund')}</span>
                </div>
              </div>
            </div>
          </div>
        )}

        {/* TAB 5: Performance & Receivables/Payables */}
        {activeReportTab === 'performance_entities' && (
          <div className="space-y-6">
            <div className="border-b pb-3 mb-4">
              <h3 className="font-extrabold text-sm text-slate-800 flex items-center gap-1.5">
                <Users className="w-5 h-5 text-sky-500" />
                <span>{language === 'ar' ? 'تقرير تحليل أداء الكوادر والمبيعات والديون المترابطة' : 'Wages, Cashier Performance & Debt Analytics'}</span>
              </h3>
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
              
              {/* Cashiers Leaderboard */}
              <div className="bg-slate-50 rounded-2xl p-4 border border-slate-150">
                <h4 className="font-extrabold text-xs text-slate-700 border-b pb-2 mb-3 flex items-center justify-between">
                  <span>{language === 'ar' ? 'لائحة إنتاجية موظفي الكاشير والمبيعات' : 'Salesmen & Cashiers Leaderboard'}</span>
                  <span className="text-[9px] bg-indigo-50 text-indigo-600 px-1.5 py-0.5 rounded uppercase font-bold">Live Link</span>
                </h4>
                <div className="space-y-3">
                  {performanceMetrics.cashiers.length === 0 ? (
                    <p className="text-center text-slate-400 p-4 text-xs">{language === 'ar' ? 'لم تسجل مبيعات لأي موظف كاشير مخصص بعد' : 'No sales registered yet.'}</p>
                  ) : (
                    performanceMetrics.cashiers.map(c => (
                      <div key={c.name} className="flex justify-between items-center text-xs border-b pb-2 last:border-none">
                        <div className="space-y-0.5">
                          <span className="font-bold text-slate-800 block">{c.name}</span>
                          <span className="text-[10px] text-slate-400 font-bold">
                            {c.bills} {language === 'ar' ? 'فواتير مصدرة' : 'completed invoices'}
                          </span>
                        </div>
                        <span className="font-extrabold text-indigo-600 font-mono text-xs">
                          {currencySymbol}{c.sales.toFixed(2)}
                        </span>
                      </div>
                    ))
                  )}
                </div>
              </div>

              {/* Best selling products */}
              <div className="bg-slate-50 rounded-2xl p-4 border border-slate-150">
                <h4 className="font-extrabold text-xs text-slate-700 border-b pb-2 mb-3 flex items-center justify-between">
                  <span>{language === 'ar' ? 'الأصناف الأكثر طلباً ومبيعاً (الأعلى ربحاً)' : 'Best Selling Products List (Highest Revenue)'}</span>
                  <span className="text-[9px] bg-amber-50 text-amber-600 px-1.5 py-0.5 rounded uppercase font-bold">Top 10</span>
                </h4>
                <div className="space-y-3">
                  {performanceMetrics.topProducts.length === 0 ? (
                    <p className="text-center text-slate-400 p-4 text-xs">{language === 'ar' ? 'لا توجد مبيعات متوفرة حالياً' : 'No products sold in this period.'}</p>
                  ) : (
                    performanceMetrics.topProducts.map((p, i) => (
                      <div key={p.name} className="flex justify-between items-center text-xs border-b pb-2 last:border-none">
                        <div className="flex items-center gap-2">
                          <span className="w-5 h-5 rounded bg-indigo-50 text-indigo-600 font-bold flex items-center justify-center text-[10px]">{i + 1}</span>
                          <span className="font-bold text-slate-750">{p.name}</span>
                        </div>
                        <div className="text-end font-mono">
                          <span className="font-bold text-slate-800">{currencySymbol}{p.revenue.toFixed(2)}</span>
                          <span className="text-[9px] text-slate-400 block">
                            {p.qty} {language === 'ar' ? 'قطعة مباعة' : 'units sold'}
                          </span>
                        </div>
                      </div>
                    ))
                  )}
                </div>
              </div>

              {/* Top Debt Clients */}
              <div className="bg-slate-50 rounded-2xl p-4 border border-slate-150">
                <h4 className="font-extrabold text-xs text-slate-700 border-b pb-2 mb-3">
                  {language === 'ar' ? 'العملاء المستحق عليهم مبالغ آجلة (المديونيات الواجب تحصيلها)' : 'Outstanding Receivables (Top Customer Debtors)'}
                </h4>
                <div className="space-y-3">
                  {performanceMetrics.customerDebts.length === 0 ? (
                    <p className="text-center text-slate-400 p-4 text-xs">{language === 'ar' ? 'لا توجد مديونيات أو مستحقات آجلة متأخرة على العملاء' : 'Excellent! No outstanding customer debts.'}</p>
                  ) : (
                    performanceMetrics.customerDebts.map(c => (
                      <div key={c.id} className="flex justify-between items-center text-xs border-b pb-2 last:border-none">
                        <div>
                          <span className="font-bold text-slate-800 block">{c.name}</span>
                          <span className="text-[9px] text-slate-400 block">{c.phone || (language === 'ar' ? 'لا يوجد هاتف' : 'No phone')}</span>
                        </div>
                        <span className="font-mono font-bold text-rose-600 bg-rose-50 px-2.5 py-0.5 rounded-full text-[10px]">
                          {currencySymbol}{c.balance.toFixed(2)}
                        </span>
                      </div>
                    ))
                  )}
                </div>
              </div>

              {/* Suppliers payable */}
              <div className="bg-slate-50 rounded-2xl p-4 border border-slate-150">
                <h4 className="font-extrabold text-xs text-slate-700 border-b pb-2 mb-3">
                  {language === 'ar' ? 'التزامات ومستحقات الموردين الآجلة (المبالغ المستحق دفعها)' : 'Supplier Payables (Outstanding Balances We Owe)'}
                </h4>
                <div className="space-y-3">
                  {performanceMetrics.supplierPayables.length === 0 ? (
                    <p className="text-center text-slate-400 p-4 text-xs">{language === 'ar' ? 'كل الالتزامات مع الموردين مسواة بالكامل' : 'No outstanding payables to suppliers.'}</p>
                  ) : (
                    performanceMetrics.supplierPayables.map(s => (
                      <div key={s.id} className="flex justify-between items-center text-xs border-b pb-2 last:border-none">
                        <div>
                          <span className="font-bold text-slate-800 block">{s.name}</span>
                          <span className="text-[9px] text-slate-400 block">{s.company || (language === 'ar' ? 'لا توجد شركة مخصصة' : 'No company specified')}</span>
                        </div>
                        <span className="font-mono font-bold text-orange-600 bg-orange-50 px-2.5 py-0.5 rounded-full text-[10px]">
                          {currencySymbol}{Math.abs(s.balance || 0).toFixed(2)}
                        </span>
                      </div>
                    ))
                  )}
                </div>
              </div>

            </div>
          </div>
        )}

      </div>

    </div>
  );
}
