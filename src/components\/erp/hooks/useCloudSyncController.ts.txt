import { useCallback, useEffect, useRef } from 'react';
import { generateBackupPayload, parseBackupPayload } from '../services/mobileBackupCodec';
import { computeCloudFingerprint, deterministicIdToInteger, getTransactionSignature, mergeById, normalizeCloudId, resolveCloudUrl } from '../services/cloudSyncUtils';

type CloudAction = 'sync' | 'backup' | 'restore';

export function useCloudSyncController(args: any) {
  const {
    products, categories, suppliers, transactions, customers, warehouses, transfers, expenses,
    teamMembers, partners, distributions, vouchers, invoices, services, chatHistory, companyName, companyPhone, companyAddress,
    setProducts, setCategories, setSuppliers, setTransactions, setCustomers, setWarehouses,
    setTransfers, setExpenses, setTeamMembers, setPartners, setDistributions, setVouchers, setInvoices, setServices, setChatHistory,
    setCompanyName, setCompanyPhone, setCompanyAddress,
    firebaseUrl, teamId, cloudSyncType, isAutoSyncEnabled, isLiveMonitoringActive,
    setIsCloudSyncing, setLastCloudSyncTime, addCloudSyncLog, setLiveNotifications,
    language, showToast, playBeep,
  } = args;

  const latestRef = useRef<any>({});
  const fingerprintRef = useRef('');
  const applyingCloudRef = useRef(false);
  const pollInFlightRef = useRef(false);
  const uploadTimerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  useEffect(() => {
    latestRef.current = { products, categories, suppliers, transactions, customers, warehouses, transfers, expenses, teamMembers, partners, distributions, vouchers, invoices, services, chatHistory };
  }, [products, categories, suppliers, transactions, customers, warehouses, transfers, expenses, teamMembers, partners, distributions, vouchers, invoices, services, chatHistory]);

  const buildPayload = useCallback(() => generateBackupPayload({
    latestStates: latestRef.current,
    deterministicIdToInteger,
    companyName,
    companyPhone,
    companyAddress,
    services: latestRef.current.services,
    transfers: latestRef.current.transfers,
  }), [companyName, companyPhone, companyAddress]);

  const markSynced = useCallback(() => {
    const now = new Date().toLocaleString();
    setLastCloudSyncTime(now);
    localStorage.setItem('erp_last_cloud_sync_time', now);
  }, [setLastCloudSyncTime]);

  const applyParsed = useCallback((parsed: any, mode: 'replace' | 'merge') => {
    applyingCloudRef.current = true;
    const update = (setter: any, values: any[]) => setter?.((previous: any[]) => mode === 'replace' ? values : mergeById(previous, values));
    update(setWarehouses, parsed.importedWarehouses || []);
    update(setCustomers, parsed.importedCustomers || []);
    update(setSuppliers, parsed.importedSuppliers || []);
    update(setCategories, parsed.importedCategories || []);
    update(setProducts, parsed.importedProducts || []);
    update(setExpenses, parsed.importedExpenses || []);
    if (mode === 'replace') setTransactions?.(parsed.importedTransactions || []);
    else setTransactions?.((previous: any[]) => {
      const ids = new Set(previous.map(item => normalizeCloudId(item.id)));
      const signatures = new Set(previous.map(getTransactionSignature));
      return [...previous, ...(parsed.importedTransactions || []).filter((item: any) => !ids.has(normalizeCloudId(item.id)) && !signatures.has(getTransactionSignature(item)))];
    });
    update(setVouchers, parsed.importedVouchers || []);
    if (mode === 'replace') setInvoices?.(parsed.importedInvoices || []);
    else setInvoices?.((previous: any[]) => {
      const ids = new Set(previous.map(item => normalizeCloudId(item.id)));
      const numbers = new Set(previous.map(item => normalizeCloudId(item.invoiceNo)));
      return [...previous, ...(parsed.importedInvoices || []).filter((item: any) => !ids.has(normalizeCloudId(item.id)) && (!item.invoiceNo || !numbers.has(normalizeCloudId(item.invoiceNo))))];
    });
    update(setServices, parsed.importedServices || []);
    update(setTransfers, parsed.importedTransfers || []);
    if (Array.isArray(parsed.importedTeamMembers)) update(setTeamMembers, parsed.importedTeamMembers);
    if (Array.isArray(parsed.importedPartners)) update(setPartners, parsed.importedPartners);
    if (Array.isArray(parsed.importedDistributions)) update(setDistributions, parsed.importedDistributions);
    if (Array.isArray(parsed.importedChatHistory)) update(setChatHistory, parsed.importedChatHistory);
    if (mode === 'replace' && parsed.settings) {
      if (parsed.settings.storeName) setCompanyName?.(parsed.settings.storeName);
      if (parsed.settings.projectPhone) setCompanyPhone?.(parsed.settings.projectPhone);
      if (parsed.settings.projectAbout) setCompanyAddress?.(parsed.settings.projectAbout);
    }
    window.setTimeout(() => { applyingCloudRef.current = false; }, 1200);
  }, [setWarehouses, setCustomers, setSuppliers, setCategories, setProducts, setExpenses, setTransactions, setVouchers, setInvoices, setServices, setTransfers, setTeamMembers, setPartners, setDistributions, setChatHistory, setCompanyName, setCompanyPhone, setCompanyAddress]);

  const uploadPayload = useCallback(async (url: string, payload: any) => {
    const response = await fetch(url, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
    if (!response.ok) throw new Error(`Upload failed with code: ${response.status}`);
    fingerprintRef.current = computeCloudFingerprint(payload, parseBackupPayload);
    markSynced();
  }, [markSynced]);

  const handleCloudSyncAction = useCallback(async (action: CloudAction) => {
    if (!teamId || !firebaseUrl) {
      showToast(language === 'ar' ? 'أدخل رابط Firebase وكود الفريق أولاً' : 'Enter Firebase URL and team code first', 'error');
      return;
    }
    const url = resolveCloudUrl(firebaseUrl, teamId, cloudSyncType === 'default');
    setIsCloudSyncing(true);
    addCloudSyncLog(language === 'ar' ? 'بدء الاتصال بالخادم السحابي...' : 'Connecting to cloud server...');
    try {
      const response = await fetch(url);
      if (!response.ok) throw new Error(`Cloud server returned code: ${response.status}`);
      const cloudPayload = await response.json();
      const cloudEmpty = !cloudPayload || Object.keys(cloudPayload).length === 0;
      if (action === 'backup') {
        await uploadPayload(url, buildPayload());
      } else if (action === 'restore') {
        if (cloudEmpty) throw new Error(language === 'ar' ? 'السحابة فارغة' : 'Cloud is empty');
        applyParsed(parseBackupPayload(cloudPayload), 'replace');
        fingerprintRef.current = computeCloudFingerprint(cloudPayload, parseBackupPayload);
        markSynced();
      } else {
        if (!cloudEmpty) applyParsed(parseBackupPayload(cloudPayload), 'merge');
        // React state/ref reconciliation may require more than one microtask. Two animation frames
        // ensure latestRef contains the merged state before it is uploaded back to Firebase.
        await new Promise<void>(resolve => window.requestAnimationFrame(() => window.requestAnimationFrame(() => resolve())));
        await uploadPayload(url, buildPayload());
      }
      addCloudSyncLog(language === 'ar' ? 'اكتملت العملية السحابية بنجاح.' : 'Cloud operation completed successfully.');
      showToast(language === 'ar' ? 'اكتملت المزامنة السحابية بنجاح' : 'Cloud synchronization completed', 'success');
      playBeep('success');
    } catch (error: any) {
      addCloudSyncLog(`⚠️ ${error?.message || error}`);
      showToast(language === 'ar' ? 'فشل الاتصال أو المزامنة السحابية' : 'Cloud synchronization failed', 'error');
      playBeep('error');
    } finally {
      setIsCloudSyncing(false);
    }
  }, [teamId, firebaseUrl, cloudSyncType, setIsCloudSyncing, addCloudSyncLog, language, showToast, playBeep, uploadPayload, buildPayload, applyParsed, markSynced]);

  useEffect(() => {
    try { fingerprintRef.current = computeCloudFingerprint(buildPayload(), parseBackupPayload); } catch { fingerprintRef.current = ''; }
  }, []);

  useEffect(() => {
    if (!isAutoSyncEnabled || applyingCloudRef.current || !teamId || !firebaseUrl) return;
    if (uploadTimerRef.current) clearTimeout(uploadTimerRef.current);
    uploadTimerRef.current = setTimeout(async () => {
      try {
        const payload = buildPayload();
        const next = computeCloudFingerprint(payload, parseBackupPayload);
        if (next === fingerprintRef.current) return;
        await uploadPayload(resolveCloudUrl(firebaseUrl, teamId, cloudSyncType === 'default'), payload);
        addCloudSyncLog(language === 'ar' ? '💾 تم الحفظ التلقائي في السحابة.' : '💾 Auto-saved to cloud.');
      } catch (error) { console.warn('Auto cloud backup failed:', error); }
    }, 2500);
    return () => { if (uploadTimerRef.current) clearTimeout(uploadTimerRef.current); };
  }, [products, categories, suppliers, transactions, customers, warehouses, transfers, expenses, teamMembers, partners, distributions, vouchers, invoices, services, chatHistory, isAutoSyncEnabled, teamId, firebaseUrl, cloudSyncType, buildPayload, uploadPayload, addCloudSyncLog, language]);

  useEffect(() => {
    if (!isLiveMonitoringActive || !teamId || !firebaseUrl) return;
    let disposed = false;
    const poll = async () => {
      if (disposed || pollInFlightRef.current || document.visibilityState !== 'visible') return;
      pollInFlightRef.current = true;
      try {
        const url = resolveCloudUrl(firebaseUrl, teamId, cloudSyncType === 'default');
        const response = await fetch(url);
        if (!response.ok) return;
        const payload = await response.json();
        if (!payload || Object.keys(payload).length === 0) return;
        const next = computeCloudFingerprint(payload, parseBackupPayload);
        if (fingerprintRef.current && next !== fingerprintRef.current) {
          const before = latestRef.current;
          const parsed = parseBackupPayload(payload);
          applyParsed(parsed, 'merge');
          fingerprintRef.current = next;
          const addedCount = Math.max(0, (parsed.importedProducts?.length || 0) - (before.products?.length || 0)) + Math.max(0, (parsed.importedTransactions?.length || 0) - (before.transactions?.length || 0));
          const message = language === 'ar' ? `☁️ وصلت تحديثات سحابية${addedCount ? ` (${addedCount})` : ''}` : `☁️ Cloud updates received${addedCount ? ` (${addedCount})` : ''}`;
          setLiveNotifications?.((previous: any[]) => [{ id: `${Date.now()}`, message, timestamp: new Date() }, ...previous].slice(0, 20));
        } else if (!fingerprintRef.current) fingerprintRef.current = next;
      } catch (error) { console.warn('Cloud live polling failed:', error); }
      finally { pollInFlightRef.current = false; }
    };
    poll();
    const onFocus = () => poll();
    window.addEventListener('focus', onFocus);
    const interval = window.setInterval(poll, 5000);
    return () => { disposed = true; window.removeEventListener('focus', onFocus); window.clearInterval(interval); };
  }, [isLiveMonitoringActive, teamId, firebaseUrl, cloudSyncType, applyParsed, language, setLiveNotifications]);

  return { handleCloudSyncAction };
}
