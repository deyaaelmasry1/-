import { useCallback, useEffect, useRef, useState } from 'react';
import type { RepositoryPageSource } from '../../components/erp/repositories/BaseRepository';
import type { PageResult } from '../../utils/desktopDatabase';

const EMPTY: PageResult<never> = { rows: [], count: 0, page: 1, pageSize: 50, pages: 1 };

export function useEntityPage<T>(repo: RepositoryPageSource<T>, pageSize = 50) {
  const [result, setResult] = useState<PageResult<T>>({ ...EMPTY, pageSize } as PageResult<T>);
  const [search, setSearch] = useState('');
  const [loading, setLoading] = useState(false);
  const requestId = useRef(0);

  const load = useCallback(async (page = 1, term = search) => {
    const id = ++requestId.current;
    setLoading(true);
    try {
      const next = await repo.page({ page, pageSize, search: term });
      if (id === requestId.current) setResult(next);
    } finally {
      if (id === requestId.current) setLoading(false);
    }
  }, [pageSize, repo, search]);

  useEffect(() => {
    const timer = window.setTimeout(() => void load(1, search), 180);
    return () => window.clearTimeout(timer);
  }, [load, search]);

  return {
    ...result,
    search,
    setSearch,
    loading,
    reload: () => load(result.page, search),
    goToPage: (page: number) => load(page, search),
  };
}
