export interface Product {
  id: string;
  code: string; // SKU or Barcode
  name: string;
  categoryId: string;
  quantity: number;
  minQuantity: number; // Reorder level alert
  costPrice: number;
  sellingPrice: number;
  supplierId?: string;
  location?: string; // e.g., Shelf A4
  description?: string;
  warehouseId?: string; // Currently stored warehouse
}

export interface Category {
  id: string;
  name: string;
}

export type TransactionType = 'IN' | 'OUT' | 'SALE_RETURN' | 'PURCHASE_RETURN' | 'SAFE_DEPOSIT' | 'SAFE_WITHDRAWAL';

export interface ProfessionalService {
  id: string;
  name: string;
  type: 'shipping' | 'maintenance' | 'craft' | 'profession' | 'other';
  price: number;
  cost: number;
  dateReceived: string; // ISO date string (YYYY-MM-DD)
  dateDelivery: string; // ISO date string (YYYY-MM-DD)
  status: 'PENDING' | 'IN_PROGRESS' | 'COMPLETED' | 'DELIVERED';
  customerId?: string;
  description?: string;
}

export interface Transaction {
  id: string;
  productId?: string;
  serviceId?: string;
  type: TransactionType;
  quantity: number;
  unitPrice: number;
  date: string; // ISO date string
  notes?: string;
  customerId?: string; // link to customer for sale/return
  supplierId?: string; // link to supplier for purchase/return
  warehouseId?: string; // relevant warehouse location
  cashierId?: string; // team member who performed the transaction
  paymentMethod?: string; // payment method used (e.g., CASH, CARD, DEBT, INSTALLMENT, SETTLEMENT)
}

export interface Supplier {
  id: string;
  name: string;
  company?: string;
  phone?: string;
  email?: string;
  address?: string;
  balance?: number; // negative means we owe supplier, positive means credit
}

export interface Customer {
  id: string;
  name: string;
  phone?: string;
  email?: string;
  address?: string;
  balance: number; // outstanding customer debt/balance
}

export interface Warehouse {
  id: string;
  name: string;
  location?: string;
}

export interface WarehouseTransfer {
  id: string;
  productId: string;
  fromWarehouseId: string;
  toWarehouseId: string;
  quantity: number;
  date: string;
  notes?: string;
  performedBy?: string;
  performedById?: string;
}

export interface Expense {
  id: string;
  title: string;
  amount: number;
  category: 'salaries' | 'rent' | 'utilities' | 'marketing' | 'other';
  date: string;
  notes?: string;
}

export interface TeamMember {
  id: string;
  name: string;
  role: 'admin' | 'cashier' | 'storekeeper';
  phone?: string;
  active: boolean;
  username?: string;
  password?: string;
}

export interface Partner {
  id: string;
  name: string;
  equityPercentage: number; // e.g., 40 for 40%
  investedAmount: number;
  notes?: string;
}

export interface PartnerDistribution {
  id: string;
  partnerId: string; // 'all' or specific Partner ID
  partnerName: string; // 'الكل' or specific Partner Name
  amount: number;
  date: string;
  notes?: string;
  status: 'PAID' | 'PENDING';
}

export interface FinancialVoucher {
  id: string;
  voucherNumber: string; // e.g., VOUCH-1001
  type: 'RECEIPT' | 'PAYMENT'; // RECEIPT (قبض), PAYMENT (صرف)
  partyType: 'CUSTOMER' | 'SUPPLIER' | 'OTHER';
  partyId?: string; // customerId or supplierId
  amount: number;
  date: string;
  paymentMethod: 'CASH' | 'BANK' | 'CHECK' | 'SETTLEMENT';
  notes?: string;
  referenceInvoiceId?: string; // link to transaction/invoice
  createdBy?: string; // name of team member
}

export interface ChatMessage {
  id: string;
  role: 'user' | 'model';
  text: string;
  timestamp: string;
}

export interface InvoiceItem {
  id: string;
  name: string;
  type: 'product' | 'service';
  productId?: string;
  serviceId?: string;
  quantity: number;
  unitPrice: number;
}

export interface Invoice {
  id: string;
  invoiceNo: string;
  customerName: string;
  customerPhone: string;
  customerId?: string;
  date: string;
  paymentMethod: 'CASH' | 'CARD' | 'DEBT' | 'INSTALLMENT';
  discount: number;
  items: InvoiceItem[];
  subtotal: number;
  total: number;
  cashierId?: string;
}
