export type PageResult<T> = { rows: T[]; count: number; page: number; pageSize: number; pages: number };

const api = () => window.electronAPI?.database;
export const desktopDatabase = {
  available: () => Boolean(api()),
  list: <T>(args: Record<string, unknown>) => api()!.list(args) as Promise<PageResult<T>>,
  createQuotation: (data: unknown) => api()!.createQuotation(data),
  updateQuotation: (data: unknown) => api()!.updateQuotation(data),
  getQuotation: (id: string) => api()!.getQuotation(id),
  updateQuotationStatus: (data: unknown) => api()!.updateQuotationStatus(data),
  createInstallment: (data: unknown) => api()!.createInstallment(data),
  collectInstallment: (data: unknown) => api()!.collectInstallment(data),
  savePayrollEmployee: (data: unknown) => api()!.savePayrollEmployee(data),
  addPayrollAdjustment: (data: unknown) => api()!.addPayrollAdjustment(data),
  generatePayrollRun: (data: unknown) => api()!.generatePayrollRun(data),
  setPayrollRunStatus: (data: unknown) => api()!.setPayrollRunStatus(data),
  getPayrollRun: (id: string) => api()!.getPayrollRun(id),
  saveAttendanceDevice: (data: unknown) => api()!.saveAttendanceDevice(data),
  recordAttendance: (data: unknown) => api()!.recordAttendance(data),
  importAttendance: (rows: unknown[]) => api()!.importAttendance(rows),
  saveEmployeeSignature: (data: unknown) => api()!.saveEmployeeSignature(data),
  attendanceSummary: (args?: Record<string, unknown>) => api()!.attendanceSummary(args),
  testBiometricDevice: (device: unknown) => api()!.testBiometricDevice(device),
  biometricDeviceCapabilities: (device: unknown) => api()!.biometricDeviceCapabilities(device),
  syncBiometricDevice: (device: unknown) => api()!.syncBiometricDevice(device),
  softDelete: (data: unknown) => api()!.softDelete(data),
  restoreDeleted: (id: string) => api()!.restoreDeleted(id),
  purgeDeleted: (id: string) => api()!.purgeDeleted(id),
  importLegacy: (data: unknown) => api()!.importLegacy(data),
  stats: () => api()!.stats(),
  getCollections: (names?: string[]) => api()!.getCollections(names),
  saveCollections: (collections: Record<string, unknown[]>, options?: Record<string, unknown>) => api()!.saveCollections(collections, options),
  hasMigratedLegacy: () => api()!.hasMigratedLegacy(),
  exportSnapshot: () => api()!.exportSnapshot(),
  importSnapshot: (snapshot: unknown) => api()!.importSnapshot(snapshot),
  batchSaveCollections: (batches: Record<string, unknown[]>[], options?: Record<string, unknown>) => api()!.batchSaveCollections(batches, options),
  enqueueSyncChange: (data: {entityType:string;entityId:string;operation:'UPSERT'|'DELETE';payload?:unknown}) => api()!.enqueueSyncChange(data),
  pendingSync: (args?: Record<string, unknown>) => api()!.pendingSync(args),
  optimize: () => api()!.optimize(),
  hashPassword: (password: string) => window.electronAPI!.hashPassword(password),
  verifyPassword: (password: string, encoded: string) => window.electronAPI!.verifyPassword(password, encoded),
};
